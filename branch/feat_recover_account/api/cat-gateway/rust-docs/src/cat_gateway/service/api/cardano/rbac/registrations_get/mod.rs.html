47/// Get RBAC registration endpoint.
48pub(crate) async fn endpoint(
49    lookup: Option<CatIdOrStake>, token: Option<CatalystRBACTokenV1>,
50) -> AllResponses {
51    let Some(persistent_session) = CassandraSession::get(true) else {
52        let err = anyhow!("Failed to acquire persistent db session");
53        error!("{err:?}");
54        return AllResponses::service_unavailable(&err, RetryAfterOption::Default);
55    };
56    let Some(volatile_session) = CassandraSession::get(false) else {
57        let err = anyhow!("Failed to acquire volatile db session");
58        error!("{err:?}");
59        return AllResponses::service_unavailable(&err, RetryAfterOption::Default);
60    };
61
62    let catalyst_id = match lookup {
63        Some(CatIdOrStake::CatId(v)) => v.into(),
64        Some(CatIdOrStake::Address(address)) => {
65            let address: StakeAddress = match address.try_into() {
66                Ok(a) => a,
67                Err(e) => return AllResponses::handle_error(&e),
68            };
69            // TODO: For now we want to select the latest registration (so query the volatile
70            // database first), but ideally the logic should be more sophisticated. See the issue
71            // for more details: https://github.com/input-output-hk/catalyst-voices/issues/2152
72            match catalyst_id_from_stake(&volatile_session, address.clone()).await {
73                Ok(Some(v)) => v,
74                Err(e) => return AllResponses::handle_error(&e),
75                Ok(None) => {
76                    match catalyst_id_from_stake(&persistent_session, address).await {
77                        Ok(Some(v)) => v,
78                        Err(e) => return AllResponses::handle_error(&e),
79                        Ok(None) => return Responses::NotFound.into(),
80                    }
81                },
82            }
83        },
84        None => {
85            match token {
86                Some(token) => token.catalyst_id().clone(),
87                None => {
88                    return Responses::UnprocessableContent(Json(RbacUnprocessableContent::new(
89                        "Either lookup parameter or token must be provided",
90                    )))
91                    .into()
92                },
93            }
94        },
95    };
96
97    match ChainInfo::new(&persistent_session, &volatile_session, &catalyst_id).await {
98        Ok(Some(info)) => {
99            match RbacRegistrationChain::new(&info) {
100                Ok(c) => Responses::Ok(Json(Box::new(c))).into(),
101                Err(e) => AllResponses::handle_error(&e),
102            }
103        },
104        Ok(None) => Responses::NotFound.into(),
105        Err(e) => AllResponses::handle_error(&e),
106    }
107}

109/// Returns a Catalyst ID for the given stake address.
110async fn catalyst_id_from_stake(
111    session: &CassandraSession, address: StakeAddress,
112) -> anyhow::Result<Option<IdUri>> {
113    let mut result: Vec<_> = Query::execute(session, QueryParams {
114        stake_address: address.into(),
115    })
116    .and_then(|r| r.try_collect().map_err(Into::into))
117    .await
118    .map(|mut res: Vec<_>| {
119        res.sort_by(|a, b| a.slot_no.cmp(&b.slot_no));
120        res
121    })
122    .context("Failed to query Catalyst ID from stake address")?;
123
124    Ok(result.pop().map(|v| v.catalyst_id.into()))
125}
