15/// Get document from the database
16pub(crate) async fn get_document(
17    document_id: &uuid::Uuid, version: Option<&uuid::Uuid>,
18) -> anyhow::Result<CatalystSignedDocument> {
19    // Find the doc in the static templates first
20    if let Some(doc) = get_doc_static_template(document_id) {
21        return Ok(doc);
22    }
23
24    // If doesn't exist in the static templates, try to find it in the database
25    let db_doc = FullSignedDoc::retrieve(document_id, version).await?;
26    db_doc.raw().try_into()
27}

29/// A struct which implements a
30/// `catalyst_signed_doc::providers::CatalystSignedDocumentProvider` trait
31pub(crate) struct DocProvider;
32
33impl catalyst_signed_doc::providers::CatalystSignedDocumentProvider for DocProvider {
34    async fn try_get_doc(
35        &self, doc_ref: &catalyst_signed_doc::DocumentRef,
36    ) -> anyhow::Result<Option<CatalystSignedDocument>> {
37        let id = doc_ref.id.uuid();
38        let ver = doc_ref.ver.uuid();
39        match get_document(&id, Some(&ver)).await {
40            Ok(doc) => Ok(Some(doc)),
41            Err(err) if err.is::<NotFoundError>() => Ok(None),
42            Err(err) => Err(err),
43        }
44    }
45
46    fn future_threshold(&self) -> Option<std::time::Duration> {
47        let signed_doc_cfg = Settings::signed_doc_cfg();
48        Some(signed_doc_cfg.future_threshold())
49    }
50
51    fn past_threshold(&self) -> Option<std::time::Duration> {
52        let signed_doc_cfg = Settings::signed_doc_cfg();
53        Some(signed_doc_cfg.past_threshold())
54    }
55}

66        &self, kid: &catalyst_signed_doc::CatalystId,
67    ) -> anyhow::Result<Option<ed25519_dalek::VerifyingKey>> {
68        Ok(self.0.get(kid).copied())
69    }

71
72impl VerifyingKeyProvider {
73    /// Attempts to construct an instance of `Self` by validating and resolving a list of
74    /// Catalyst Document KIDs against a provided RBAC token.
75    ///
76    /// This method performs the following steps:
77    /// 1. Verifies that only a single KID is provided with a document (as multi-signature
78    ///    is currently unsupported).
79    /// 2. Verifies that **all** provided KIDs match the Catalyst ID from the RBAC token.
80    /// 3. Verifies that each provided KID is actually a signing key.
81    /// 4. Extracts the role index and rotation from each KID.
82    /// 5. Retrieves the latest signing public key and rotation state associated with the
83    ///    role for each KID from the registration chain.
84    /// 6. Verifies that each provided KID uses its latest rotation.
85    /// 7. Collects and returns a vector of tuples containing the KID, and its latest
86    ///    signing key.
87    ///
88    /// # Errors
89    ///
90    /// Returns an `anyhow::Error` if:
91    /// - Any KID's short Catalyst ID does not match the one in the token.
92    /// - Indexed registration queries or chain building fail.
93    /// - The KID's role index and rotation parsing fails.
94    /// - The KID is not a singing key.
95    /// - The latest signing key for a required role cannot be found.
96    /// - The KID is not using the latest rotation.
97    pub(crate) async fn try_from_kids(
98        token: &mut CatalystRBACTokenV1, kids: &[catalyst_signed_doc::CatalystId],
99    ) -> anyhow::Result<Self> {
100        if kids.len() > 1 {
101            anyhow::bail!("Multi-signature document is currently unsupported");
102        }
103
104        if kids
105            .iter()
106            .any(|kid| kid.as_short_id() != token.catalyst_id().as_short_id())
107        {
108            anyhow::bail!("RBAC Token CatID does not match with the document KIDs");
109        }
110
111        let Some(reg_chain) = token.reg_chain().await? else {
112            anyhow::bail!("Failed to retrieve a registration from corresponding Catalyst ID");
113        };
114
115        let result = kids.iter().map(|kid| {
116            if !kid.is_signature_key() {
117                anyhow::bail!("Invalid KID {kid}: KID must be a signing key not an encryption key");
118            }
119
120            let (kid_role_index, kid_rotation) = kid.role_and_rotation();
121            let (latest_pk, rotation) = reg_chain
122                .get_latest_signing_pk_for_role(&kid_role_index)
123                .ok_or_else(|| {
124                anyhow::anyhow!(
125                    "Failed to get last signing key for the proposer role for {kid} Catalyst ID"
126                )
127            })?;
128
129            if rotation != kid_rotation {
130                anyhow::bail!("Invalid KID {kid}: KID's rotation ({kid_rotation}) is not the latest rotation ({rotation})");
131            }
132
133            Ok((kid.clone(), latest_pk))
134        })
135        .collect::<Result<_, _>>()?;
136
137        Ok(Self(result))
138    }
139}
