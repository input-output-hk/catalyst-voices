11/// Data we return from a sync task.
12#[derive(Clone)]
13pub(crate) struct SyncParams {
14    /// What blockchain are we syncing.
15    chain: Network,
16    /// The starting point of this sync.
17    start: Point,
18    /// The ending point of this sync.
19    end: Point,
20    /// The first block we successfully synced.
21    first_indexed_block: Option<Point>,
22    /// Is the starting point immutable? (True = immutable, false = don't know.)
23    first_is_immutable: bool,
24    /// The last block we successfully synced.
25    last_indexed_block: Option<Point>,
26    /// Is the ending point immutable? (True = immutable, false = don't know.)
27    last_is_immutable: bool,
28    /// The number of blocks we successfully synced overall.
29    total_blocks_synced: u64,
30    /// The number of blocks we successfully synced, in the last attempt.
31    last_blocks_synced: u64,
32    /// The number of retries so far on this sync task.
33    retries: u64,
34    /// The number of retries so far on this sync task.
35    backoff_delay: Option<Duration>,
36    /// If the sync completed without error or not.
37    result: Arc<Option<anyhow::Result<()>>>,
38    /// Chain follower roll forward.
39    follower_roll_forward: Option<Point>,
40}
41
42impl Display for SyncParams {
43    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
44        if self.result.is_none() {
45            write!(f, "Sync_Params {{ ")?;
46        } else {
47            write!(f, "Sync_Result {{ ")?;
48        }
49
50        write!(f, "start: {}, end: {}", self.start, self.end)?;
51
52        if let Some(first) = self.first_indexed_block.as_ref() {
53            write!(
54                f,
55                ", first_indexed_block: {first}{}",
56                if self.first_is_immutable { ":I" } else { "" }
57            )?;
58        }
59
60        if let Some(last) = self.last_indexed_block.as_ref() {
61            write!(
62                f,
63                ", last_indexed_block: {last}{}",
64                if self.last_is_immutable { ":I" } else { "" }
65            )?;
66        }
67
68        if self.retries > 0 {
69            write!(f, ", retries: {}", self.retries)?;
70        }
71
72        if self.retries > 0 || self.result.is_some() {
73            write!(f, ", synced_blocks: {}", self.total_blocks_synced)?;
74        }
75
76        if self.result.is_some() {
77            write!(f, ", last_sync: {}", self.last_blocks_synced)?;
78        }
79
80        if let Some(backoff) = self.backoff_delay.as_ref() {
81            write!(f, ", backoff: {}", DurationString::from(*backoff))?;
82        }
83
84        if let Some(result) = self.result.as_ref() {
85            match result {
86                Ok(()) => write!(f, ", Success")?,
87                Err(error) => write!(f, ", {error}")?,
88            };
89        }
90
91        f.write_str(" }")
92    }
93}

95/// The range we generate random backoffs within given a base backoff value.
96const BACKOFF_RANGE_MULTIPLIER: u32 = 3;
97
98impl SyncParams {
99    /// Return `SyncParams::chain` field
100    pub(crate) fn chain(&self) -> Network {
101        self.chain
102    }
103
104    /// Returns `SuncParams::last_indexed_block` field
105    pub(crate) fn last_indexed_block(&self) -> Option<&Point> {
106        self.last_indexed_block.as_ref()
107    }
108
109    /// Returns `SuncParams::follower_roll_forward` field
110    pub(crate) fn follower_roll_forward(&self) -> Option<&Point> {
111        self.follower_roll_forward.as_ref()
112    }
113
114    /// Sets `SuncParams::follower_roll_forward` field
115    pub(crate) fn set_follower_roll_forward(&mut self, roll_forward: Point) {
116        self.follower_roll_forward = Some(roll_forward);
117    }
118
119    /// Returns `SuncParams::result` field
120    pub(crate) fn result(&self) -> Option<&anyhow::Result<()>> {
121        self.result.as_ref().as_ref()
122    }
123
124    /// Create a new `SyncParams`.
125    pub(crate) fn new(chain: Network, start: Point, end: Point) -> Self {
126        Self {
127            chain,
128            start,
129            end,
130            first_indexed_block: None,
131            first_is_immutable: false,
132            last_indexed_block: None,
133            last_is_immutable: false,
134            total_blocks_synced: 0,
135            last_blocks_synced: 0,
136            retries: 0,
137            backoff_delay: None,
138            result: Arc::new(None),
139            follower_roll_forward: None,
140        }
141    }
142
143    /// Convert a result back into parameters for a retry.
144    pub(crate) fn retry(&self) -> Self {
145        let retry_count = self.retries.saturating_add(1);
146
147        let mut backoff = None;
148
149        // If we did sync any blocks last time, first retry is immediate.
150        // Otherwise we backoff progressively more as we do more retries.
151        if self.last_blocks_synced == 0 {
152            // Calculate backoff based on number of retries so far.
153            backoff = match retry_count {
154                1 => Some(Duration::from_secs(1)),     // 1-3 seconds
155                2..5 => Some(Duration::from_secs(10)), // 10-30 seconds
156                _ => Some(Duration::from_secs(30)),    // 30-90 seconds.
157            };
158        }
159
160        let mut retry = self.clone();
161        retry.last_blocks_synced = 0;
162        retry.retries = retry_count;
163        retry.backoff_delay = backoff;
164        retry.result = Arc::new(None);
165        retry.follower_roll_forward = None;
166
167        retry
168    }
169
170    /// Convert Params into the result of the sync.
171    pub(crate) fn done(&self, result: anyhow::Result<()>) -> Self {
172        if result.is_ok() && self.end != Point::TIP {
173            // Update sync status in the Immutable DB.
174            // Can fire and forget, because failure to update DB will simply cause the chunk to be
175            // re-indexed, on recovery.
176            update_sync_status(self.end.slot_or_default(), self.start.slot_or_default());
177        }
178        let mut done = self.clone();
179        done.total_blocks_synced = done
180            .total_blocks_synced
181            .saturating_add(done.last_blocks_synced);
182        done.result = Arc::new(Some(result));
183
184        done
185    }
186
187    /// Apply the upcoming block, update the underlying state
188    pub(crate) fn update_block_state(&mut self, block: &MultiEraBlock) {
189        self.last_is_immutable = block.is_immutable();
190        self.last_indexed_block = Some(block.point());
191
192        if self.first_indexed_block.is_none() {
193            self.first_is_immutable = block.is_immutable();
194            self.first_indexed_block = Some(block.point());
195        }
196
197        self.last_blocks_synced = self.last_blocks_synced.saturating_add(1);
198    }
199
200    /// Get where this sync run actually needs to start from.
201    pub(crate) fn actual_start(&self) -> Point {
202        self.last_indexed_block
203            .as_ref()
204            .unwrap_or(&self.start)
205            .clone()
206    }
207
208    /// Returns `SyncParams.end` field.
209    pub(crate) fn end(&self) -> Point {
210        self.end.clone()
211    }
212
213    /// Do the backoff delay processing.
214    ///
215    /// The actual delay is a random time from the Delay itself to
216    /// `BACKOFF_RANGE_MULTIPLIER` times the delay. This is to prevent hammering the
217    /// service at regular intervals.
218    pub(crate) async fn backoff(&self) {
219        if let Some(backoff) = self.backoff_delay {
220            let mut rng = rand::rngs::StdRng::from_entropy();
221            let actual_backoff =
222                rng.gen_range(backoff..backoff.saturating_mul(BACKOFF_RANGE_MULTIPLIER));
223
224            tokio::time::sleep(actual_backoff).await;
225        }
226    }
