22/// Index RBAC 509 Registration Query Parameters
23#[derive(Debug)]
24pub(crate) struct Rbac509InsertQuery {
25    /// RBAC Registration Data captured during indexing.
26    pub(crate) registrations: Vec<insert_rbac509::Params>,
27    /// An invalid RBAC registration data.
28    pub(crate) invalid: Vec<insert_rbac509_invalid::Params>,
29}
30
31impl Rbac509InsertQuery {
32    /// Creates new data set for RBAC 509 Registrations Insert Query Batch.
33    pub(crate) fn new() -> Self {
34        Rbac509InsertQuery {
35            registrations: Vec::new(),
36            invalid: Vec::new(),
37        }
38    }
39
40    /// Prepare Batch of Insert RBAC 509 Registration Data Queries
41    pub(crate) async fn prepare_batch(
42        session: &Arc<Session>, cfg: &EnvVars,
43    ) -> anyhow::Result<(SizedBatch, SizedBatch)> {
44        Ok((
45            insert_rbac509::Params::prepare_batch(session, cfg).await?,
46            insert_rbac509_invalid::Params::prepare_batch(session, cfg).await?,
47        ))
48    }
49
50    /// Index the RBAC 509 registrations in a transaction.
51    pub(crate) fn index(
52        &mut self, txn_hash: TransactionId, index: TxnIndex, block: &MultiEraBlock,
53    ) {
54        let slot = block.slot();
55        let cip509 = match Cip509::new(block, index, &[]) {
56            Ok(Some(v)) => v,
57            Ok(None) => {
58                // Nothing to index.
59                return;
60            },
61            Err(e) => {
62                // This registration is either completely corrupted or someone else is using "our"
63                // label (`MetadatumLabel::CIP509_RBAC`). We don't want to index it even as
64                // incorrect.
65                debug!(
66                    slot = ?slot,
67                    index = ?index,
68                    "Invalid RBAC Registration Metadata in transaction: {e:?}"
69                );
70                return;
71            },
72        };
73
74        // This should never happen, but let's check anyway.
75        if slot != cip509.origin().point().slot_or_default() {
76            error!(
77                "Cip509 slot mismatch: expected {slot:?}, got {:?}",
78                cip509.origin().point().slot_or_default()
79            );
80        }
81        if txn_hash != cip509.txn_hash() {
82            error!(
83                "Cip509 txn hash mismatch: expected {txn_hash}, got {}",
84                cip509.txn_hash()
85            );
86        }
87
88        let previous_transaction = cip509.previous_transaction();
89        match RBAC_CACHE.add(cip509, block.is_immutable()) {
90            Ok(RbacCacheAddSuccess { catalyst_id }) => {
91                self.registrations.push(insert_rbac509::Params::new(
92                    catalyst_id.clone(),
93                    txn_hash,
94                    slot,
95                    index,
96                    previous_transaction,
97                ));
98            },
99            Err(RbacCacheAddError::InvalidRegistration {
100                catalyst_id,
101                purpose,
102                report,
103            }) => {
104                self.invalid.push(insert_rbac509_invalid::Params::new(
105                    catalyst_id,
106                    txn_hash,
107                    slot,
108                    index,
109                    purpose,
110                    previous_transaction,
111                    &report,
112                ));
113            },
114            Err(RbacCacheAddError::UnknownCatalystId) => {
115                debug!("Unable to determine Catalyst id for registration: slot = {slot:?}, index = {index:?}, txn_hash = {txn_hash:?}");
116            },
117        }
118    }
119
120    /// Execute the RBAC 509 Registration Indexing Queries.
121    ///
122    /// Consumes the `self` and returns a vector of futures.
123    pub(crate) fn execute(self, session: &Arc<CassandraSession>) -> FallibleQueryTasks {
124        let mut query_handles: FallibleQueryTasks = Vec::new();
125
126        if !self.registrations.is_empty() {
127            let inner_session = session.clone();
128            query_handles.push(tokio::spawn(async move {
129                inner_session
130                    .execute_batch(PreparedQuery::Rbac509InsertQuery, self.registrations)
131                    .await
132            }));
133        }
134
135        if !self.invalid.is_empty() {
136            let inner_session = session.clone();
137            query_handles.push(tokio::spawn(async move {
138                inner_session
139                    .execute_batch(PreparedQuery::Rbac509InvalidInsertQuery, self.invalid)
140                    .await
141            }));
142        }
143
144        query_handles
145    }
146}

148#[cfg(test)]
149mod tests {
150    use super::*;
151    use crate::db::index::tests::test_utils;
152
153    #[tokio::test]
154    async fn index() {
155        let block = test_utils::block_3();
156        let mut query = Rbac509InsertQuery::new();
157        let txn_hash = "1bf8eb4da8fe5910cc890025deb9740ba5fa4fd2ac418ccbebfd6a09ed10e88b"
158            .parse()
159            .unwrap();
160        query.index(txn_hash, 0.into(), &block);
161        assert!(query.invalid.is_empty());
162        assert_eq!(1, query.registrations.len());
163    }
164
165    // The invalid vec is empty in this test because `RBAC_CACHE.add()` returns
166    // `UnknownCatalystId` for that registration.
167    #[tokio::test]
168    async fn index_invalid() {
169        let block = test_utils::block_4();
170        let mut query = Rbac509InsertQuery::new();
171        let txn_hash = "337d35026efaa48b5ee092d38419e102add1b535364799eb8adec8ac6d573b79"
172            .parse()
173            .unwrap();
174        query.index(txn_hash, 0.into(), &block);
175        assert!(query.registrations.is_empty());
176        assert!(query.invalid.is_empty());
177    }
