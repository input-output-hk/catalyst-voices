30/// How long we wait between checks for connection to the indexing DB to be ready.
31const INDEXING_DB_READY_WAIT_INTERVAL: Duration = Duration::from_secs(1);

33/// The synchronisation task, and its state.
34/// There should ONLY ever be one of these at any time.
35pub(crate) struct SyncTask {
36    /// Chain follower configuration.
37    cfg: chain_follower::EnvVars,
38
39    /// The current running sync tasks.
40    sync_tasks: FuturesUnordered<tokio::task::JoinHandle<SyncParams>>,
41
42    /// Start for the next block we would sync.
43    start_slot: Slot,
44
45    /// The immutable tip slot.
46    immutable_tip_slot: Slot,
47
48    /// The live tip slot.
49    live_tip_slot: Slot,
50
51    /// Current Sync Status.
52    sync_status: Vec<SyncStatus>,
53
54    /// Event sender during the process of sync tasks.
55    event_channel: (
56        broadcast::Sender<ChainIndexerEvent>,
57        broadcast::Receiver<ChainIndexerEvent>,
58    ),
59}
60
61impl SyncTask {
62    /// Create a new `SyncTask`.
63    pub(crate) fn new(cfg: chain_follower::EnvVars) -> SyncTask {
64        Self {
65            cfg,
66            sync_tasks: FuturesUnordered::new(),
67            start_slot: 0.into(),
68            immutable_tip_slot: 0.into(),
69            live_tip_slot: 0.into(),
70            sync_status: Vec::new(),
71            event_channel: broadcast::channel(10),
72        }
73    }
74
75    /// Primary Chain Follower task.
76    ///
77    /// This continuously runs in the background, and never terminates.
78    pub(crate) async fn run(&mut self) {
79        // We can't sync until the local chain data is synced.
80        // This call will wait until we sync.
81        let tips = ChainFollower::get_tips(self.cfg.chain).await;
82        self.immutable_tip_slot = tips.0.slot_or_default();
83        self.live_tip_slot = tips.1.slot_or_default();
84        info!(chain=%self.cfg.chain, immutable_tip=?self.immutable_tip_slot, live_tip=?self.live_tip_slot, "Blockchain ready to sync from.");
85
86        self.dispatch_event(ChainIndexerEvent::ImmutableTipSlotChanged {
87            slot: self.immutable_tip_slot,
88        });
89        self.dispatch_event(ChainIndexerEvent::LiveTipSlotChanged {
90            slot: self.live_tip_slot,
91        });
92
93        // Wait for indexing DB to be ready before continuing.
94        // We do this after the above, because other nodes may have finished already, and we don't
95        // want to wait do any work they already completed while we were fetching the blockchain.
96        drop(CassandraSession::wait_until_ready(INDEXING_DB_READY_WAIT_INTERVAL, true).await);
97        info!(chain=%self.cfg.chain, "Indexing DB is ready - Getting recovery state");
98        self.sync_status = get_sync_status().await;
99        debug!(chain=%self.cfg.chain, "Sync Status: {:?}", self.sync_status);
100
101        if !self.try_start_immutable_followers() {
102            // immutable chain already synced, start live chain
103            if !self.try_start_live_chain_follower() {
104                error!(chain=%self.cfg.chain, "BUG: Cannot start any follower task");
105            }
106        }
107
108        self.dispatch_event(ChainIndexerEvent::SyncStarted);
109
110        while let Some(completed) = self.sync_tasks.next().await {
111            self.process_completed_task(completed.map_err(Into::into));
112
113            // IF there is no chain followers left in sync_tasks, then all
114            // immutable followers have finished.
115            // When this happens we need to purge the live index of any records that exist
116            // before the current immutable tip.
117            // Note: to prevent a data race when multiple nodes are syncing, we probably
118            // want to put a gap in this, so that there are X slots of overlap
119            // between the live chain and immutable chain.  This gap should be
120            // a parameter.
121            if self.sync_tasks.is_empty() {
122                info!(chain=%self.cfg.chain, immutable_tip=?self.immutable_tip_slot, "Immutable State Indexing Finished");
123                self.dispatch_event(ChainIndexerEvent::SyncCompleted);
124
125                // Purge data up to this slot
126                // Slots arithmetic has saturating semantic, so this is ok.
127                #[allow(clippy::arithmetic_side_effects)]
128                let purge_to_slot =
129                    self.immutable_tip_slot - Settings::purge_backward_slot_buffer();
130                let purge_condition = PurgeCondition::PurgeBackwards(purge_to_slot);
131                if let Err(error) = roll_forward::purge_live_index(purge_condition).await {
132                    error!(chain=%self.cfg.chain, error=%error, "BUG: Purging volatile data task failed.");
133                } else {
134                    self.dispatch_event(ChainIndexerEvent::BackwardDataPurged);
135                }
136
137                if !self.try_start_live_chain_follower() {
138                    error!(chain=%self.cfg.chain,
139                        "BUG: Cannot start live chain follower task, after completed immutable sync."
140                    );
141                }
142            }
143        }
144
145        error!(chain=%self.cfg.chain,"BUG: Sync tasks have all stopped.  This is an unexpected error!");
146    }

148    /// Try to start immutable followers.
149    /// Returns `true` if at least one task was started, `false` otherwise.
150    fn try_start_immutable_followers(&mut self) -> bool {
151        let mut started = false;
152        // Nothing to do if the start_slot is not less than the end of the immutable chain.
153        if self.start_slot < self.immutable_tip_slot {
154            // We will start at most the number of configured sync tasks.
155            while self.sync_tasks.len() < usize::from(self.cfg.sync_tasks) {
156                let end_slot = self.immutable_tip_slot.min(
157                    (u64::from(self.start_slot).saturating_add(self.cfg.sync_chunk_max_slots))
158                        .into(),
159                );
160
161                if let Some((first_point, last_point)) =
162                    self.get_syncable_range(self.start_slot, end_slot)
163                {
164                    self.sync_tasks.push(sync_subchain(
165                        SyncParams::new_immutable(self.cfg.chain, first_point, last_point.clone()),
166                        self.event_channel.0.clone(),
167                    ));
168                    started = true;
169                    self.dispatch_event(ChainIndexerEvent::SyncTasksChanged {
170                        current_sync_tasks: self
171                            .sync_tasks
172                            .len()
173                            .try_into()
174                            .inspect(|_| {
175                                error!(
176                                    sync_tasks = self.sync_tasks.len(),
177                                    max_sync_tasks = self.cfg.sync_tasks,
178                                    "Follower sync tasks amount overflow."
179                                );
180                            })
181                            .unwrap_or(self.cfg.sync_tasks),
182                    });
183                }
184
185                // The one slot overlap is deliberate, it doesn't hurt anything and prevents all off
186                // by one problems that may occur otherwise.
187                self.start_slot = end_slot;
188
189                if end_slot == self.immutable_tip_slot {
190                    break;
191                }
192            }
193        }
194        started
195    }
196
197    /// Try to start only one live chain follower.
198    /// Returns `true` if task was started, `false` otherwise.
199    fn try_start_live_chain_follower(&mut self) -> bool {
200        if self.sync_tasks.len() < usize::from(self.cfg.sync_tasks) {
201            self.sync_tasks.push(sync_subchain(
202                SyncParams::new_live(self.cfg.chain, Point::fuzzy(self.immutable_tip_slot)),
203                self.event_channel.0.clone(),
204            ));
205
206            self.dispatch_event(ChainIndexerEvent::SyncTasksChanged {
207                current_sync_tasks: self
208                    .sync_tasks
209                    .len()
210                    .try_into()
211                    .inspect(|_| {
212                        error!(
213                            sync_tasks = self.sync_tasks.len(),
214                            max_sync_tasks = self.cfg.sync_tasks,
215                            "Follower sync tasks amount overflow."
216                        );
217                    })
218                    .unwrap_or(self.cfg.sync_tasks),
219            });
220
221            return true;
222        }
223        false
224    }

226    /// Process completed sync tasks
227    /// If they fail and have not completed, reschedule them.
228    // If an immutable sync task ends OK, and we still have immutable data to sync then
229    // start a new task.
230    // They will return from this iterator in the order they complete.
231    // This iterator actually never ends, because the live sync task is always restarted.
232    fn process_completed_task(&mut self, completed: anyhow::Result<SyncParams>) {
233        let finished = match completed {
234            Ok(finished) => finished,
235            Err(error) => {
236                error!(chain=%self.cfg.chain, error=%error,
237                    "BUG: Sync task failed. Can not restart it, not enough information. Sync is probably failed at this point."
238                );
239                return;
240            },
241        };
242        // If there is a roll forward we want to start indexing immutable followers to catch up
243        // with the immutable state
244        if let Some(roll_forward_point) = finished.follower_roll_forward() {
245            // Advance the known immutable tip, and try and start followers to reach
246            // it.
247            self.immutable_tip_slot = roll_forward_point.slot_or_default();
248
249            self.dispatch_event(ChainIndexerEvent::ImmutableTipSlotChanged {
250                slot: self.immutable_tip_slot,
251            });
252
253            self.try_start_immutable_followers();
254        }
255
256        // Sync task finished.  Check if it completed OK or had an error.
257        // If it failed, we need to reschedule it.
258        if let Some(error) = finished.error() {
259            error!(chain=%self.cfg.chain, report=%finished, error=%error,
260                "An Follower sync task failed, restarting it."
261            );
262            // Restart the Follower Chain sync task again from where it left
263            // off.
264            self.sync_tasks.push(sync_subchain(
265                finished.retry(),
266                self.event_channel.0.clone(),
267            ));
268        } else {
269            info!(chain=%self.cfg.chain, report=%finished,
270                "The Follower sync task completed successfully."
271            );
272
273            if !finished.is_live() {
274                // Update sync status in the Immutable DB.
275                // Can fire and forget, because failure to update DB will simply cause the chunk to
276                // be re-indexed, on recovery.
277                update_sync_status(
278                    finished.end().slot_or_default(),
279                    finished.start().slot_or_default(),
280                );
281            }
282
283            finished.last_indexed_block().inspect(|block| {
284                self.dispatch_event(ChainIndexerEvent::IndexedSlotProgressed {
285                    slot: block.slot_or_default(),
286                });
287            });
288            self.dispatch_event(ChainIndexerEvent::SyncTasksChanged {
289                current_sync_tasks: self
290                    .sync_tasks
291                    .len()
292                    .try_into()
293                    .inspect(|_| {
294                        error!(
295                            sync_tasks = self.sync_tasks.len(),
296                            max_sync_tasks = self.cfg.sync_tasks,
297                            "Follower sync tasks amount overflow."
298                        );
299                    })
300                    .unwrap_or(self.cfg.sync_tasks),
301            });
302
303            // If we need more immutable chain followers to sync the block
304            // chain, we can now start them.
305            self.try_start_immutable_followers();
306        }
307    }
308
309    /// Check if the requested range has already been indexed.
310    /// If it hasn't just return the slots as points.
311    /// If it has, return a subset that hasn't been indexed if any, or None if its been
312    /// completely indexed already.
313    fn get_syncable_range(&self, start: Slot, end: Slot) -> Option<(Point, Point)> {
314        for sync_block in &self.sync_status {
315            // Check if we start within a previously synchronized block.
316            if start >= sync_block.start_slot && start <= sync_block.end_slot {
317                // Check if we are fully contained by the sync block, if so, nothing to sync.
318                if end <= sync_block.end_slot {
319                    return None;
320                }
321
322                // In theory, we could extend into another sync block, but because we could extend
323                // into an unbounded number of sync blocks, we would need to bust
324                // this range into an unbounded number of sub chunks.
325                // It is not a problem to sync the same data mutiple times, so for simplicity we do
326                // not account for this, if the requested range goes beyond the sync
327                // block it starts within we assume that the rest is not synced.
328                return Some((Point::fuzzy(sync_block.end_slot), Point::fuzzy(end)));
329            }
330        }
331
332        let start_slot = if start == 0.into() {
333            Point::ORIGIN
334        } else {
335            Point::fuzzy(start)
336        };
337
338        Some((start_slot, Point::fuzzy(end)))
339    }

342impl EventTarget<ChainIndexerEvent> for SyncTask {
343    fn add_event_listener(&mut self, listener: EventListenerFn<ChainIndexerEvent>) {
344        let mut rx = self.event_channel.0.subscribe();
345        tokio::spawn(async move {
346            while let Ok(event) = rx.recv().await {
347                (listener)(&event);
348            }
349        });
350    }
351
352    fn dispatch_event(&self, message: ChainIndexerEvent) {
353        let _ = self.event_channel.0.send(message);
354    }
355}

357/// Sync a portion of the blockchain.
358/// Set end to `Point::TIP` to sync the tip continuously.
359fn sync_subchain(
360    mut params: SyncParams, event_sender: broadcast::Sender<ChainIndexerEvent>,
361) -> tokio::task::JoinHandle<SyncParams> {
362    tokio::spawn(async move {
363        info!(chain = %params.chain(), params=%params, "Indexing Blockchain");
364
365        // Backoff hitting the database if we need to.
366        params.backoff().await;
367
368        // Wait for indexing DB to be ready before continuing.
369        drop(CassandraSession::wait_until_ready(INDEXING_DB_READY_WAIT_INTERVAL, true).await);
370        info!(chain=%params.chain(), params=%params,"Indexing DB is ready");
371
372        let mut follower =
373            ChainFollower::new(*params.chain(), params.actual_start(), params.end().clone()).await;
374        while let Some(chain_update) = follower.next().await {
375            match chain_update.kind {
376                cardano_chain_follower::Kind::ImmutableBlockRollForward => {
377                    // What we need to do here is tell the primary follower to start a new sync
378                    // for the new immutable data, and then purge the volatile database of the
379                    // old data (after the immutable data has synced).
380                    info!(chain=%params.chain(), "Immutable chain rolled forward, point {}.", chain_update.block_data().point());
381                    // Signal the point the immutable chain rolled forward to.
382                    params.set_follower_roll_forward(chain_update.block_data().point());
383                    // If this is live chain immediately stops to later run immutable sync tasks
384                    if params.is_live() {
385                        return params.done(None);
386                    }
387                },
388                cardano_chain_follower::Kind::Block => {
389                    let block = chain_update.block_data();
390
391                    if let Err(error) = index_block(block).await {
392                        let error =
393                            error.context(format!("Failed to index block {}", block.point()));
394                        return params.done(Some(error));
395                    }
396
397                    params.update_block_state(block);
398                },
399                cardano_chain_follower::Kind::Rollback => {
400                    if params.is_live() {
401                        // Rollback occurs, need to purge forward
402                        let rollback_slot = chain_update.block_data().slot();
403
404                        let purge_condition = PurgeCondition::PurgeForwards(rollback_slot);
405                        if let Err(error) = roll_forward::purge_live_index(purge_condition).await {
406                            let error = error.context(
407                                "Chain follower rollback, purging volatile data task failed.",
408                            );
409                            error!(chain=%params.chain(), error=%error);
410                            return params.done(Some(error));
411                        }
412                        // How many slots are purged
413                        #[allow(clippy::arithmetic_side_effects)]
414                        let purge_slots = params
415                            .last_indexed_block()
416                            // Slots arithmetic has saturating semantic, so this is ok
417                            .map_or(0.into(), |l| l.slot_or_default() - rollback_slot);
418
419                        let _ = event_sender.send(ChainIndexerEvent::ForwardDataPurged {
420                            purge_slots: purge_slots.into(),
421                        });
422
423                        // Purge success, now index the current block
424                        let block = chain_update.block_data();
425                        if let Err(error) = index_block(block).await {
426                            let error = error.context(format!(
427                                "Failed to index block after rollback {}",
428                                block.point()
429                            ));
430                            return params.done(Some(error));
431                        }
432
433                        params.update_block_state(block);
434                    }
435                },
436            }
437        }
438
439        let result = params.done(None);
440        info!(chain = %result.chain(), result=%result, "Indexing Blockchain Completed: OK");
441        result
442    })
