1//! Implementation of the GET `/cardano/cip36` endpoint
2
3use poem::http::HeaderMap;

6use self::cardano::query::stake_or_voter::StakeAddressOrPublicKey;
7use super::{
8    cardano::{self},

10    response, SlotNo,
11};
12use crate::{
13    db::index::session::CassandraSession,
14    service::{

20/// Process the endpoint operation
21pub(crate) async fn cip36_registrations(
22    lookup: Option<cardano::query::stake_or_voter::StakeOrVoter>, asat: Option<SlotNo>,
23    _page: common::types::generic::query::pagination::Page,
24    _limit: common::types::generic::query::pagination::Limit, _headers: &HeaderMap,
25) -> AllRegistration {
26    let Some(session) = CassandraSession::get(true) else {
27        error!("Failed to acquire db session");
28        return AllRegistration::service_unavailable(
29            &anyhow::anyhow!("Failed to acquire db session"),
30            RetryAfterOption::Default,
31        );
32    };
33
34    if let Some(stake_or_voter) = lookup {
35        match StakeAddressOrPublicKey::from(stake_or_voter) {
36            StakeAddressOrPublicKey::Address(cip19_stake_address) => {
37                // Typically, a stake address will start with 'stake1',
38                // We need to convert this to a stake hash as per our data model to then find the,
39                // Full Stake Public Key (32 byte Ed25519 Public key, not hashed).
40                // We then get the latest registration or from a specific time as optionally
41                // specified in the query parameter. This can be represented as either
42                // the blockchains slot number or a unix timestamp.
43                let address = match cip19_stake_address.try_into() {
44                    Ok(a) => a,
45                    Err(err) => {
46                        return AllRegistration::handle_error(&anyhow::anyhow!(
47                            "Given stake pub key is corrupt {:?}",
48                            err
49                        ));
50                    },
51                };
52
53                return get_registration_given_stake_key_hash(address, session, asat).await;
54            },
55            StakeAddressOrPublicKey::PublicKey(ed25519_hex_encoded_public_key) => {
56                // As above...
57                // Except using a voting key.
58                return get_registration_given_vote_key(
59                    ed25519_hex_encoded_public_key,
60                    session,
61                    asat,
62                )
63                .await;
64            },
65            StakeAddressOrPublicKey::All =>
66            // As above...
67            // Snapshot replacement, returns all registrations or returns a
68            // subset of registrations if constrained by a given time.
69            {
70                return snapshot(session, asat).await
71            },
72        };
73    };
74
75    // If _for is not defined, use the stake addresses defined for Role0 in the _auth
76    // parameter. _auth not yet implemented, so put placeholder for that, and return not
77    // found until _auth is implemented.
78
79    response::Cip36Registration::NotFound.into()
80}
