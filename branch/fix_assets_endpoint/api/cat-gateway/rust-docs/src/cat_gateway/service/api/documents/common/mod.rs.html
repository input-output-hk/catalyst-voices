12/// Get document from the database
13pub(crate) async fn get_document(
14    document_id: &uuid::Uuid, version: Option<&uuid::Uuid>,
15) -> anyhow::Result<CatalystSignedDocument> {
16    // Find the doc in the static templates first
17    if let Some(doc) = get_doc_static_template(document_id) {
18        return Ok(doc);
19    }
20
21    // If doesn't exist in the static templates, try to find it in the database
22    let db_doc = FullSignedDoc::retrieve(document_id, version).await?;
23    db_doc.raw().try_into()
24}

26/// A struct which implements a
27/// `catalyst_signed_doc::providers::CatalystSignedDocumentProvider` trait
28pub(crate) struct DocProvider;
29
30impl catalyst_signed_doc::providers::CatalystSignedDocumentProvider for DocProvider {
31    async fn try_get_doc(
32        &self, doc_ref: &catalyst_signed_doc::DocumentRef,
33    ) -> anyhow::Result<Option<CatalystSignedDocument>> {
34        let id = doc_ref.id.uuid();
35        let ver = doc_ref.ver.map(|uuid| uuid.uuid());
36        match get_document(&id, ver.as_ref()).await {
37            Ok(doc) => Ok(Some(doc)),
38            Err(err) if err.is::<NotFoundError>() => Ok(None),
39            Err(err) => Err(err),
40        }
41    }
42}

44/// A struct which implements a
45/// `catalyst_signed_doc::providers::CatalystSignedDocumentProvider` trait
46pub(crate) struct VerifyingKeyProvider(
47    Vec<(catalyst_signed_doc::IdUri, ed25519_dalek::VerifyingKey)>,
48);
49
50impl catalyst_signed_doc::providers::VerifyingKeyProvider for VerifyingKeyProvider {
51    async fn try_get_key(
52        &self, kid: &catalyst_signed_doc::IdUri,
53    ) -> anyhow::Result<Option<ed25519_dalek::VerifyingKey>> {
54        Ok(self.0.iter().find(|item| &(item.0) == kid).map(|v| v.1))
55    }
56}
57
58impl VerifyingKeyProvider {
59    /// Prepares a list of registrations related to `catid` from the token.
60    pub(crate) async fn try_from_token(token: CatalystRBACTokenV1) -> anyhow::Result<Self> {
61        let cat_id = token.catalyst_id();
62
63        let registrations = scheme::indexed_registrations(cat_id).await?;
64
65        if registrations.is_empty() {
66            return Err(anyhow::anyhow!(
67                "Unable to find registrations for {cat_id} Catalyst ID"
68            ));
69        }
70
71        let pk = scheme::last_signing_key(token.network(), &registrations)
72            .await
73            .map_err(|e| {
74                anyhow::anyhow!("Unable to get last signing key for {cat_id} Catalyst ID: {e:?}")
75            })?;
76
77        Ok(Self(Vec::from([(cat_id.clone().as_uri(), pk)])))
78    }
