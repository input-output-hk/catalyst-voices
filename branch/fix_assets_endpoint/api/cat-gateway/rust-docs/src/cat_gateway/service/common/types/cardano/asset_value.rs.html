12/// Title.
13const TITLE: &str = "Cardano native Asset Value";
14/// Description.
15const DESCRIPTION: &str = "This is a non-zero signed integer.";
16/// Example.
17const EXAMPLE: i128 = 1_234_567;
18/// Minimum.
19/// From: <https://github.com/IntersectMBO/cardano-ledger/blob/78b32d585fd4a0340fb2b184959fb0d46f32c8d2/eras/conway/impl/cddl-files/conway.cddl#L568>
20/// This is NOT `i128::MIN`.
21const MINIMUM: i128 = -9_223_372_036_854_775_808;
22/// Maximum.
23/// From: <https://github.com/IntersectMBO/cardano-ledger/blob/78b32d585fd4a0340fb2b184959fb0d46f32c8d2/eras/conway/impl/cddl-files/conway.cddl#L569>
24/// This is NOT `i128::MAX`.
25const MAXIMUM: i128 = 9_223_372_036_854_775_808;
26
27/// Schema.
28#[allow(clippy::cast_precision_loss)]
29static SCHEMA: LazyLock<MetaSchema> = LazyLock::new(|| {
30    MetaSchema {
31        title: Some(TITLE.to_owned()),
32        description: Some(DESCRIPTION),
33        example: Some(EXAMPLE.into()),
34        maximum: Some(MAXIMUM as f64),
35        minimum: Some(MINIMUM as f64),
36        ..poem_openapi::registry::MetaSchema::ANY
37    }
38});
39
40/// Value of a Cardano Native Asset (may not be zero)
41#[derive(Debug, Clone, Eq, PartialEq, Hash)]
42pub(crate) struct AssetValue(i128);
43
44/// Is the `AssetValue` valid?
45fn is_valid(value: i128) -> bool {
46    value != 0 && (MINIMUM..=MAXIMUM).contains(&value)
47}

49impl Type for AssetValue {
50    type RawElementValueType = Self;
51    type RawValueType = Self;
52
53    const IS_REQUIRED: bool = true;

60        let schema_ref =
61            MetaSchemaRef::Inline(Box::new(MetaSchema::new_with_format("integer", "i128")));

63    }

67    }

73    }
74}

76impl ParseFromJSON for AssetValue {
77    fn parse_from_json(value: Option<Value>) -> ParseResult<Self> {
78        let value = value.unwrap_or_default();
79        if let Value::Number(value) = value {
80            let value = value.as_i128().unwrap_or_default();
81            if !is_valid(value) {
82                return Err("invalid AssetValue".into());
83            }
84            Ok(Self(value))
85        } else {
86            Err(ParseError::expected_type(value))
87        }
88    }
89}
90
91impl ToJSON for AssetValue {
92    fn to_json(&self) -> Option<Value> {
93        Some(self.0.into())
94    }
95}

97impl TryFrom<num_bigint::BigInt> for AssetValue {
98    type Error = anyhow::Error;
99
100    fn try_from(value: num_bigint::BigInt) -> Result<Self, Self::Error> {
101        let value: i128 = value.try_into()?;
102        if !is_valid(value) {
103            bail!("Invalid Asset Value: {value}");
104        }
105        Ok(Self(value))
106    }
107}
