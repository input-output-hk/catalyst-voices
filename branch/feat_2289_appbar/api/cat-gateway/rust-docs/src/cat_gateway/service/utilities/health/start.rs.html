1//! Utilities used for system `Started` functionality.
2
3use std::sync::atomic::{
4    AtomicBool,
5    Ordering::{Acquire, Release},
6};
7
8/// Flag to determine if the service has started
9static STARTED: AtomicBool = AtomicBool::new(false);
10
11/// Flag to determine if the Indexing DB has started.
12static LIVE_INDEX_DB: AtomicBool = AtomicBool::new(false);
13
14/// Flag to determine if the Event DB has started.
15static LIVE_EVENT_DB: AtomicBool = AtomicBool::new(false);
16
17/// Flag to determine if the chain follower has synchronized with the Tip for the first
18/// time.
19static INITIAL_FOLLOWER_TIP_REACHED: AtomicBool = AtomicBool::new(false);
20
21/// Returns whether the service has started or not.
22pub(crate) fn service_has_started() -> bool {
23    STARTED.load(Acquire)
24}
25
26/// Set the `STARTED` flag to `true`
27pub(crate) fn set_to_started() {
28    STARTED.store(true, Release);
29}
30
31/// Returns whether the service has started or not.
32pub(crate) fn condition_for_started() -> bool {
33    event_db_is_live() && index_db_is_live() && follower_has_first_reached_tip()
34}
35
36/// Returns whether the Event DB is live or not.
37///
38/// Gets the stored value from `LIVE_EVENT_DB` flag.
39pub(crate) fn event_db_is_live() -> bool {
40    LIVE_EVENT_DB.load(Acquire)
41}
42
43/// Set the `LIVE_EVENT_DB` flag.
44pub(crate) fn set_event_db_liveness(flag: bool) {
45    LIVE_EVENT_DB.store(flag, Release);
46}
47
48/// Returns whether the Index DB is live or not.
49///
50/// Gets the stored value from `LIVE_INDEX_DB` flag.
51pub(crate) fn index_db_is_live() -> bool {
52    LIVE_INDEX_DB.load(Acquire)
53}
54
55/// Set the `LIVE_INDEX_DB` flag.
56pub(crate) fn set_index_db_liveness(flag: bool) {
57    LIVE_INDEX_DB.store(flag, Release);
58}
59
60/// Returns whether the Chain Follower has reached Tip or not.
61///
62/// Gets the stored value from `INITIAL_FOLLOWER_TIP_REACHED` flag.
63pub(crate) fn follower_has_first_reached_tip() -> bool {
64    INITIAL_FOLLOWER_TIP_REACHED.load(Acquire)
65}
66
67/// Set the `INITIAL_FOLLOWER_TIP_REACHED` as `true`.
68///
69/// This value can not be set to `false` afterwards.
70pub(crate) fn set_follower_first_reached_tip() {
71    INITIAL_FOLLOWER_TIP_REACHED.store(true, Release);
72}
