1//! API Key authorization scheme is used ONLY by internal endpoints.
2//!
3//! Its purpose is to prevent their use externally, if they were accidentally exposed.
4//!
5//! It is NOT to be used on any endpoint intended to be publicly facing.
6
7use poem::Request;
8use poem_openapi::{auth::ApiKey, SecurityScheme};
9
10use crate::settings::Settings;
11
12/// The header name that holds the API Key
13pub(crate) const API_KEY_HEADER: &str = "X-API-Key";
14
15/// `ApiKey` authorization for Internal Endpoints
16#[derive(SecurityScheme)]
17#[oai(
18    ty = "api_key",
19    key_name = "X-API-Key", // MUST match the `API_KEY_HEADER` constant.
20    key_in = "header",
21    checker = "api_checker"
22)]
23#[allow(dead_code)]
24pub(crate) struct InternalApiKeyAuthorization(String);
25
26/// Check the provided API Key matches the API Key defined by for the deployment.
27#[allow(clippy::unused_async)]
28async fn api_checker(_req: &Request, api_key: ApiKey) -> Option<String> {
29    if Settings::check_internal_api_key(&api_key.key) {
30        Some(api_key.key)
31    } else {
32        None
33    }
34}
