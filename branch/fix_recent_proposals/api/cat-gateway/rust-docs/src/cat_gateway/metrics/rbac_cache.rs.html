11/// This is to prevent the init function from accidentally being called multiple times.
12static IS_INITIALIZED: AtomicBool = AtomicBool::new(false);
13
14/// Starts a background thread to periodically update Chain Follower metrics.
15///
16/// This function spawns a thread that updates the Chain Follower metrics
17/// at regular intervals defined by `METRICS_FOLLOWER_INTERVAL`.
18pub(crate) fn init_metrics_reporter() {
19    if IS_INITIALIZED.swap(true, Ordering::SeqCst) {
20        return;
21    }
22
23    let api_host_names = Settings::api_host_names().join(",");
24    let service_id = Settings::service_id();
25    let network = Settings::cardano_network().to_string();
26
27    thread::spawn(move || loop {
28        let rbac_entries = RBAC_CACHE.rbac_entries();
29        reporter::CACHING_RBAC_ENTRIES
30            .with_label_values(&[&api_host_names, service_id, &network])
31            .set(i64::try_from(rbac_entries).unwrap_or(-1));
32
33        thread::sleep(Duration::from_secs(1));
34    });
35}

37/// All the related RBAC Registration Chain Caching reporting metrics to the Prometheus
38/// service are inside this module.
39pub(crate) mod reporter {
40    use std::sync::LazyLock;
41
42    use prometheus::{
43        register_counter_vec, register_histogram_vec, register_int_gauge_vec, CounterVec,
44        HistogramVec, IntGaugeVec,
45    };
46
47    /// Labels for the metrics.
48    const METRIC_LABELS: [&str; 3] = ["api_host_names", "service_id", "network"];
49
50    /// Total count of cache hits.
51    pub(crate) static CACHE_HIT: LazyLock<CounterVec> = LazyLock::new(|| {
52        register_counter_vec!("cache_hit", "Total count of cache hits", &METRIC_LABELS).unwrap()
53    });
54
55    /// Total count of cache misses.
56    pub(crate) static CACHE_MISS: LazyLock<CounterVec> = LazyLock::new(|| {
57        register_counter_vec!("cache_miss", "Total count of cache misses", &METRIC_LABELS).unwrap()
58    });
59
60    /// Total count of cache access attempts.
61    pub(crate) static CACHE_ACCESS: LazyLock<CounterVec> = LazyLock::new(|| {
62        register_counter_vec!(
63            "cache_access",
64            "Total count of cache access attempts",
65            &METRIC_LABELS
66        )
67        .unwrap()
68    });
69
70    /// Duration measured in milliseconds for accessing cache on a cache hit.
71    pub(crate) static LATENCY: LazyLock<HistogramVec> = LazyLock::new(|| {
72        register_histogram_vec!(
73            "latency",
74            "Duration measured in milliseconds for accessing cache on a cache hit",
75            &METRIC_LABELS
76        )
77        .unwrap()
78    });
79
80    /// Total space in bytes currently using for RBAC caching.
81    pub(crate) static CACHE_SIZE: LazyLock<IntGaugeVec> = LazyLock::new(|| {
82        register_int_gauge_vec!(

84            "Total space in bytes currently using for RBAC caching",
85            &METRIC_LABELS
86        )
87        .unwrap()
88    });
89
90    /// Maximum space in bytes for RBAC caching.
91    pub(crate) static MAX_CACHE_SIZE: LazyLock<IntGaugeVec> = LazyLock::new(|| {
92        register_int_gauge_vec!(
93            "max_cache_size",
94            "Maximum space in bytes for RBAC caching",
95            &METRIC_LABELS
96        )
97        .unwrap()
98    });
99
100    /// Total time in milliseconds taken to set up RBAC caching service to be operational.
101    pub(crate) static START_UP_TIME: LazyLock<IntGaugeVec> = LazyLock::new(|| {
102        register_int_gauge_vec!(
103            "start_up_time",
104            "Total time in milliseconds taken to set up RBAC caching service to be operational",
105            &METRIC_LABELS
106        )
107        .unwrap()
108    });

110    /// Total number of RBAC registration chain entries being stored as cache.
