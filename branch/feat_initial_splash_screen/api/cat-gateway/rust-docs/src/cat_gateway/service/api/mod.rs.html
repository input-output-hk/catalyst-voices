52/// Create the `OpenAPI` definition
53pub(crate) fn mk_api() -> OpenApiService<(HealthApi, CardanoApi, ConfigApi, DocumentApi), ()> {
54    let mut service = OpenApiService::new(
55        (
56            HealthApi,
57            (cardano::Api, cardano::staking::Api, cardano::cip36::Api),
58            ConfigApi,
59            DocumentApi,
60        ),
61        API_TITLE,
62        API_VERSION,
63    )
64    .contact(get_api_contact())
65    .description(API_DESCRIPTION)
66    .license(get_api_license())
67    .terms_of_service(TERMS_OF_SERVICE)
68    .url_prefix(Settings::api_url_prefix());
69
70    let hosts = Settings::api_host_names();
71
72    if hosts.is_empty() {
73        service = set_localhost_addresses(service);
74    } else {
75        for host in &hosts {
76            service = service.server(
77                ServerObject::new(host).description("Server host staging/production location."),
78            );
79        }
80    }
81
82    // Add server name if it is set
83    if let Some(name) = Settings::server_name() {
84        service = service.server(ServerObject::new(name).description("Server at server name."));
85    }
86    service
87}

89/// Set the localhost addresses descriptions.
90fn set_localhost_addresses<T>(mut service: OpenApiService<T, ()>) -> OpenApiService<T, ()> {
91    let port = Settings::bound_address().port();
92
93    // Get localhost name
94    if let Ok(hostname) = gethostname().into_string() {
95        let hostname_address = format!("http://{hostname}:{port}",);
96        service = service
97            .server(ServerObject::new(hostname_address).description("Server at localhost name."));
98    }
99
100    // Get local IP address v4 and v6
101    if let Ok(network_interfaces) = list_afinet_netifas() {
102        for (name, ip) in &network_interfaces {
103            if *name == "en0" {
104                let (address, desc) = match ip {
105                    IpAddr::V4(_) => {
106                        (
107                            format!("http://{ip}:{port}"),
108                            "Server at local IPv4 address.",
109                        )
110                    },
111                    IpAddr::V6(_) => {
112                        (
113                            format!("http://[{ip}]:{port}"),
114                            "Server at local IPv6 address.",
115                        )
116                    },
117                };
118                service = service.server(ServerObject::new(address).description(desc));
119            }
120        }
121    }
122    service
123}
