43/// Get RBAC registration endpoint.
44pub(crate) async fn endpoint(
45    lookup: Option<CatIdOrStake>, token: Option<CatalystRBACTokenV1>,
46) -> AllResponses {
47    let Some(persistent_session) = CassandraSession::get(true) else {
48        let err = anyhow!("Failed to acquire persistent db session");
49        error!("{err:?}");
50        return AllResponses::service_unavailable(&err, RetryAfterOption::Default);
51    };
52    let Some(volatile_session) = CassandraSession::get(false) else {
53        let err = anyhow!("Failed to acquire volatile db session");
54        error!("{err:?}");
55        return AllResponses::service_unavailable(&err, RetryAfterOption::Default);
56    };
57
58    let catalyst_id = match lookup {
59        Some(CatIdOrStake::CatId(v)) => v.into(),
60        Some(CatIdOrStake::Address(address)) => {
61            let address: StakeAddress = match address.try_into() {
62                Ok(a) => a,
63                Err(e) => return AllResponses::handle_error(&e),
64            };
65            // TODO: For now we want to select the latest registration (so query the volatile
66            // database first), but ideally the logic should be more sophisticated. See the issue
67            // for more details: https://github.com/input-output-hk/catalyst-voices/issues/2152
68            match catalyst_id_from_stake(&volatile_session, address.clone()).await {
69                Ok(Some(v)) => v,
70                Err(e) => return AllResponses::handle_error(&e),
71                Ok(None) => {
72                    match catalyst_id_from_stake(&persistent_session, address).await {
73                        Ok(Some(v)) => v,
74                        Err(e) => return AllResponses::handle_error(&e),
75                        Ok(None) => return Responses::NotFound.into(),
76                    }
77                },
78            }
79        },
80        None => {
81            match token {
82                Some(token) => token.catalyst_id().clone(),
83                None => {
84                    return Responses::UnprocessableContent(Json(RbacUnprocessableContent::new(
85                        "Either lookup parameter or token must be provided",
86                    )))
87                    .into()
88                },
89            }
90        },
91    };
92
93    match ChainInfo::new(&persistent_session, &volatile_session, &catalyst_id).await {
94        Ok(Some(info)) => {
95            match RbacRegistrationChain::new(&info) {
96                Ok(c) => Responses::Ok(Json(Box::new(c))).into(),
97                Err(e) => AllResponses::handle_error(&e),
98            }
99        },
100        Ok(None) => Responses::NotFound.into(),
101        Err(e) => AllResponses::handle_error(&e),
102    }
103}
