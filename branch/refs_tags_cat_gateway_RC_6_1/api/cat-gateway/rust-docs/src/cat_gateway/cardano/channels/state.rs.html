11/// Chain Indexer State structure
12#[derive(Debug, Clone)]
13#[allow(dead_code)]
14pub(crate) struct ChainIndexerState {
15    /// The immutable tip slot.
16    pub immutable_tip_slot: Slot,
17
18    /// Current Immutable db status of indexed data.
19    pub immutable_indexed_status: IndexedStatus,
20}
21
22impl ChainIndexerState {
23    /// Returns true if the Chain indexer reached immutable tip, so the all immutable data
24    /// if fully is indexed. False otherwise.
25    pub(crate) fn is_reached_immutable_tip(&self) -> bool {
26        // If the immutable part is fully indexed it must be only a one chunk
27        if let Some((_, end)) = self.immutable_indexed_status.first() {
28            end == &self.immutable_tip_slot && self.immutable_indexed_status.len() == 1
29        } else {
30            false
31        }
32    }
33
34    /// Returns true of the immutable part is fully consecutive indexed up to the
35    /// provided slot.
36    pub(crate) fn is_immutable_fully_indexed_up_to(&self, slot: Slot) -> bool {
37        // Looking into the first chunk, which represents a definitely consecutive indexed
38        // chunk of blocks.
39        if let Some((start, end)) = self.immutable_indexed_status.first() {
40            let slot = slot.sub(1.into());
41            start <= &slot && &slot <= end
42        } else {
43            false
44        }
45    }
46}

48/// Chain Indexer state sender multi-producer channel
49#[derive(Debug, Clone)]
50pub(crate) struct ChainIndexerStateSender(broadcast::Sender<ChainIndexerState>);
51
52/// Chain Indexer state receiver multi-consumer channel
53#[derive(Debug)]
54pub(crate) struct ChainIndexerStateReceiver {
55    /// The chain indexer's stored state, from the last `latest_state` call from the
56    /// `receiver`.
57    state: Option<ChainIndexerState>,
58    /// A `ChainIndexerState` receiver.
59    receiver: broadcast::Receiver<ChainIndexerState>,
60}
61
62impl ChainIndexerStateSender {
63    /// Creates a multi-producer channel for processing
64    /// Chain Indexer state events.
65    /// To subscribe on processing events and instantiate `ChainIndexerStateReceiver` call
66    /// `subscribe` method of the `ChainIndexerStateSender`
67    pub(crate) fn new() -> Self {
68        Self(broadcast::channel(1).0)
69    }
70
71    /// Dispatches the latest state to all registered listeners.
72    pub(crate) fn dispatch_state(&self, state: ChainIndexerState) {
73        dispatch_message(&self.0, state);
74    }
75
76    /// Creates a new multi-consumer `ChainIndexerStateReceiver` that will receive values
77    /// sent **after** this call to `subscribe`.
78    pub(crate) fn subscribe(&self) -> ChainIndexerStateReceiver {
79        ChainIndexerStateReceiver {
80            state: None,
81            receiver: self.0.subscribe(),
82        }
83    }
84}
85
86impl ChainIndexerStateReceiver {
87    /// Returns the stored state, from the last `latest_state` call. Returns `None` if the
88    /// state was never updated.
89    pub(crate) fn stored_state(&self) -> Option<&ChainIndexerState> {
90        self.state.as_ref()
91    }
92
93    /// Receives the latest Chain Indexer state from the channel.
94    /// Return `None` if the channel is closed.
95    pub(crate) async fn latest_state(&mut self) -> Option<&ChainIndexerState> {
96        self.state = receive_msg(&mut self.receiver).await;
97        self.stored_state()
98    }
