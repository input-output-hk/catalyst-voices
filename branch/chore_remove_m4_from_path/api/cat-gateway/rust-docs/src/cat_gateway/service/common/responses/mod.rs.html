41/// Default error responses
42#[derive(ApiResponse)]
43pub(crate) enum ErrorResponses {
44    /// ## Bad Request
45    ///
46    /// The client has not sent valid request, could be an invalid HTTP in general or
47    /// provided not correct headers, path or query arguments.
48    #[oai(status = 400)]
49    #[allow(dead_code)]
50    BadRequest,
51
52    /// ## Unauthorized
53    ///
54    /// The client has not sent valid authentication credentials for the requested
55    /// resource.
56    #[oai(status = 401)]
57    Unauthorized(Json<Unauthorized>),
58
59    /// ## Forbidden
60    ///
61    /// The client has not sent valid authentication credentials for the requested
62    /// resource.
63    #[oai(status = 403)]
64    Forbidden(Json<Forbidden>),
65
66    /// ## URI Too Long
67    ///
68    /// The client sent a request with the URI is longer than the server is willing to
69    /// interpret
70    #[oai(status = 414)]
71    #[allow(dead_code)]
72    UriTooLong,
73
74    /// ## Precondition Failed
75    ///
76    /// The client has not sent valid data in its request, headers, parameters or body.
77    #[oai(status = 412)]
78    PreconditionFailed(Json<PreconditionFailed>),
79
80    /// ## Too Many Requests
81    ///
82    /// The client has sent too many requests in a given amount of time.
83    #[oai(status = 429)]
84    TooManyRequests(
85        Json<TooManyRequests>,
86        #[oai(header = "Retry-After")] RetryAfterHeader,
87    ),
88
89    /// ## Request Header Fields Too Large
90    ///
91    /// The client sent a request with too large header fields.
92    #[oai(status = 431)]
93    #[allow(dead_code)]
94    RequestHeaderFieldsTooLarge,
95
96    /// ## Internal Server Error.
97    ///
98    /// An internal server error occurred.
99    ///
100    /// *The contents of this response should be reported to the projects issue tracker.*
101    #[oai(status = 500)]
102    ServerError(Json<InternalServerError>),
103
104    /// ## Service Unavailable
105    ///
106    /// The service is not available, try again later.
107    ///
108    /// *This is returned when the service either has not started,
109    /// or has become unavailable.*
110    #[oai(status = 503)]
111    ServiceUnavailable(
112        Json<ServiceUnavailable>,
113        #[oai(header = "Retry-After")] Option<RetryAfterHeader>,
114    ),
115}

117impl ErrorResponses {
118    /// Handle a 401 unauthorized response.
119    ///
120    /// Returns a 401 Unauthorized response.
121    /// Its OK if we actually never call this.  Required for the API.
122    /// May be generated by the ingress.
123    pub(crate) fn unauthorized(text: String) -> Self {
124        let error = Unauthorized::new(Some(text));
125        ErrorResponses::Unauthorized(Json(error))
126    }
127
128    /// Handle a 403 forbidden response.
129    ///
130    /// Returns a 403 Forbidden response.
131    /// Its OK if we actually never call this.  Required for the API.
132    /// May be generated by the ingress.
133    pub(crate) fn forbidden(roles: Option<Vec<String>>) -> Self {
134        let error = Forbidden::new(None, roles);
135        ErrorResponses::Forbidden(Json(error))
136    }
137}

139/// Combine provided responses type with the default responses under one type.
140pub(crate) enum WithErrorResponses<T> {
141    /// Provided responses
142    With(T),
143    /// Error responses
144    Error(ErrorResponses),
145}
146
147impl<T> WithErrorResponses<T> {
148    /// Handle a 5xx response.
149    /// Returns a Server Error or a Service Unavailable response.
150    pub(crate) fn handle_error(err: &anyhow::Error) -> Self {
151        match err {
152            err if err.is::<bb8::RunError<tokio_postgres::Error>>() => {
153                // Event DB failed
154                set_event_db_liveness(false);
155                Self::service_unavailable(err, RetryAfterOption::Default)
156            },
157            err if err.is::<CassandraSessionError>() => {
158                // Index DB failed
159                set_index_db_liveness(false);
160                Self::service_unavailable(err, RetryAfterOption::Default)
161            },
162            err => Self::internal_error(err),
163        }
164    }
165
166    /// Handle a 503 service unavailable error response.
167    ///
168    /// Returns a 503 Service unavailable Error response.
169    pub(crate) fn service_unavailable(err: &anyhow::Error, retry: RetryAfterOption) -> Self {
170        let error = ServiceUnavailable::new(None);
171        error!(id=%error.id(), error=?err, retry_after=?retry);
172        let retry = match retry {
173            RetryAfterOption::Default => Some(RetryAfterHeader::default()),
174            RetryAfterOption::None => None,
175            RetryAfterOption::Some(value) => Some(value),
176        };
177        WithErrorResponses::Error(ErrorResponses::ServiceUnavailable(Json(error), retry))
178    }
179
180    /// Handle a 500 internal error response.
181    ///
182    /// Returns a 500 Internal Error response.
183    pub(crate) fn internal_error(err: &anyhow::Error) -> Self {
184        let error = InternalServerError::new(None);
185        error!(id=%error.id(), error=?err);
186        WithErrorResponses::Error(ErrorResponses::ServerError(Json(error)))
187    }
188
189    /// Handle a 401 unauthorized response.
190    ///
191    /// Returns a 401 Unauthorized response.
192    /// Its OK if we actually never call this.  Required for the API.
193    /// May be generated by the ingress.
194    pub(crate) fn unauthorized(text: String) -> Self {
195        WithErrorResponses::Error(ErrorResponses::unauthorized(text))
196    }
197
198    /// Handle a 403 forbidden response.
199    ///
200    /// Returns a 403 Forbidden response.
201    /// Its OK if we actually never call this.  Required for the API.
202    /// May be generated by the ingress.
203    #[allow(dead_code)]
204    pub(crate) fn forbidden(roles: Option<Vec<String>>) -> Self {
205        WithErrorResponses::Error(ErrorResponses::forbidden(roles))
206    }
207
208    /// Handle a 412 precondition failed response.
209    ///
210    /// Returns a 412 precondition failed response.
211    fn precondition_failed(errors: Vec<poem::Error>) -> Self {
212        let error = PreconditionFailed::new(errors);
213        WithErrorResponses::Error(ErrorResponses::PreconditionFailed(Json(error)))
214    }
215
216    /// Handle a 429 rate limiting response.
217    ///
218    /// Returns a 429 Rate limit response.
219    /// Its OK if we actually never call this.  Required for the API.
220    /// May be generated by the ingress.
221    #[allow(dead_code)]
222    pub(crate) fn rate_limit(retry_after: Option<RetryAfterHeader>) -> Self {
223        let retry_after = retry_after.unwrap_or_default();
224        let error = TooManyRequests::new(None);
225        WithErrorResponses::Error(ErrorResponses::TooManyRequests(Json(error), retry_after))
226    }
227}

229impl<T: ApiResponse> From<T> for WithErrorResponses<T> {
230    fn from(val: T) -> Self {
231        Self::With(val)
232    }

235impl<T: ApiResponse> ApiResponse for WithErrorResponses<T> {
236    const BAD_REQUEST_HANDLER: bool = true;
237
238    fn meta() -> MetaResponses {
239        let t_meta = T::meta();
240        let default_meta = ErrorResponses::meta();
241
242        let mut responses = HashSet::new();
243        responses.extend(
244            t_meta
245                .responses
246                .into_iter()
247                .map(FilteredByStatusCodeResponse),
248        );
249        responses.extend(
250            default_meta
251                .responses
252                .into_iter()
253                .map(FilteredByStatusCodeResponse),
254        );
255
256        let responses =
257            responses
258                .into_iter()
259                .map(|val| {
260                    let mut response = val.0;
261                    // Make modifications to the responses to set common headers
262                    if let Some(status) = response.status {
263                        // Only 2xx and 4xx responses get RateLimit Headers.
264                        if (200..300).contains(&status) || (400..500).contains(&status) {
265                            response.headers.insert(0usize, MetaHeader {
266                                name: "RateLimit".to_string(),
267                                description: Some("RateLimit Header.".to_string()),
268                                required: false,
269                                deprecated: false,
270                                schema: <RateLimitHeader as poem_openapi::types::Type>::schema_ref(),
271                            });
272                        }
273
274                        // All responses get Access-Control-Allow-Origin headers
275                        response.headers.insert(0usize, MetaHeader {
276                            name: "Access-Control-Allow-Origin".to_string(),
277                            description: Some("Access-Control-Allow-Origin Header.".to_string()),
278                            required: false,
279                            deprecated: false,
280                            schema: <AccessControlAllowOriginHeader as poem_openapi::types::Type>::schema_ref(),
281                        });
282                    }
283                    response
284                })
285                .collect();
286
287        // Add Rate limiting headers to ALL 2xx and 4xx responses
288        // for response in responses.iter_mut() {
289        //    response.
290        //    debug!(response = response);
291        //}
292
293        MetaResponses { responses }
294    }
295
296    fn register(registry: &mut Registry) {
297        ErrorResponses::register(registry);
298        T::register(registry);
299    }
300
301    fn from_parse_request_error(err: poem::Error) -> Self {
302        if err.status() == StatusCode::UNAUTHORIZED {
303            WithErrorResponses::unauthorized(err.to_string())
304        } else if err.status() == StatusCode::FORBIDDEN {
305            WithErrorResponses::forbidden(Some(vec![err.to_string()]))
306        } else {
307            WithErrorResponses::precondition_failed(vec![err])
308        }
309    }
310}

312impl<T: IntoResponse + Send> IntoResponse for WithErrorResponses<T> {
313    fn into_response(self) -> poem::Response {
314        match self {
315            Self::With(t) => t.into_response(),
316            Self::Error(default) => default.into_response(),
317        }
318    }

321/// `FilteredByStatusCodeResponse` is used to filter out duplicate responses by status
322/// code.
323struct FilteredByStatusCodeResponse(MetaResponse);
324impl PartialEq for FilteredByStatusCodeResponse {
325    fn eq(&self, other: &Self) -> bool {
326        self.0.status.eq(&other.0.status)
327    }
328}
329impl Eq for FilteredByStatusCodeResponse {}
330impl Hash for FilteredByStatusCodeResponse {
331    fn hash<H: Hasher>(&self, state: &mut H) {
332        self.0.status.hash(state);
333    }
