13/// Auth token in the form of catv1..
14pub type EncodedAuthToken = String;
15
16/// The header name that holds the authorization RBAC token
17pub(crate) const AUTHORIZATION_HEADER: &str = "Authorization";
18
19/// Cached auth tokens
20static CACHE: LazyLock<Cache<EncodedAuthToken, CatalystRBACTokenV1>> = LazyLock::new(|| {
21    Cache::builder()
22        // Time to live (TTL): 30 minutes
23        .time_to_live(Duration::from_secs(30 * 60))
24        // Time to idle (TTI):  5 minutes
25        .time_to_idle(Duration::from_secs(5 * 60))
26        // Create the cache.
27        .build()
28});
29
30/// Dummy Public Key
31const DUMMY_PUB_KEY_BYTES: [u8; PUBLIC_KEY_LENGTH] = [
32    180, 91, 130, 149, 226, 112, 29, 45, 188, 141, 64, 147, 250, 233, 75, 151, 151, 53, 248, 197,
33    225, 122, 24, 67, 207, 100, 162, 152, 232, 102, 89, 162,
34];

36/// Catalyst RBAC Access Token
37#[derive(SecurityScheme)]
38#[oai(
39    ty = "bearer",
40    key_name = "Authorization", // MUST match the `AUTHORIZATION_HEADER` constant.
41    bearer_format = "catalyst-rbac-token",
42    checker = "checker_api_catalyst_auth"
43)]
44#[allow(dead_code, clippy::module_name_repetitions)]
45pub struct CatalystRBACSecurityScheme(pub CatalystRBACTokenV1);

47/// Error with the Authorization Token
48///
49/// We can not parse it, so its a 401 response.
50#[derive(Debug, thiserror::Error)]
51#[error("Invalid Catalyst RBAC Auth Token")]
52pub struct AuthTokenError;
53
54impl ResponseError for AuthTokenError {
55    fn status(&self) -> StatusCode {
56        StatusCode::UNAUTHORIZED
57    }
58
59    /// Convert this error to a HTTP response.
60    fn as_response(&self) -> poem::Response
61    where Self: Error + Send + Sync + 'static {
62        ErrorResponses::unauthorized().into_response()
63    }
64}
65
66/// Token does not have required access rights
67///
68/// Not enough access rights, so its a 403 response.
69#[derive(Debug, thiserror::Error)]
70#[error("Insufficient Permission for Catalyst RBAC Token")]
71pub struct AuthTokenAccessViolation(Vec<String>);
72
73impl ResponseError for AuthTokenAccessViolation {
74    fn status(&self) -> StatusCode {
75        StatusCode::FORBIDDEN
76    }
77
78    /// Convert this error to a HTTP response.
79    fn as_response(&self) -> poem::Response
80    where Self: Error + Send + Sync + 'static {
81        // TODO: Actually check permissions needed for an endpoint.
82        ErrorResponses::forbidden(Some(self.0.clone())).into_response()
83    }
84}

86/// Time in the past the Token can be valid for.
87const MAX_TOKEN_AGE: Duration = Duration::from_secs(60 * 60); // 1 hour.
88
89/// Time in the future the Token can be valid for.
90const MAX_TOKEN_SKEW: Duration = Duration::from_secs(5 * 60); // 5 minutes
91
92/// When added to an endpoint, this hook is called per request to verify the bearer token
93/// is valid.
94async fn checker_api_catalyst_auth(
95    _req: &Request, bearer: Bearer,
96) -> poem::Result<CatalystRBACTokenV1> {
97    /// Temporary: Conditional RBAC for testing
98    const RBAC_OFF: &str = "RBAC_OFF";
99
100    // First check the token can be deserialized.
101    let token = match CatalystRBACTokenV1::decode(&bearer.token) {
102        Ok(token) => token,
103        Err(err) => {
104            // Corrupted Authorisation Token received
105            error!("Corrupt auth token: {:?}", err);
106            Err(AuthTokenError)?
107        },
108    };
109
110    // If env var explicitly set by SRE, switch off full verification
111    if env::var(RBAC_OFF).is_ok() {
112        return Ok(token);
113    };
114
115    // Check if the token is young enough.
116    if !token.is_young(MAX_TOKEN_AGE, MAX_TOKEN_SKEW) {
117        // Token is too old or too far in the future.
118        error!("Auth token expired: {:?}", token);
119        Err(AuthTokenAccessViolation(vec!["EXPIRED".to_string()]))?;
120    }
121
122    // Its valid and young enough, check if its in the auth cache.
123    // This get() will extend the entry life for another 5 minutes.
124    // Even though we keep calling get(), the entry will expire
125    // after 30 minutes (TTL) from the origin insert().
126    // This is an optimization which saves us constantly looking up registrations we have
127    // already validated.
128    if let Some(token) = CACHE.get(&bearer.token).await {
129        return Ok(token);
130    }
131
132    // Ok, so its validly decoded, but we haven't seen it before.
133    // Check that the token is able to be authorized.
134
135    // Get pub key from CERTS state given decoded KID from decoded bearer token
136    // TODO: Look up certs from the Kid based on RBAC Registrations.
137    // let pub_key_bytes = if let Some(cert) = CERTS.get(&token.kid) {
138    //    *cert
139    //} else {
140    // error!(kid = %token.kid, "Invalid KID");
141    // Err(AuthTokenAccessViolation(vec!["UNREGISTERED".to_string()]))?;
142    //};
143
144    // Verify the token signature using the public key.
145    let public_key = match VerifyingKey::from_bytes(&DUMMY_PUB_KEY_BYTES) {
146        Ok(pub_key) => pub_key,
147        Err(err) => {
148            // In theory this should never happen.
149            error!("Invalid public key: {:?}", err);
150            Err(AuthTokenAccessViolation(vec![
151                "INVALID PUBLIC KEY".to_string()
152            ]))?
153        },
154    };
155
156    if let Err(error) = token.verify(&public_key) {
157        error!(error=%error, "Token Invalidly Signed");
158        Err(AuthTokenAccessViolation(vec![
159            "INVALID SIGNATURE".to_string()
160        ]))?;
161    }
162
163    // This entry will expire after 5 minutes (TTI) if there is no more ().
164    CACHE.insert(bearer.token, token.clone()).await;
165
166    Ok(token)
167}
