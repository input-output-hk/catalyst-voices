17/// Title.
18const TITLE: &str = "Cardano Native Asset Name";
19/// Description.
20const DESCRIPTION: &str = r"The name given to the native asset when it was minted.
21If the name can be converted to UTF8, its string is represented directly.
22Otherwise it will be represented as escaped ascii.
23Any `\` present in the name will be replaced with `\\` in all cases.";
24/// Example.
25const EXAMPLE: &str = "My Cool\nAsset";
26/// Minimum length.
27const MIN_LENGTH: usize = 0;
28/// Maximum length. (True length is 32, but escaping can double its size).
29const MAX_LENGTH: usize = 64;
30/// Validation Regex Pattern
31const PATTERN: &str = r"[\S\s]{0,64}";
32
33/// Schema.
34static SCHEMA: LazyLock<MetaSchema> = LazyLock::new(|| {
35    MetaSchema {
36        title: Some(TITLE.to_owned()),
37        description: Some(DESCRIPTION),
38        example: Some(Value::String(EXAMPLE.to_string())),
39        max_length: Some(MAX_LENGTH),
40        min_length: Some(MIN_LENGTH),
41        pattern: Some(PATTERN.to_string()),
42        ..poem_openapi::registry::MetaSchema::ANY
43    }
44});
45
46/// Because ALL the constraints are defined above, we do not ever need to define them in
47/// the API. BUT we do need to make a validator.
48/// This helps enforce uniform validation.
49fn is_valid(_name: &str) -> bool {
50    // Anything is valid.
51    true
52}
53
54impl_string_types!(
55    AssetName,
56    "string",
57    "cardano:asset_name",
58    Some(SCHEMA.clone()),
59    is_valid
60);
61
62impl Example for AssetName {
63    fn example() -> Self {
64        Self(EXAMPLE.to_owned())
65    }
66}
67
68impl From<Vec<u8>> for AssetName {
69    fn from(value: Vec<u8>) -> Self {
70        match String::from_utf8(value.clone()) {
71            Ok(name) => {
72                // UTF8 - Yay
73                // Escape any `\` so its consistent with escaped ascii below.
74                let name = name.replace('\\', r"\\");
75                Self(name)
76            },
77            Err(_) => Self(value.escape_ascii().to_string()),
78        }
79    }
80}
81
82impl From<String> for AssetName {
83    fn from(value: String) -> Self {
84        Self(value)
85    }
