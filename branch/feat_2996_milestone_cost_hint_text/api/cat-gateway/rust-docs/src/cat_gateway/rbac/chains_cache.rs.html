1//! A cache of RBAC chains.
2
3use std::sync::LazyLock;
4
5use catalyst_types::catalyst_id::CatalystId;
6use moka::{policy::EvictionPolicy, sync::Cache};
7use rbac_registration::registration::cardano::RegistrationChain;
8
9use crate::{
10    metrics::rbac_cache::reporter::{PERSISTENT_CHAINS_CACHE_HIT, PERSISTENT_CHAINS_CACHE_MISS},
11    settings::Settings,
12};
13
14/// A cache of persistent RBAC chains.
15static PERSISTENT_CHAINS: LazyLock<Cache<CatalystId, RegistrationChain>> = LazyLock::new(|| {
16    Cache::builder()
17        .eviction_policy(EvictionPolicy::lru())
18        .max_capacity(Settings::rbac_cfg().persistent_chains_cache_size)
19        .build()
20});
21
22/// Add (or update) a persistent chain to the cache.
23pub fn cache_persistent_rbac_chain(id: CatalystId, chain: RegistrationChain) {
24    PERSISTENT_CHAINS.insert(id, chain);
25}
26
27/// Returns a cached persistent chain by the given Catalyst ID.
28pub fn cached_persistent_rbac_chain(id: &CatalystId) -> Option<RegistrationChain> {
29    let api_host_names = Settings::api_host_names().join(",");
30    let service_id = Settings::service_id();
31
32    let res = PERSISTENT_CHAINS.get(id);
33    if res.is_some() {
34        PERSISTENT_CHAINS_CACHE_HIT
35            .with_label_values(&[&api_host_names, service_id])
36            .inc();
37    } else {
38        PERSISTENT_CHAINS_CACHE_MISS
39            .with_label_values(&[&api_host_names, service_id])
40            .inc();
41    }
42    res
43}
44
45/// Returns an approximate number of entries in the `PERSISTENT_CHAINS` cache.
46pub fn persistent_rbac_chains_cache_size() -> u64 {
47    PERSISTENT_CHAINS.entry_count()
48}
