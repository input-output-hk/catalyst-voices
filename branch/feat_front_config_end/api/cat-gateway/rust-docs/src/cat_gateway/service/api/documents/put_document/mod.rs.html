29/// Endpoint responses.
30#[derive(ApiResponse)]
31pub(crate) enum Responses {
32    /// ## Created
33    ///
34    /// The Document was stored OK for the first time.
35    #[oai(status = 201)]
36    Created,
37    /// ## No Content
38    ///
39    /// The Document was already stored, and has not changed.
40    #[oai(status = 204)]
41    NoContent,
42    /// ## Unprocessable Content
43    ///
44    /// Error Response. The document submitted is invalid.
45    #[oai(status = 422)]
46    UnprocessableContent(Json<PutDocumentUnprocessableContent>),
47    /// ## Content Too Large
48    ///
49    /// Payload Too Large. The document exceeds the maximum size of a legitimate single
50    /// document.
51    #[oai(status = 413)]
52    PayloadTooLarge,
53}
54
55/// All responses.
56pub(crate) type AllResponses = WithErrorResponses<Responses>;

58/// # PUT `/document`
59pub(crate) async fn endpoint(doc_bytes: Vec<u8>, mut token: CatalystRBACTokenV1) -> AllResponses {
60    let Ok(doc): Result<catalyst_signed_doc::CatalystSignedDocument, _> =
61        doc_bytes.as_slice().try_into()
62    else {
63        return Responses::UnprocessableContent(Json(PutDocumentUnprocessableContent::new(
64            "Invalid CBOR bytes, cannot decode Catalyst Signed Document",
65            None,
66        )))
67        .into();
68    };
69
70    // validate document integrity
71    match catalyst_signed_doc::validator::validate(&doc, &DocProvider).await {
72        Ok(true) => (),
73        Ok(false) => {
74            return Responses::UnprocessableContent(Json(PutDocumentUnprocessableContent::new(
75                "Failed validating document integrity",
76                serde_json::to_value(doc.problem_report()).ok(),
77            )))
78            .into();
79        },
80        Err(err) => {
81            // means that something happened inside the `DocProvider`, some db error.
82            return AllResponses::handle_error(&err);
83        },
84    }
85
86    // validate document signatures
87    let verifying_key_provider =
88        match VerifyingKeyProvider::try_from_kids(&mut token, &doc.kids()).await {
89            Ok(value) => value,
90            Err(err) if err.is::<CassandraSessionError>() => {
91                return AllResponses::service_unavailable(&err, RetryAfterOption::Default)
92            },
93            Err(err) => {
94                return Responses::UnprocessableContent(Json(PutDocumentUnprocessableContent::new(
95                    &err, None,
96                )))
97                .into()
98            },
99        };
100    match catalyst_signed_doc::validator::validate_signatures(&doc, &verifying_key_provider).await {
101        Ok(true) => (),
102        Ok(false) => {
103            return Responses::UnprocessableContent(Json(PutDocumentUnprocessableContent::new(
104                "Failed validating document signatures",
105                serde_json::to_value(doc.problem_report()).ok(),
106            )))
107            .into();
108        },
109        Err(err) => {
110            return AllResponses::handle_error(&err);
111        },
112    };
113
114    if doc.problem_report().is_problematic() {
115        return Responses::UnprocessableContent(Json(PutDocumentUnprocessableContent::new(
116            "Invalid Catalyst Signed Document",
117            serde_json::to_value(doc.problem_report()).ok(),
118        )))
119        .into();
120    }
121
122    match validate_against_original_doc(&doc).await {
123        Ok(true) => (),
124        Ok(false) => {
125            return Responses::UnprocessableContent(Json(
126                PutDocumentUnprocessableContent::new("Failed validating document: catalyst-id or role does not match the current version.", None),
127            ))
128            .into();
129        },
130        Err(err) => return AllResponses::handle_error(&err),
131    };
132
133    // update the document storing in the db
134    match store_document_in_db(&doc, doc_bytes).await {
135        Ok(true) => Responses::Created.into(),
136        Ok(false) => Responses::NoContent.into(),
137        Err(err) if err.is::<StoreError>() => {
138            Responses::UnprocessableContent(Json(PutDocumentUnprocessableContent::new(
139                "Document with the same `id` and `ver` already exists",
140                serde_json::to_value(doc.problem_report()).ok(),
141            )))
142            .into()
143        },
144        Err(err) => AllResponses::handle_error(&err),
145    }
146}

148/// Fetch the latest version and ensure its catalyst-id match those in the newer version.
149async fn validate_against_original_doc(doc: &CatalystSignedDocument) -> anyhow::Result<bool> {
150    let original_doc = match FullSignedDoc::retrieve(&doc.doc_id()?.uuid(), None).await {
151        Ok(doc) => doc,
152        Err(e) if e.is::<error::NotFoundError>() => return Ok(true),
153        Err(e) => anyhow::bail!("Database error: {e}"),
154    };
155
156    let original_authors = original_doc
157        .body()
158        .authors()
159        .iter()
160        .map(|author| catalyst_signed_doc::IdUri::from_str(author))
161        .collect::<Result<HashSet<_>, _>>()?;
162    let authors: HashSet<_> = doc.authors().into_iter().collect();
163    Ok(authors == original_authors)
164}

166/// Store a provided and validated document inside the db.
167/// Returns `true` if its a new document.
168/// Returns `false` if the same document already exists.
169async fn store_document_in_db(
170    doc: &catalyst_signed_doc::CatalystSignedDocument, doc_bytes: Vec<u8>,
171) -> anyhow::Result<bool> {
172    let authors = doc.authors().iter().map(ToString::to_string).collect();
173
174    let doc_meta_json = match serde_json::to_value(doc.doc_meta()) {
175        Ok(json) => json,
176        Err(e) => {
177            anyhow::bail!("Cannot decode document metadata into JSON, err: {e}");
178        },
179    };
180
181    let payload = if matches!(
182        doc.doc_content_type()?,
183        catalyst_signed_doc::ContentType::Json
184    ) {
185        match serde_json::from_slice(doc.doc_content().decoded_bytes()?) {
186            Ok(payload) => Some(payload),
187            Err(e) => {
188                anyhow::bail!("Invalid Document Content, not Json encoded: {e}");
189            },
190        }
191    } else {
192        None
193    };
194
195    let doc_body = SignedDocBody::new(
196        doc.doc_id()?.into(),
197        doc.doc_ver()?.into(),
198        doc.doc_type()?.into(),
199        authors,
200        Some(doc_meta_json),
201    );
202
203    FullSignedDoc::new(doc_body, payload, doc_bytes)
204        .store()
205        .await
