1//! Define `Precondition Failed` response type.
2
3use poem_openapi::{types::Example, Object};
4
5use crate::service::common;
6
7/// The client has not sent valid data in its request, headers, parameters or body.
8#[derive(Object)]
9#[oai(example)]
10pub(crate) struct PreconditionFailed {
11    #[oai(validator(max_items = "1000", min_items = "1"))]
12    /// Details of each error in the content that was detected.
13    ///
14    /// Note: This may not be ALL errors in the content, as validation of content can stop
15    /// at any point an error is detected.
16    detail: Vec<ContentErrorDetail>,
17}
18
19impl PreconditionFailed {
20    /// Create a new `ContentErrorDetail` Response Payload.
21    pub(crate) fn new(errors: Vec<poem::Error>) -> Self {
22        let mut detail = vec![];
23        for error in errors {
24            detail.push(ContentErrorDetail::new(&error));
25        }
26
27        Self { detail }
28    }
29}
30
31impl Example for PreconditionFailed {
32    /// Example for the Too Many Requests Payload.
33    fn example() -> Self {
34        Self {
35            detail: vec![ContentErrorDetail::example()],
36        }
37    }
38}
39

41
42#[derive(Object)]
43#[oai(example)]
44/// Individual details of a single error that was detected with the content of the
45/// request.
46pub(crate) struct ContentErrorDetail {
47    /// The location of the error
48    #[oai(validator(max_items = 100), nullable)]
49    loc: Option<Vec<common::types::generic::error_msg::ErrorMessage>>,
50    /// The error message.
51    #[oai(validator(max_length = "1000", pattern = "^[0-9a-zA-Z].*$"), nullable)]
52    msg: Option<common::types::generic::error_msg::ErrorMessage>,
53    /// The type of error
54    #[oai(
55        rename = "type",
56        validator(max_length = "1000", pattern = "^[0-9a-zA-Z].*$"),
57        nullable
58    )]
59    err_type: Option<common::types::generic::error_msg::ErrorMessage>,
60}
61
62impl Example for ContentErrorDetail {
63    /// Example for the `ContentErrorDetail` Payload.
64    fn example() -> Self {
65        Self {
66            loc: Some(vec!["body".into()]),
67            msg: Some("Value is not a valid dict.".into()),
68            err_type: Some("type_error.dict".into()),
69        }
70    }
71}
72
73impl ContentErrorDetail {
74    /// Create a new `ContentErrorDetail` Response Payload.
75    pub(crate) fn new(error: &poem::Error) -> Self {
76        // TODO: See if we can get more info from the error than this.
77        Self {
78            loc: None,
79            msg: Some(error.to_string().into()),
80            err_type: None,
81        }
82    }
