39/// Endpoint responses.
40#[derive(ApiResponse)]
41pub(crate) enum Responses {
42    /// ## Ok
43    ///
44    /// Success returns a list of registration transaction ids.
45    #[oai(status = 200)]
46    Ok(Json<Box<RbacRegistrations>>),
47
48    /// No valid registration.
49    #[oai(status = 404)]
50    NotFound,
51
52    /// Response for unprocessable content.
53    #[oai(status = 422)]
54    UnprocessableContent(Json<RbacUnprocessableContent>),
55}
56
57pub(crate) type AllResponses = WithErrorResponses<Responses>;
58
59/// Get RBAC registration endpoint.
60pub(crate) async fn endpoint(
61    lookup: Option<CatIdOrStake>, auth_catalyst_id: Option<IdUri>, detailed: bool,
62) -> AllResponses {
63    let Some(session) = CassandraSession::get(true) else {
64        error!("Failed to acquire db session");
65        let err = anyhow!("Failed to acquire db session");
66        return AllResponses::service_unavailable(&err, RetryAfterOption::Default);
67    };
68
69    let catalyst_id = match lookup {
70        Some(CatIdOrStake::CatId(v)) => v.into(),
71        Some(CatIdOrStake::Address(address)) => {
72            let address = match address.try_into() {
73                Ok(a) => a,
74                Err(e) => return AllResponses::handle_error(&e),
75            };
76            match catalyst_id_from_stake(&session, address).await {
77                Ok(Some(v)) => v,
78                Ok(None) => return Responses::NotFound.into(),
79                Err(e) => return AllResponses::handle_error(&e),
80            }
81        },
82        None => {
83            match auth_catalyst_id {
84                Some(id) => id,
85                None => {
86                    return Responses::UnprocessableContent(Json(RbacUnprocessableContent::new(
87                        "Either lookup parameter or token must be provided",
88                    )))
89                    .into()
90                },
91            }
92        },
93    };
94
95    let registrations = match registrations(&session, &catalyst_id).await {
96        Ok(v) => v,
97        Err(e) => return AllResponses::handle_error(&e),
98    };
99    match build_rbac_reg(registrations, detailed) {
100        Ok(Some(reg)) => {
101            Responses::Ok(Json(Box::new(RbacRegistrations {
102                catalyst_id: catalyst_id.into(),
103                finalized: Some(reg),
104                volatile: None,
105            })))
106            .into()
107        },
108        Ok(None) => Responses::NotFound.into(),
109        Err(e) => AllResponses::handle_error(&e),
110    }
111}

113/// Returns a Catalyst ID for the given stake address.
114async fn catalyst_id_from_stake(
115    session: &CassandraSession, address: StakeAddress,
116) -> anyhow::Result<Option<IdUri>> {
117    let mut iter = CatalystIdQuery::execute(session, CatalystIdQueryParams {
118        stake_address: address.into(),
119    })
120    .await
121    .context("Failed to query Catalyst ID from stake address {e:?}")?;
122
123    match iter.next().await {
124        Some(Ok(v)) => Ok(Some(v.catalyst_id.into())),
125        Some(Err(e)) => {
126            Err(anyhow!(
127                "Failed to query Catalyst ID from stake address {e:?}"
128            ))
129        },
130        None => Ok(None),
131    }
132}

134/// Returns a list of `Cip509` for the given Catalyst ID.
135async fn registrations(
136    session: &CassandraSession, catalyst_id: &IdUri,
137) -> anyhow::Result<Vec<rbac_registration::cardano::cip509::Cip509>> {
138    let network = Settings::cardano_network();
139
140    let mut iter = RbacQuery::execute(session, RbacQueryParams {
141        catalyst_id: catalyst_id.clone().into(),
142    })
143    .await?;
144
145    let mut result = Vec::new();
146    while let Some(reg) = iter.next().await.transpose()? {
147        let point = Point::fuzzy(reg.slot_no.into());
148        let block = ChainFollower::get_block(network, point)
149            .await
150            .context("Unable to get block")?
151            .data;
152        if block.point().slot_or_default() != reg.slot_no.into() {
153            // The `ChainFollower::get_block` function can return the next consecutive block if it
154            // cannot find the exact one. This shouldn't happen, but we need to check anyway.
155            bail!("Unable to find exact block");
156        }
157        let reg =
158            rbac_registration::cardano::cip509::Cip509::new(&block, reg.txn_index.into(), &[])
159                .context("Invalid RBAC registration")?
160                .context("No RBAC registration at this block and txn index")?;
161        result.push(reg);
162    }
163    Ok(result)
164}

166/// Build a response `RbacRegistration` from the provided CIP 509 registrations lists
167fn build_rbac_reg(
168    mut regs: Vec<rbac_registration::cardano::cip509::Cip509>, detailed: bool,
169) -> anyhow::Result<Option<RbacReg>> {
170    // sort registrations in the ascending order from the oldest to the latest
171    regs.sort_by(|a, b| {
172        let a = a.origin().point();
173        let b = b.origin().point();
174        a.cmp(b)
175    });
176
177    let details: Cip509List = if detailed {
178        regs.iter()
179            .map(TryInto::try_into)
180            .collect::<anyhow::Result<Vec<_>>>()?
181            .into()
182    } else {
183        Vec::new().into()
184    };
185    let chain = build_chain(regs);
186    if details.is_empty() && chain.is_none() {
187        return Ok(None);
188    }
189    Ok(Some(RbacRegistration { chain, details }.into()))
190}
191
192/// Try to build a `RegistrationChain` from the provided registration list
193fn build_chain(regs: Vec<rbac_registration::cardano::cip509::Cip509>) -> Option<RegChain> {
194    let mut regs_iter = regs.into_iter();
195    while let Some(try_first) = regs_iter.next() {
196        if let Ok(mut chain) =
197            rbac_registration::registration::cardano::RegistrationChain::new(try_first)
198        {
199            for reg in regs_iter {
200                let Ok(updated_chain) = chain.update(reg) else {
201                    continue;
202                };
203                chain = updated_chain;
204            }
205            return Some(chain.into());
206        }
207        continue;
208    }
209    None
