1//! Get stake addr registrations
2
3use std::sync::Arc;
4
5use cardano_blockchain_types::Slot;
6use scylla::{
7    prepared_statement::PreparedStatement, transport::iterator::TypedRowStream, DeserializeRow,
8    SerializeRow, Session,
9};
10use tracing::error;
11
12use crate::db::{
13    index::{
14        queries::{PreparedQueries, PreparedSelectQuery},
15        session::CassandraSession,
16    },
17    types::{DbSlot, DbTxnIndex},
18};
19
20/// Get registrations from stake public key query.
21const GET_REGISTRATIONS_FROM_STAKE_PK_QUERY: &str =
22    include_str!("../cql/get_registrations_w_stake_addr.cql");
23
24/// Get registration
25#[derive(SerializeRow)]
26pub(crate) struct GetRegistrationParams {
27    /// Stake public key.
28    stake_public_key: Vec<u8>,
29    /// Block Slot Number.
30    slot_no: DbSlot,
31}
32
33impl GetRegistrationParams {
34    /// Create a new instance of [`GetRegistrationParams`]
35    pub(crate) fn new(stake_public_key: Vec<u8>, slot_no: Slot) -> Self {
36        Self {
37            stake_public_key,
38            slot_no: slot_no.into(),
39        }
40    }
41}
42
43/// Get registration query.
44#[derive(DeserializeRow)]
45pub(crate) struct GetRegistrationQuery {
46    /// Nonce value after normalization.
47    pub nonce: num_bigint::BigInt,
48    /// Slot Number the cert is in.
49    pub slot_no: DbSlot,
50    /// Transaction Index.
51    pub txn_index: DbTxnIndex,
52    /// Raw Nonce value.
53    pub raw_nonce: num_bigint::BigInt,
54    /// Voting Public Key
55    pub vote_key: Vec<u8>,
56    /// Full Payment Address (not hashed, 32 byte ED25519 Public key).
57    pub payment_address: Vec<u8>,
58    /// Is the stake address a script or not.
59    pub is_payable: bool,
60    /// Is the Registration CIP36 format, or CIP15
61    pub cip36: bool,
62}
63
64impl GetRegistrationQuery {
65    /// Prepares a get registration query.
66    pub(crate) async fn prepare(session: Arc<Session>) -> anyhow::Result<PreparedStatement> {
67        PreparedQueries::prepare(
68            session,
69            GET_REGISTRATIONS_FROM_STAKE_PK_QUERY,
70            scylla::statement::Consistency::All,
71            true,
72        )
73        .await
74        .inspect_err(|error| error!(error=%error, "Failed to prepare get registration from stake address query."))
75        .map_err(|error| anyhow::anyhow!("{error}\n--\n{GET_REGISTRATIONS_FROM_STAKE_PK_QUERY}"))
76    }
77
78    /// Executes get registration info for given stake addr query.
79    pub(crate) async fn execute(
80        session: &CassandraSession, params: GetRegistrationParams,
81    ) -> anyhow::Result<TypedRowStream<GetRegistrationQuery>> {
82        let iter = session
83            .execute_iter(PreparedSelectQuery::RegistrationFromStakeAddr, params)
84            .await?
85            .rows_stream::<GetRegistrationQuery>()?;
86
87        Ok(iter)
88    }
89}
