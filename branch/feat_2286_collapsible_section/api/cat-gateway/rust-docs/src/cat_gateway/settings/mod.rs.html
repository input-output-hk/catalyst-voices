157// Lazy initialization of all env vars which are not command line parameters.
158// All env vars used by the application should be listed here and all should have a
159// default. The default for all NON Secret values should be suitable for Production, and
160// NOT development. Secrets however should only be used with the default value in
161// development
162
163/// Handle to the mithril sync thread. One for each Network ONLY.
164static ENV_VARS: LazyLock<EnvVars> = LazyLock::new(|| {
165    // Support env vars in a `.env` file,  doesn't need to exist.
166    dotenv().ok();
167
168    let address = StringEnvVar::new("ADDRESS", ADDRESS_DEFAULT.to_string().into());
169    let address = SocketAddr::from_str(address.as_str())
170        .inspect(|err| {
171            error!(
172                "Invalid binding address {}, err: {err}. Using default binding address value {ADDRESS_DEFAULT}.",
173                address.as_str(),
174            );
175        }).unwrap_or(ADDRESS_DEFAULT);
176
177    let purge_backward_slot_buffer = StringEnvVar::new_as_int(
178        "PURGE_BACKWARD_SLOT_BUFFER",
179        PURGE_BACKWARD_SLOT_BUFFER_DEFAULT,
180        0,
181        u64::MAX,
182    );
183
184    EnvVars {
185        github_repo_owner: StringEnvVar::new("GITHUB_REPO_OWNER", GITHUB_REPO_OWNER_DEFAULT.into()),
186        github_repo_name: StringEnvVar::new("GITHUB_REPO_NAME", GITHUB_REPO_NAME_DEFAULT.into()),
187        github_issue_template: StringEnvVar::new(

193        service_id: StringEnvVar::new("SERVICE_ID", calculate_service_uuid().into()),

237    /// Validate env vars in ways we couldn't when they were first loaded.
238    pub(crate) fn validate() -> anyhow::Result<()> {
239        let mut status = Ok(());
240
241        let url = ENV_VARS.event_db.url.as_str();
242        if let Err(error) = tokio_postgres::config::Config::from_str(url) {
243            error!(error=%error, url=url, "Invalid Postgres DB URL.");
244            status = Err(anyhow!("Environment Variable Validation Error."));
245        }
246
247        status
248    }
249}

262        if SERVICE_SETTINGS.set(settings).is_err() {
263            // We use println here, because logger not yet configured.

265        }
266
267        // Init the logger.
268        logger::init(log_level);

270        log_build_info();
271
272        // Validate any settings we couldn't validate when loaded.
273        EnvVars::validate()
274    }

394    /// Generate a github issue url with a given title
395    ///
396    /// ## Arguments
397    ///
398    /// * `title`: &str - the title to give the issue
399    ///
400    /// ## Returns
401    ///
402    /// * String - the url
403    ///
404    /// ## Example
405    ///
406    /// ```rust,no_run
407    /// # use cat_data_service::settings::generate_github_issue_url;
408    /// assert_eq!(
409    ///     generate_github_issue_url("Hello, World! How are you?"),
410    ///     "https://github.com/input-output-hk/catalyst-voices/issues/new?template=bug_report.yml&title=Hello%2C%20World%21%20How%20are%20you%3F"
411    /// );
412    /// ```
413    pub(crate) fn generate_github_issue_url(title: &str) -> Option<Url> {
414        let path = format!(
415            "https://github.com/{}/{}/issues/new",
416            ENV_VARS.github_repo_owner.as_str(),
417            ENV_VARS.github_repo_name.as_str()
418        );
419
420        match Url::parse_with_params(&path, &[
421            ("template", ENV_VARS.github_issue_template.as_str()),
422            ("title", title),
423        ]) {
424            Ok(url) => Some(url),
425            Err(e) => {
426                error!("Failed to generate github issue url {:?}", e.to_string());
427                None
428            },
429        }
430    }

457/// Transform a string list of host names into a vec of host names.
458fn string_to_api_host_names(hosts: &str) -> Vec<String> {
459    /// Log an invalid hostname.
460    fn invalid_hostname(hostname: &str) -> String {
461        error!("Invalid host name for API: {}", hostname);
462        String::new()
463    }
464
465    let configured_hosts: Vec<String> = hosts
466        .split(',')
467        .map(|s| {
468            let url = Url::parse(s.trim());
469            match url {
470                Ok(url) => {
471                    // Get the scheme, and if its empty, use http
472                    let scheme = url.scheme();
473
474                    let port = url.port();
475
476                    // Rebuild the scheme + hostname
477                    match url.host() {
478                        Some(host) => {
479                            let host = host.to_string();
480                            if host.is_empty() {
481                                invalid_hostname(s)
482                            } else {
483                                match port {
484                                    Some(port) => {
485                                        format! {"{scheme}://{host}:{port}"}
486                                        // scheme.to_owned() + "://" + &host + ":" +
487                                        // &port.to_string()
488                                    },
489                                    None => {
490                                        format! {"{scheme}://{host}"}
491                                    },
492                                }
493                            }
494                        },
495                        None => invalid_hostname(s),
496                    }
497                },
498                Err(_) => invalid_hostname(s),
499            }
500        })
501        .filter(|s| !s.is_empty())
502        .collect();
503
504    configured_hosts
505}
