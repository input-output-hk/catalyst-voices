3use std::fmt::{Display, Formatter};

5use cardano_blockchain_types::TransactionId;

10        value::SerializeValue,

16/// A `TransactionHash` wrapper that can be stored to and load from a database.
17#[allow(clippy::module_name_repetitions)]
18#[derive(Debug, Copy, Clone, PartialEq, Eq, PartialOrd, Ord, Hash)]
19pub struct DbTransactionId(TransactionId);
20
21impl Display for DbTransactionId {
22    fn fmt(&self, f: &mut Formatter<'_>) -> std::fmt::Result {
23        f.write_str(&format!("{}", self.0))
24    }
25}
26
27impl From<TransactionId> for DbTransactionId {
28    fn from(value: TransactionId) -> Self {
29        Self(value)
30    }
31}
32
33impl From<DbTransactionId> for TransactionId {
34    fn from(value: DbTransactionId) -> Self {
35        value.0
36    }
37}
38
39impl SerializeValue for DbTransactionId {
40    fn serialize<'b>(
41        &self, typ: &ColumnType, writer: CellWriter<'b>,
42    ) -> Result<WrittenCellProof<'b>, SerializationError> {
43        let bytes: Vec<u8> = self.0.into();
44        bytes.serialize(typ, writer)
45    }
46}
47
48impl<'frame, 'metadata> DeserializeValue<'frame, 'metadata> for DbTransactionId {
49    fn type_check(typ: &ColumnType) -> Result<(), TypeCheckError> {
50        <Vec<u8>>::type_check(typ)
51    }
52
53    fn deserialize(
54        typ: &'metadata ColumnType<'metadata>, v: Option<FrameSlice<'frame>>,
55    ) -> Result<Self, DeserializationError> {
56        let bytes = <Vec<u8>>::deserialize(typ, v)?;
57        let hash = TransactionId::try_from(bytes).map_err(DeserializationError::new)?;
58        Ok(Self(hash))
59    }
60}
