1//! Implementation of the GET `/rbac/chain_root` endpoint.
2use anyhow::anyhow;
3use der_parser::asn1_rs::ToDer;
4use futures::StreamExt;
5use poem_openapi::{payload::Json, ApiResponse, Object};
6use tracing::error;
7
8use crate::{
9    db::index::{
10        queries::rbac::get_chain_root::{GetChainRootQuery, GetChainRootQueryParams},
11        session::CassandraSession,
12    },
13    service::common::{
14        responses::WithErrorResponses,
15        types::{
16            cardano::cip19_stake_address::Cip19StakeAddress, headers::retry_after::RetryAfterOption,
17        },
18    },
19};
20
21/// GET RBAC chain root response.
22#[derive(Object)]
23pub(crate) struct Response {
24    /// RBAC certificate chain root.
25    #[oai(validator(max_length = 66, min_length = 64, pattern = "0x[0-9a-f]{64}"))]
26    chain_root: String,
27}
28
29/// Endpoint responses.
30#[derive(ApiResponse)]
31pub(crate) enum Responses {
32    /// ## Ok
33    ///
34    /// Success returns the chain root hash.
35    #[oai(status = 200)]
36    Ok(Json<Response>),
37    /// ## Not Found
38    ///
39    /// No chain root found for the given stake address.
40    #[oai(status = 404)]
41    NotFound,
42}
43
44pub(crate) type AllResponses = WithErrorResponses<Responses>;
45
46/// Get chain root endpoint.
47pub(crate) async fn endpoint(stake_address: Cip19StakeAddress) -> AllResponses {
48    let Some(session) = CassandraSession::get(true) else {
49        error!("Failed to acquire db session");
50        let err = anyhow::anyhow!("Failed to acquire db session");
51        return AllResponses::service_unavailable(&err, RetryAfterOption::Default);
52    };
53
54    let Ok(stake_address) = stake_address.to_der_vec() else {
55        error!("Failed to create stake address vec");
56        let err = anyhow!("Failed to create stake address vec");
57        return AllResponses::internal_error(&err);
58    };
59
60    let query_res =
61        GetChainRootQuery::execute(&session, GetChainRootQueryParams { stake_address }).await;
62
63    match query_res {
64        Ok(mut row_iter) => {
65            if let Some(row_res) = row_iter.next().await {
66                let row = match row_res {
67                    Ok(row) => row,
68                    Err(err) => {
69                        error!(error = ?err, "Failed to parse get chain root by stake address query row");
70                        let err = anyhow!(err);
71                        return AllResponses::internal_error(&err);
72                    },
73                };
74
75                let res = Response {
76                    chain_root: format!("0x{}", hex::encode(row.chain_root)),
77                };
78
79                Responses::Ok(Json(res)).into()
80            } else {
81                Responses::NotFound.into()
82            }
83        },
84        Err(err) => {
85            error!(error = ?err, "Failed to execute get chain root by stake address query");
86            let err = anyhow!(err);
87            AllResponses::internal_error(&err)
88        },
89    }
90}
