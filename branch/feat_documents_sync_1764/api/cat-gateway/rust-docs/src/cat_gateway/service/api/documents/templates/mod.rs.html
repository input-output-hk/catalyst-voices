19/// A map of signed document templates to its ID.
20pub(crate) static TEMPLATES: LazyLock<Option<HashMap<Uuid, CatalystSignedDocument>>> =
21    LazyLock::new(|| {
22        let sk = load_doc_signing_key()?;
23
24        let mut map = HashMap::new();
25        map.extend(
26            CATEGORY_DOCUMENTS
27                .into_iter()
28                .map(|t| build_signed_doc(&t.into(), &sk)),
29        );
30        map.extend(
31            PROPOSAL_TEMPLATES
32                .into_iter()
33                .map(|t| build_signed_doc(&t.into(), &sk)),
34        );
35        map.extend(
36            COMMENT_TEMPLATES
37                .into_iter()
38                .map(|t| build_signed_doc(&t.into(), &sk)),
39        );
40        Some(map)
41    });

43/// Loads a Signing Key which is used to sign hardcoded Catalyst Signed Document objects
44fn load_doc_signing_key() -> Option<SigningKey> {
45    // Load the secret key from the environment
46    // Expect hex encoded secret key that start with 0x
47    // If no secret key is found, cannot generate sign documents
48    let Ok(signed_doc_sk_hex) = env::var("SIGNED_DOC_SK") else {
49        tracing::error!(
50            id = "static_template_env",
51            "Secret key not found in environment"
52        );
53        return None;
54    };
55    // Strip the prefix and convert to 32 bytes array
56    let Some(byte_array) = signed_doc_sk_hex
57        .strip_prefix("0x")
58        .and_then(|s| Vec::from_hex(s).ok())
59        .and_then(|bytes| bytes.as_slice().try_into().ok())
60    else {
61        tracing::error!(
62            id = "static_template",
63            "Failed to convert secret key to bytes"
64        );
65        return None;
66    };
67
68    let sk = SigningKey::from_bytes(&byte_array);
69    Some(sk)
70}

72#[derive(Clone)]
73/// Signed document hardcoded data structure.
74pub(crate) struct SignedDocData {
75    /// ID.
76    id: &'static str,
77    /// Version.
78    ver: &'static str,
79    /// Document type.
80    doc_type: Uuid,
81    /// Content bytes.
82    content: &'static [u8],
83    /// Category ID.
84    category_id: Option<&'static str>,
85}
86
87/// Build a `CatalystSignedDocument` from the hardcoded `SignedDocData`, return a document
88/// with its id.
89/// This function should not fail, because template details are hardcoded.
90#[allow(clippy::expect_used)]
91fn build_signed_doc(data: &SignedDocData, sk: &SigningKey) -> (Uuid, CatalystSignedDocument) {
92    /// ID URI network.
93    const KID_NETWORK: &str = "cardano";
94
95    let metadata = serde_json::json!({
96        "alg": catalyst_signed_doc::Algorithm::EdDSA.to_string(),
97        "type": data.doc_type,
98        "id": data.id,
99        "ver": data.ver,
100        "category_id": data.category_id.map(|v| serde_json::json!({"id": v})),
101        "content-type": ContentType::Json.to_string(),
102        "content-encoding": ContentEncoding::Brotli.to_string(),
103        "campaign_id": {"id": CAMPAIGN_ID},
104        "brand_id":  {"id": BRAND_ID},
105    });
106
107    let kid = IdUri::new(KID_NETWORK, None, sk.verifying_key());
108    let doc = Builder::new()
109        .with_json_metadata(metadata.clone())
110        .expect("Failed to build Metadata from template")
111        .with_decoded_content(data.content.to_vec())
112        .add_signature(sk.to_bytes(), kid)
113        .expect("Failed to add signature for template")
114        .build();
115    let doc_id = doc
116        .doc_id()
117        .expect("Template document must have id field")
118        .uuid();
119    (doc_id, doc)
120}

122/// Get a static document template from ID and version.
123pub(crate) fn get_doc_static_template(document_id: &uuid::Uuid) -> Option<CatalystSignedDocument> {
124    TEMPLATES
125        .as_ref()
126        .and_then(|templates| templates.get(document_id))
127        .cloned()
128}
