1//! Metrics related to RBAC Registration Chain Caching analytics.
2
3/// All the related RBAC Registration Chain Caching reporting metrics to the Prometheus
4/// service are inside this module.
5pub(crate) mod reporter {
6    use std::sync::LazyLock;
7
8    use prometheus::{
9        register_counter_vec, register_histogram_vec, register_int_gauge_vec, CounterVec,
10        HistogramVec, IntGaugeVec,
11    };
12
13    /// Labels for the metrics.
14    const METRIC_LABELS: [&str; 3] = ["api_host_names", "service_id", "network"];
15
16    /// Total count of cache hits.
17    pub(crate) static CACHE_HIT: LazyLock<CounterVec> = LazyLock::new(|| {
18        register_counter_vec!("cache_hit", "Total count of cache hits", &METRIC_LABELS).unwrap()
19    });
20
21    /// Total count of cache misses.
22    pub(crate) static CACHE_MISS: LazyLock<CounterVec> = LazyLock::new(|| {
23        register_counter_vec!("cache_miss", "Total count of cache misses", &METRIC_LABELS).unwrap()
24    });
25
26    /// Total count of cache access attempts.
27    pub(crate) static CACHE_ACCESS: LazyLock<CounterVec> = LazyLock::new(|| {
28        register_counter_vec!(
29            "cache_access",
30            "Total count of cache access attempts",
31            &METRIC_LABELS
32        )
33        .unwrap()
34    });
35
36    /// Duration measured in milliseconds for accessing cache on a cache hit.
37    pub(crate) static LATENCY: LazyLock<HistogramVec> = LazyLock::new(|| {
38        register_histogram_vec!(
39            "latency",
40            "Duration measured in milliseconds for accessing cache on a cache hit",
41            &METRIC_LABELS
42        )
43        .unwrap()
44    });
45
46    /// Total space in bytes currently using for RBAC caching.
47    pub(crate) static CACHE_SIZE: LazyLock<IntGaugeVec> = LazyLock::new(|| {
48        register_int_gauge_vec!(
49            "cache_size",
50            "Total space in bytes currently using for RBAC caching",
51            &METRIC_LABELS
52        )
53        .unwrap()
54    });
55
56    /// Maximum space in bytes for RBAC caching.
57    pub(crate) static MAX_CACHE_SIZE: LazyLock<IntGaugeVec> = LazyLock::new(|| {
58        register_int_gauge_vec!(
59            "max_cache_size",
60            "Maximum space in bytes for RBAC caching",
61            &METRIC_LABELS
62        )
63        .unwrap()
64    });
65
66    /// Total time in milliseconds taken to set up RBAC caching service to be operational.
67    pub(crate) static START_UP_TIME: LazyLock<IntGaugeVec> = LazyLock::new(|| {
68        register_int_gauge_vec!(
69            "start_up_time",
70            "Total time in milliseconds taken to set up RBAC caching service to be operational",
71            &METRIC_LABELS
72        )
73        .unwrap()
74    });
75
76    /// Total number of RBAC registration chain entries being stored as cache.
77    pub(crate) static CACHING_RBAC_ENTRIES: LazyLock<IntGaugeVec> = LazyLock::new(|| {
78        register_int_gauge_vec!(
79            "caching_rbac_entries",
80            "Total number of RBAC registration chain entries being stored as cache",
81            &METRIC_LABELS
82        )
83        .unwrap()
84    });
85}
