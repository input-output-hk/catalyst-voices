1//! Index RBAC Chain Root For Stake Address Insert Query.
2use std::{fmt::Debug, sync::Arc};
3
4use scylla::{SerializeRow, Session};
5use tracing::error;
6
7use crate::{
8    db::index::queries::{PreparedQueries, SizedBatch},
9    settings::cassandra_db::EnvVars,
10};
11
12/// Index RBAC Chain Root by Stake Address
13const INSERT_CHAIN_ROOT_FOR_STAKE_ADDRESS_QUERY: &str =
14    include_str!("./cql/insert_chain_root_for_stake_address.cql");
15
16/// Insert Chain Root For Stake Address Query Parameters
17#[derive(SerializeRow)]
18pub(crate) struct Params {
19    /// Stake Address Hash. 32 bytes.
20    stake_addr: Vec<u8>,
21    /// Block Slot Number
22    slot_no: num_bigint::BigInt,
23    /// Transaction Offset inside the block.
24    txn: i16,
25    /// Chain Root Hash. 32 bytes.
26    chain_root: Vec<u8>,
27}
28
29impl Debug for Params {
30    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
31        f.debug_struct("Params")
32            .field("stake_addr", &self.stake_addr)
33            .field("slot_no", &self.slot_no)
34            .field("txn", &self.txn)
35            .field("chain_root", &self.chain_root)
36            .finish()
37    }
38}
39
40impl Params {
41    /// Create a new record for this transaction.
42    pub(crate) fn new(stake_addr: &[u8], chain_root: &[u8], slot_no: u64, txn: i16) -> Self {
43        Params {
44            stake_addr: stake_addr.to_vec(),
45            slot_no: num_bigint::BigInt::from(slot_no),
46            txn,
47            chain_root: chain_root.to_vec(),
48        }
49    }
50
51    /// Prepare Batch of RBAC Registration Index Data Queries
52    pub(crate) async fn prepare_batch(
53        session: &Arc<Session>, cfg: &EnvVars,
54    ) -> anyhow::Result<SizedBatch> {
55        PreparedQueries::prepare_batch(
56            session.clone(),
57            INSERT_CHAIN_ROOT_FOR_STAKE_ADDRESS_QUERY,
58            cfg,
59            scylla::statement::Consistency::Any,
60            true,
61            false,
62        )
63        .await
64        .inspect_err(|error| error!(error=%error,"Failed to prepare Insert Chain Root For Stake Address Query."))
65        .map_err(|error| anyhow::anyhow!("{error}\n--\n{INSERT_CHAIN_ROOT_FOR_STAKE_ADDRESS_QUERY}"))
66    }
67}
