1//! A middleware module wrapping around `prometheus::Registry`.
2//! It acts as an endpoint to export metrics data to the Prometheus service.
3//! For every request to this endpoint, it will call the `updater` function to update
4//! metrics to the latest before sending to the service.
5
6use poem::{
7    http::{Method, StatusCode},
8    Endpoint, Request, Response, Result,
9};
10use prometheus::{Encoder, Registry, TextEncoder};
11use tracing::error;
12
13/// The function type to call when a request sent to the endpoint.
14type UpdateFn = fn();
15
16/// A Middleware wrapping the Prometheus registry to report as metrics.
17///
18/// The middleware is originally from `poem::endpoint::PrometheusExporter`.
19pub struct MetricsUpdaterMiddleware {
20    /// The Prometheus registry.
21    registry: Registry,
22    /// The updater function, called for every request for this endpoint.
23    updater: UpdateFn,
24}
25
26impl MetricsUpdaterMiddleware {
27    /// Create a `PrometheusExporter` endpoint.
28    pub fn new(registry: Registry, updater: UpdateFn) -> Self {
29        Self { registry, updater }
30    }
31}
32
33impl Endpoint for MetricsUpdaterMiddleware {
34    type Output = Response;
35
36    async fn call(&self, req: Request) -> Result<Self::Output> {
37        if req.method() != Method::GET {
38            return Ok(StatusCode::METHOD_NOT_ALLOWED.into());
39        }
40
41        // perform updating as another task
42        let updater = self.updater.clone();
43        let handle = tokio::task::spawn(async move { (updater)() });
44
45        if let Err(err) = handle.await {
46            error!(?err, ?req, "Failed to perform metrics updating");
47        }
48
49        let encoder = TextEncoder::new();
50        let metric_families = self.registry.gather();
51        let mut result = Vec::new();
52        match encoder.encode(&metric_families, &mut result) {
53            Ok(()) => {
54                Ok(Response::builder()
55                    .content_type(encoder.format_type())
56                    .body(result))
57            },
58            Err(_) => Err(StatusCode::INTERNAL_SERVER_ERROR.into()),
59        }
60    }
61}
