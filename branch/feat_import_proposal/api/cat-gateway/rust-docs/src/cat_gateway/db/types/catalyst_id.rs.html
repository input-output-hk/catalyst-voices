1//! A `IdUri` wrapper that can be stored to and load from a database.
2
3use catalyst_types::id_uri::IdUri;
4use scylla::_macro_internal::{
5    CellWriter, ColumnType, DeserializationError, DeserializeValue, FrameSlice, SerializationError,
6    SerializeValue, TypeCheckError, WrittenCellProof,
7};
8
9/// A `IdUri` wrapper that can be stored to and load from a database.
10#[allow(clippy::module_name_repetitions)]
11#[derive(Debug, Clone, PartialEq, Hash)]
12pub struct DbCatalystId(IdUri);
13
14impl From<IdUri> for DbCatalystId {
15    fn from(value: IdUri) -> Self {
16        Self(value)
17    }
18}
19
20impl From<DbCatalystId> for IdUri {
21    fn from(value: DbCatalystId) -> Self {
22        value.0
23    }
24}
25
26impl SerializeValue for DbCatalystId {
27    fn serialize<'b>(
28        &self, typ: &ColumnType, writer: CellWriter<'b>,
29    ) -> Result<WrittenCellProof<'b>, SerializationError> {
30        self.0.as_short_id().to_string().serialize(typ, writer)
31    }
32}
33
34impl<'frame, 'metadata> DeserializeValue<'frame, 'metadata> for DbCatalystId {
35    fn type_check(typ: &ColumnType) -> Result<(), TypeCheckError> {
36        String::type_check(typ)
37    }
38
39    fn deserialize(
40        typ: &'metadata ColumnType<'metadata>, v: Option<FrameSlice<'frame>>,
41    ) -> Result<Self, DeserializationError> {
42        let id = String::deserialize(typ, v)?;
43        let id: IdUri = id.parse().map_err(DeserializationError::new)?;
44        Ok(Self(id))
45    }
46}
