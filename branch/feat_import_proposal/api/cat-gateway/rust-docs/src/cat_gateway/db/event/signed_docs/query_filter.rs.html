8/// A `select_signed_docs` query filtering argument.
9/// If all fields would be `None` the query will search for all entries from the db.
10#[derive(Clone, Debug, Default)]
11pub(crate) struct DocsQueryFilter {
12    /// `type` field
13    doc_type: Option<uuid::Uuid>,
14    /// `id` field
15    id: Option<EqOrRangedUuid>,
16    /// `ver` field
17    ver: Option<EqOrRangedUuid>,
18    /// `metadata->'ref'` field
19    doc_ref: Option<DocumentRef>,
20    /// `metadata->'template'` field
21    template: Option<DocumentRef>,
22    /// `metadata->'reply'` field
23    reply: Option<DocumentRef>,
24    /// `metadata->'brand_id'` field
25    brand_id: Option<DocumentRef>,
26    /// `metadata->'campaign_id'` field
27    campaign_id: Option<DocumentRef>,
28    /// `metadata->'category_id'` field
29    category_id: Option<DocumentRef>,
30}
31
32impl Display for DocsQueryFilter {
33    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
34        use std::fmt::Write;
35        let mut query = "TRUE".to_string();
36
37        if let Some(doc_type) = &self.doc_type {
38            write!(&mut query, " AND signed_docs.type = '{doc_type}'")?;
39        }
40
41        if let Some(id) = &self.id {
42            write!(&mut query, " AND {}", id.conditional_stmt("signed_docs.id"))?;
43        }
44        if let Some(ver) = &self.ver {
45            write!(
46                &mut query,
47                " AND {}",
48                ver.conditional_stmt("signed_docs.ver")
49            )?;
50        }
51        if let Some(doc_ref) = &self.doc_ref {
52            write!(
53                &mut query,
54                " AND {}",
55                doc_ref.conditional_stmt("metadata->'ref'")
56            )?;
57        }
58        if let Some(template) = &self.template {
59            write!(
60                &mut query,
61                " AND {}",
62                template.conditional_stmt("metadata->'template'")
63            )?;
64        }
65        if let Some(reply) = &self.reply {
66            write!(
67                &mut query,
68                " AND {}",
69                reply.conditional_stmt("metadata->'reply'")
70            )?;
71        }
72        if let Some(brand_id) = &self.brand_id {
73            write!(
74                &mut query,
75                " AND {}",
76                brand_id.conditional_stmt("metadata->'brand_id'")
77            )?;
78        }
79        if let Some(campaign_id) = &self.campaign_id {
80            write!(
81                &mut query,
82                " AND {}",
83                campaign_id.conditional_stmt("metadata->'campaign_id'")
84            )?;
85        }
86        if let Some(category_id) = &self.category_id {
87            write!(
88                &mut query,
89                " AND {}",
90                category_id.conditional_stmt("metadata->'category_id'")
91            )?;
92        }
93
94        write!(f, "{query}")
95    }
96}
