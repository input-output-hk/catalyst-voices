28/// A map of signed document templates to its ID.
29pub(crate) static TEMPLATES: LazyLock<Option<HashMap<Uuid, CatalystSignedDocument>>> =
30    LazyLock::new(|| {
31        // Load the secret key from the environment
32        // Expect hex encoded secret key that start with 0x
33        let value = env::var("SIGNED_DOC_SK");
34
35        // If no secret key is found, cannot generate sign documents
36        if let Ok(v) = value {
37            // Strip the prefix and convert to 32 bytes array
38            let byte_array: [u8; 32] = if let Some(bytes) = v
39                .strip_prefix("0x")
40                .and_then(|s| Vec::from_hex(s).ok())
41                .and_then(|bytes| bytes.as_slice().try_into().ok())
42            {
43                bytes
44            } else {
45                error!(
46                    id = "static_template",
47                    "Failed to convert secret key to bytes"
48                );
49                return None;
50            };
51            let sk = SigningKey::from_bytes(&byte_array);
52
53            let mut map = HashMap::new();
54            for t in TEMPLATE_DATA.iter() {
55                let signed_doc = t.to_signed_doc(&sk);
56                map.insert(signed_doc.doc_id().uuid(), signed_doc);
57            }
58            Some(map)
59        } else {
60            error!(
61                id = "static_template_env",
62                "Secret key not found in environment"
63            );
64            None
65        }
66    });

68#[derive(Clone)]
69/// Signed document template.
70pub(crate) struct SignedDocTemplate {
71    /// ID.
72    id: String,
73    /// Version.
74    ver: String,
75    /// Document type.
76    doc_type: String,
77    /// Content bytes.
78    content: Vec<u8>,
79    /// Category ID.
80    category_id: String,
81}
82
83impl SignedDocTemplate {
84    /// Convert the template to a catalyst signed document.
85    /// This function should not fail, because template details are hardcoded.
86    #[allow(clippy::expect_used)]
87    fn to_signed_doc(&self, sk: &SigningKey) -> CatalystSignedDocument {
88        /// ID URI network.
89        const KID_NETWORK: &str = "cardano";
90
91        let metadata: Metadata = serde_json::from_value(serde_json::json!({
92            "type": self.doc_type,
93            "id": self.id,
94            "ver": self.ver,
95            "category_id": {"id": self.category_id},
96            "content-type": CONTENT_TYPE,
97            "content-encoding": CONTENT_ENCODING,
98            "campaign_id": {"id": CAMPAIGN_ID},
99            "brand_id":  {"id": BRAND_ID},
100        }))
101        .expect("Failed to build Metadata from template");
102
103        let kid = IdUri::new(KID_NETWORK, None, sk.verifying_key());
104        Builder::new()
105            .with_metadata(metadata.clone())
106            .with_decoded_content(self.content.clone())
107            .add_signature(sk.to_bytes(), kid)
108            .expect("Failed to add signature for template")
109            .build()
110            .expect("Failed to build CatalystSignedDocument from template")
111    }
112}

114/// Get a static document template from ID and version.
115pub(crate) fn get_doc_static_template(document_id: uuid::Uuid) -> Option<Vec<u8>> {
116    let data = TEMPLATES
117        .as_ref()
118        .and_then(|templates| templates.get(&document_id));
119
120    if let Some(template) = data {
121        let mut buffer = Vec::new();
122        let mut encoder = Encoder::new(&mut buffer);
123        // Cannot encode the template, return None
124        // Should not happen
125        template.encode(&mut encoder, &mut ()).ok()?;
126        return Some(buffer);
127    }
128    None
129}
