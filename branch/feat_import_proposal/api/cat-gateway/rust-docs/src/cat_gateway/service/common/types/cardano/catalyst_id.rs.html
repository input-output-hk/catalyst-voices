15/// A schema.
16static SCHEMA: LazyLock<MetaSchema> = LazyLock::new(|| {
17    let example = Some(CatalystId::example().0.to_string().into());
18    MetaSchema {
19        title: Some("Catalyst short ID".into()),
20        description: Some("Catalyst short identifier in string format"),
21        example,
22        pattern: Some(r".+\..+\/[A-Za-z0-9_-]{43}".into()),
23        ..MetaSchema::ANY
24    }
25});

27/// A Catalyst short identifier.
28#[derive(Debug, Clone, PartialEq, Hash)]
29pub(crate) struct CatalystId(IdUri);

31impl Type for CatalystId {
32    type RawElementValueType = IdUri;
33    type RawValueType = IdUri;
34
35    const IS_REQUIRED: bool = true;

43            "string",

72impl ParseFromJSON for CatalystId {
73    fn parse_from_json(value: Option<Value>) -> ParseResult<Self> {
74        let value = value.unwrap_or_default();
75        if let Value::String(value) = value {
76            CatalystId::parse_from_parameter(&value)
77        } else {
78            Err(ParseError::expected_type(value))
79        }
80    }
81}
82
83impl ParseFromParameter for CatalystId {
84    fn parse_from_parameter(value: &str) -> ParseResult<Self> {
85        let id: IdUri = match value.parse() {
86            Ok(v) => v,
87            Err(e) => return Err(format!("Invalid Catalyst ID: {e:?}").into()),
88        };
89        Ok(Self(id.as_short_id()))
90    }
91}
92
93impl ToJSON for CatalystId {

96    }
97}
