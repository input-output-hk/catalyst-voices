19/// Title.
20const TITLE: &str = "Error Message";
21/// Description.
22const DESCRIPTION: &str = "This is an error message.";
23/// Example.
24const EXAMPLE: &str = "An error has occurred, the details of the error are ...";
25/// Max Length
26const MAX_LENGTH: usize = 256;
27/// Min Length
28const MIN_LENGTH: usize = 1;
29/// Validation Regex Pattern
30const PATTERN: &str = concatcp!("^(.){", MIN_LENGTH, ",", MAX_LENGTH, "}$");
31
32/// Schema
33static SCHEMA: LazyLock<MetaSchema> = LazyLock::new(|| {
34    MetaSchema {
35        title: Some(TITLE.to_owned()),
36        description: Some(DESCRIPTION),
37        example: Some(Value::String(EXAMPLE.to_string())),
38        max_length: Some(MAX_LENGTH),
39        min_length: Some(MIN_LENGTH),
40        pattern: Some(PATTERN.to_string()),
41        ..poem_openapi::registry::MetaSchema::ANY
42    }
43});
44
45/// Check if we match the regex.
46fn is_valid(msg: &str) -> bool {
47    /// Validation pattern
48    #[allow(clippy::unwrap_used)] // Safe because the Regex is constant.  Can never panic in prod.
49    static RE: LazyLock<Regex> = LazyLock::new(|| Regex::new(PATTERN).unwrap());
50
51    RE.is_match(msg)
52}
53
54impl_string_types!(
55    ErrorMessage,
56    "string",
57    "error",
58    Some(SCHEMA.clone()),
59    is_valid
60);
61
62impl Example for ErrorMessage {
63    /// An example 32 bytes ED25519 Public Key.
64    fn example() -> Self {
65        Self(EXAMPLE.to_owned())
66    }
67}

72    }
