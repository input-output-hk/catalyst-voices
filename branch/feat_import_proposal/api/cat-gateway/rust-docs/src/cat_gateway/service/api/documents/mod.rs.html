35/// Cardano Follower API Endpoints
36pub(crate) struct DocumentApi;
37
38#[OpenApi(tag = "ApiTags::Documents")]
39impl DocumentApi {
40    /// Get A Signed Document.
41    ///
42    /// This endpoint returns either a specific or latest version of a registered signed
43    /// document.
44    #[oai(
45        path = "/v1/document/:document_id",
46        method = "get",
47        operation_id = "getDocument",
48        transform = "schema_version_validation"
49    )]
50    async fn get_document(
51        &self, /// UUIDv7 Document ID to retrieve
52        document_id: Path<UUIDv7>,
53        /// UUIDv7 Version of the Document to retrieve, if omitted, returns the latest
54        /// version.
55        version: Query<Option<UUIDv7>>,
56        /// No Authorization required, but Token permitted.
57        _auth: CatalystRBACSecurityScheme,
58    ) -> get_document::AllResponses {
59        let Ok(doc_id) = document_id.0.try_into() else {
60            let err = anyhow!("Invalid UUIDv7"); // Should not happen as UUIDv7 is validating.
61            return get_document::AllResponses::internal_error(&err);
62        };
63        let Ok(ver_id) = version.0.map(std::convert::TryInto::try_into).transpose() else {
64            let err = anyhow!("Invalid UUIDv7"); // Should not happen as UUIDv7 is validating.
65            return get_document::AllResponses::internal_error(&err);
66        };
67        get_document::endpoint(doc_id, ver_id).await
68    }
69
70    /// Put A Signed Document.
71    ///
72    /// This endpoint returns OK if the document is valid, able to be put by the
73    /// submitter, and if it already exists, is identical to the existing document.
74    #[oai(
75        path = "/v1/document",
76        method = "put",
77        operation_id = "putDocument",
78        transform = "schema_version_validation"
79    )]
80    async fn put_document(
81        &self, /// The document to PUT
82        document: Cbor<Body>,
83        /// Authorization required.
84        _auth: CatalystRBACSecurityScheme,
85    ) -> put_document::AllResponses {
86        match document.0.into_bytes_limit(MAXIMUM_DOCUMENT_SIZE).await {
87            Ok(doc_bytes) => put_document::endpoint(doc_bytes.to_vec()).await,
88            Err(ReadBodyError::PayloadTooLarge) => put_document::Responses::PayloadTooLarge.into(),
89            Err(_) => {
90                put_document::Responses::UnprocessableContent(Json(
91                    PutDocumentUnprocessableContent::new(
92                        "Failed to read document from the request",
93                        None,
94                    ),
95                ))
96                .into()
97            },
98        }
99    }
100
101    /// Post A Signed Document Index Query.
102    ///
103    /// This endpoint produces a summary of signed documents that meet the criteria
104    /// defined in the request body.
105    ///
106    /// It does not return the actual documents, just an index of the document identifiers
107    /// which allows the documents to be retrieved by the `GET document` endpoint.
108    #[oai(
109        path = "/v1/document/index",
110        method = "post",
111        operation_id = "postDocument",
112        transform = "schema_version_validation"
113    )]
114    async fn post_document(
115        &self, /// The Query Filter Specification
116        query: Json<DocumentIndexQueryFilterBody>,
117        page: Query<Option<Page>>, limit: Query<Option<Limit>>,
118        /// Authorization required.
119        _auth: CatalystRBACSecurityScheme,
120    ) -> post_document_index_query::AllResponses {
121        post_document_index_query::endpoint(query.0 .0, page.0, limit.0).await
122    }
123}
