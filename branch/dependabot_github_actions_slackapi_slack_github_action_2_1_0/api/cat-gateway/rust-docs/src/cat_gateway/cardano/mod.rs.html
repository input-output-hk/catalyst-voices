35/// Start syncing a particular network
36async fn start_sync_for(cfg: &chain_follower::EnvVars) -> anyhow::Result<()> {
37    let chain = cfg.chain;
38    let dl_config = cfg.dl_config.clone();
39
40    let mut cfg = ChainSyncConfig::default_for(chain);
41    cfg.mithril_cfg = cfg.mithril_cfg.with_dl_config(dl_config);
42    info!(chain = %chain, "Starting Blockchain Sync");
43
44    if let Err(error) = cfg.run().await {
45        error!(chain=%chain, error=%error, "Failed to start chain sync task");
46        Err(error)?;
47    }
48
49    Ok(())
50}

52/// Data we return from a sync task.
53#[derive(Clone)]
54struct SyncParams {
55    /// What blockchain are we syncing.
56    chain: Network,
57    /// The starting point of this sync.
58    start: Point,
59    /// The ending point of this sync.
60    end: Point,
61    /// The first block we successfully synced.
62    first_indexed_block: Option<Point>,
63    /// Is the starting point immutable? (True = immutable, false = don't know.)
64    first_is_immutable: bool,
65    /// The last block we successfully synced.
66    last_indexed_block: Option<Point>,
67    /// Is the ending point immutable? (True = immutable, false = don't know.)
68    last_is_immutable: bool,
69    /// The number of blocks we successfully synced overall.
70    total_blocks_synced: u64,
71    /// The number of blocks we successfully synced, in the last attempt.
72    last_blocks_synced: u64,
73    /// The number of retries so far on this sync task.
74    retries: u64,
75    /// The number of retries so far on this sync task.
76    backoff_delay: Option<Duration>,
77    /// If the sync completed without error or not.
78    result: Arc<Option<anyhow::Result<()>>>,
79    /// Chain follower roll forward.
80    follower_roll_forward: Option<Point>,
81}

83impl Display for SyncParams {
84    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
85        if self.result.is_none() {
86            write!(f, "Sync_Params {{ ")?;
87        } else {
88            write!(f, "Sync_Result {{ ")?;
89        }
90
91        write!(f, "start: {}, end: {}", self.start, self.end)?;
92
93        if let Some(first) = self.first_indexed_block.as_ref() {
94            write!(
95                f,
96                ", first_indexed_block: {first}{}",
97                if self.first_is_immutable { ":I" } else { "" }
98            )?;
99        }
100
101        if let Some(last) = self.last_indexed_block.as_ref() {
102            write!(
103                f,
104                ", last_indexed_block: {last}{}",
105                if self.last_is_immutable { ":I" } else { "" }
106            )?;
107        }
108
109        if self.retries > 0 {
110            write!(f, ", retries: {}", self.retries)?;
111        }
112
113        if self.retries > 0 || self.result.is_some() {
114            write!(f, ", synced_blocks: {}", self.total_blocks_synced)?;
115        }
116
117        if self.result.is_some() {
118            write!(f, ", last_sync: {}", self.last_blocks_synced)?;
119        }
120
121        if let Some(backoff) = self.backoff_delay.as_ref() {
122            write!(f, ", backoff: {}", DurationString::from(*backoff))?;
123        }
124
125        if let Some(result) = self.result.as_ref() {
126            match result {
127                Ok(()) => write!(f, ", Success")?,
128                Err(error) => write!(f, ", {error}")?,
129            };
130        }
131
132        f.write_str(" }")
133    }
134}

136/// The range we generate random backoffs within given a base backoff value.
137const BACKOFF_RANGE_MULTIPLIER: u32 = 3;
138
139impl SyncParams {
140    /// Create a new `SyncParams`.
141    fn new(chain: Network, start: Point, end: Point) -> Self {
142        Self {
143            chain,
144            start,
145            end,
146            first_indexed_block: None,
147            first_is_immutable: false,
148            last_indexed_block: None,
149            last_is_immutable: false,
150            total_blocks_synced: 0,
151            last_blocks_synced: 0,
152            retries: 0,
153            backoff_delay: None,
154            result: Arc::new(None),
155            follower_roll_forward: None,
156        }
157    }
158
159    /// Convert a result back into parameters for a retry.
160    fn retry(&self) -> Self {
161        let retry_count = self.retries.saturating_add(1);
162
163        let mut backoff = None;
164
165        // If we did sync any blocks last time, first retry is immediate.
166        // Otherwise we backoff progressively more as we do more retries.
167        if self.last_blocks_synced == 0 {
168            // Calculate backoff based on number of retries so far.
169            backoff = match retry_count {
170                1 => Some(Duration::from_secs(1)),     // 1-3 seconds
171                2..5 => Some(Duration::from_secs(10)), // 10-30 seconds
172                _ => Some(Duration::from_secs(30)),    // 30-90 seconds.
173            };
174        }
175
176        let mut retry = self.clone();
177        retry.last_blocks_synced = 0;
178        retry.retries = retry_count;
179        retry.backoff_delay = backoff;
180        retry.result = Arc::new(None);
181        retry.follower_roll_forward = None;
182
183        retry
184    }
185
186    /// Convert Params into the result of the sync.
187    pub(crate) fn done(
188        &self, first: Option<Point>, first_immutable: bool, last: Option<Point>,
189        last_immutable: bool, synced: u64, result: anyhow::Result<()>,
190    ) -> Self {
191        if result.is_ok() && self.end != Point::TIP {
192            // Update sync status in the Immutable DB.
193            // Can fire and forget, because failure to update DB will simply cause the chunk to be
194            // re-indexed, on recovery.
195            update_sync_status(self.end.slot_or_default(), self.start.slot_or_default());
196        }
197        let mut done = self.clone();
198        done.first_indexed_block = first;
199        done.first_is_immutable = first_immutable;
200        done.last_indexed_block = last;
201        done.last_is_immutable = last_immutable;
202        done.total_blocks_synced = done.total_blocks_synced.saturating_add(synced);
203        done.last_blocks_synced = synced;
204        done.result = Arc::new(Some(result));
205
206        done
207    }
208
209    /// Get where this sync run actually needs to start from.
210    fn actual_start(&self) -> Point {
211        self.last_indexed_block
212            .as_ref()
213            .unwrap_or(&self.start)
214            .clone()
215    }
216
217    /// Do the backoff delay processing.
218    ///
219    /// The actual delay is a random time from the Delay itself to
220    /// `BACKOFF_RANGE_MULTIPLIER` times the delay. This is to prevent hammering the
221    /// service at regular intervals.
222    async fn backoff(&self) {
223        if let Some(backoff) = self.backoff_delay {
224            let mut rng = rand::rngs::StdRng::from_entropy();
225            let actual_backoff =
226                rng.gen_range(backoff..backoff.saturating_mul(BACKOFF_RANGE_MULTIPLIER));
227
228            tokio::time::sleep(actual_backoff).await;
229        }
230    }
231}

233/// Sync a portion of the blockchain.
234/// Set end to `Point::TIP` to sync the tip continuously.
235#[allow(clippy::too_many_lines)]
236fn sync_subchain(
237    params: SyncParams, event_sender: broadcast::Sender<event::ChainIndexerEvent>,
238) -> tokio::task::JoinHandle<SyncParams> {
239    tokio::spawn(async move {
240        info!(chain = %params.chain, params=%params, "Indexing Blockchain");
241
242        // Backoff hitting the database if we need to.
243        params.backoff().await;
244
245        // Wait for indexing DB to be ready before continuing.
246        drop(CassandraSession::wait_until_ready(INDEXING_DB_READY_WAIT_INTERVAL, true).await);
247        info!(chain=%params.chain, params=%params,"Indexing DB is ready");
248
249        let mut first_indexed_block = params.first_indexed_block.clone();
250        let mut first_immutable = params.first_is_immutable;
251        let mut last_indexed_block = params.last_indexed_block.clone();
252        let mut last_immutable = params.last_is_immutable;
253        let mut blocks_synced = 0u64;
254
255        let mut follower =
256            ChainFollower::new(params.chain, params.actual_start(), params.end.clone()).await;
257        while let Some(chain_update) = follower.next().await {
258            match chain_update.kind {
259                cardano_chain_follower::Kind::ImmutableBlockRollForward => {
260                    // We only process these on the follower tracking the TIP.
261                    if params.end == Point::TIP {
262                        // What we need to do here is tell the primary follower to start a new sync
263                        // for the new immutable data, and then purge the volatile database of the
264                        // old data (after the immutable data has synced).
265                        info!(chain=%params.chain, point=?chain_update.block_data().point(), "Immutable chain rolled forward.");
266                        let mut result = params.done(
267                            first_indexed_block,
268                            first_immutable,
269                            last_indexed_block,
270                            last_immutable,
271                            blocks_synced,
272                            Ok(()),
273                        );
274                        // Signal the point the immutable chain rolled forward to.
275                        // If this is live chain immediately stops to later run immutable sync tasks
276                        result.follower_roll_forward = Some(chain_update.block_data().point());
277                        return result;
278                    }
279                },
280                cardano_chain_follower::Kind::Block => {
281                    let block = chain_update.block_data();
282
283                    if let Err(error) = index_block(block).await {
284                        let error_msg = format!("Failed to index block {}", block.point());
285                        error!(chain=%params.chain, error=%error, params=%params, error_msg);
286                        return params.done(
287                            first_indexed_block,
288                            first_immutable,
289                            last_indexed_block,
290                            last_immutable,
291                            blocks_synced,
292                            Err(error.context(error_msg)),
293                        );
294                    }
295
296                    // Update flag if this is the first time reaching TIP.
297                    if chain_update.tip && !follower_has_first_reached_tip() {
298                        set_follower_first_reached_tip();
299                    }
300
301                    update_block_state(
302                        block,
303                        &mut first_indexed_block,
304                        &mut first_immutable,
305                        &mut last_indexed_block,
306                        &mut last_immutable,
307                        &mut blocks_synced,
308                    );
309                },
310
311                cardano_chain_follower::Kind::Rollback => {
312                    // Rollback occurs, need to purge forward
313                    let rollback_slot = chain_update.block_data().slot();
314
315                    let purge_condition = PurgeCondition::PurgeForwards(rollback_slot);
316                    if let Err(error) = roll_forward::purge_live_index(purge_condition).await {
317                        error!(chain=%params.chain, error=%error,
318                            "Chain follower rollback, purging volatile data task failed."
319                        );
320                    } else {
321                        // How many slots are purged
322                        #[allow(clippy::arithmetic_side_effects)]
323                        let purge_slots = params
324                            .last_indexed_block
325                            .as_ref()
326                            // Slots arithmetic has saturating semantic, so this is ok
327                            .map_or(0.into(), |l| l.slot_or_default() - rollback_slot);
328
329                        let _ = event_sender.send(event::ChainIndexerEvent::ForwardDataPurged {
330                            purge_slots: purge_slots.into(),
331                        });
332
333                        // Purge success, now index the current block
334                        let block = chain_update.block_data();
335                        if let Err(error) = index_block(block).await {
336                            let error_msg =
337                                format!("Failed to index block after rollback {}", block.point());
338                            error!(chain=%params.chain, error=%error, params=%params, error_msg);
339                            return params.done(
340                                first_indexed_block,
341                                first_immutable,
342                                last_indexed_block,
343                                last_immutable,
344                                blocks_synced,
345                                Err(error.context(error_msg)),
346                            );
347                        }
348
349                        update_block_state(
350                            block,
351                            &mut first_indexed_block,
352                            &mut first_immutable,
353                            &mut last_indexed_block,
354                            &mut last_immutable,
355                            &mut blocks_synced,
356                        );
357                    }
358                },
359            }
360        }
361
362        let result = params.done(
363            first_indexed_block,
364            first_immutable,
365            last_indexed_block,
366            last_immutable,
367            blocks_synced,
368            Ok(()),
369        );
370
371        info!(chain = %result.chain, result=%result, "Indexing Blockchain Completed: OK");
372
373        result
374    })
375}

377/// Update block related state.
378fn update_block_state(
379    block: &MultiEraBlock, first_indexed_block: &mut Option<Point>, first_immutable: &mut bool,
380    last_indexed_block: &mut Option<Point>, last_immutable: &mut bool, blocks_synced: &mut u64,
381) {
382    *last_immutable = block.is_immutable();
383    *last_indexed_block = Some(block.point());
384
385    if first_indexed_block.is_none() {
386        *first_immutable = *last_immutable;
387        *first_indexed_block = Some(block.point());
388    }
389
390    *blocks_synced = blocks_synced.saturating_add(1);
391}

393/// The synchronisation task, and its state.
394/// There should ONLY ever be one of these at any time.
395struct SyncTask {
396    /// Chain follower configuration.
397    cfg: chain_follower::EnvVars,
398
399    /// The current running sync tasks.
400    sync_tasks: FuturesUnordered<tokio::task::JoinHandle<SyncParams>>,
401
402    /// How many immutable chain follower sync tasks we are running.
403    current_sync_tasks: u16,
404
405    /// Start for the next block we would sync.
406    start_slot: Slot,
407
408    /// The immutable tip slot.
409    immutable_tip_slot: Slot,
410
411    /// The live tip slot.
412    live_tip_slot: Slot,
413
414    /// Current Sync Status.
415    sync_status: Vec<SyncStatus>,
416
417    /// Event sender during the process of sync tasks.
418    event_channel: (
419        broadcast::Sender<event::ChainIndexerEvent>,
420        broadcast::Receiver<event::ChainIndexerEvent>,
421    ),
422}
423
424impl SyncTask {
425    /// Create a new `SyncTask`.
426    fn new(cfg: chain_follower::EnvVars) -> SyncTask {
427        Self {
428            cfg,
429            sync_tasks: FuturesUnordered::new(),
430            start_slot: 0.into(),
431            current_sync_tasks: 0,
432            immutable_tip_slot: 0.into(),
433            live_tip_slot: 0.into(),
434            sync_status: Vec::new(),
435            event_channel: broadcast::channel(10),
436        }
437    }
438
439    /// Primary Chain Follower task.
440    ///
441    /// This continuously runs in the background, and never terminates.
442    ///
443    /// Sets the Index DB liveness flag to true if it is not already set.
444    ///
445    /// Sets the Chain Follower Has First Reached Tip flag to true if it is not already
446    /// set.
447    #[allow(clippy::too_many_lines)]
448    async fn run(&mut self) {
449        // We can't sync until the local chain data is synced.
450        // This call will wait until we sync.
451        let tips = ChainFollower::get_tips(self.cfg.chain).await;
452        self.immutable_tip_slot = tips.0.slot_or_default();
453        self.live_tip_slot = tips.1.slot_or_default();
454        info!(chain=%self.cfg.chain, immutable_tip=?self.immutable_tip_slot, live_tip=?self.live_tip_slot, "Blockchain ready to sync from.");
455
456        self.dispatch_event(event::ChainIndexerEvent::ImmutableTipSlotChanged {
457            immutable_slot: self.immutable_tip_slot,
458            live_slot: self.live_tip_slot,
459        });
460        self.dispatch_event(event::ChainIndexerEvent::LiveTipSlotChanged {
461            immutable_slot: self.immutable_tip_slot,
462            live_slot: self.live_tip_slot,
463        });
464
465        // Wait for indexing DB to be ready before continuing.
466        // We do this after the above, because other nodes may have finished already, and we don't
467        // want to wait do any work they already completed while we were fetching the blockchain.
468        //
469        // After waiting, we set the liveness flag to true if it is not already set.
470        drop(CassandraSession::wait_until_ready(INDEXING_DB_READY_WAIT_INTERVAL, true).await);
471
472        info!(chain=%self.cfg.chain, "Indexing DB is ready - Getting recovery state");
473        self.sync_status = get_sync_status().await;
474        debug!(chain=%self.cfg.chain, "Sync Status: {:?}", self.sync_status);
475
476        // Start the Live Chain sync task - This can never end because it is syncing to TIP.
477        // So, if it fails, it will automatically be restarted.
478        self.sync_tasks.push(sync_subchain(
479            SyncParams::new(
480                self.cfg.chain,
481                Point::fuzzy(self.immutable_tip_slot),
482                Point::TIP,
483            ),
484            self.event_channel.0.clone(),
485        ));
486
487        self.start_immutable_followers();
488
489        self.dispatch_event(event::ChainIndexerEvent::SyncStarted);
490
491        // Wait Sync tasks to complete.  If they fail and have not completed, reschedule them.
492        // If an immutable sync task ends OK, and we still have immutable data to sync then
493        // start a new task.
494        // They will return from this iterator in the order they complete.
495        // This iterator actually never ends, because the live sync task is always restarted.
496        while let Some(completed) = self.sync_tasks.next().await {
497            match completed {
498                Ok(finished) => {
499                    // Sync task finished.  Check if it completed OK or had an error.
500                    // If it failed, we need to reschedule it.
501
502                    // The TIP follower should NEVER end, unless there is an immutable roll forward,
503                    // or there is an error.  If this is not a roll forward, log an error.
504                    // It can fail if the index DB goes down in some way.
505                    // Restart it always.
506                    if finished.end == Point::TIP {
507                        if let Some(ref roll_forward_point) = finished.follower_roll_forward {
508                            // Advance the known immutable tip, and try and start followers to reach
509                            // it.
510                            self.immutable_tip_slot = roll_forward_point.slot_or_default();
511
512                            self.dispatch_event(
513                                event::ChainIndexerEvent::ImmutableTipSlotChanged {
514                                    immutable_slot: self.immutable_tip_slot,
515                                    live_slot: self.live_tip_slot,
516                                },
517                            );
518
519                            self.start_immutable_followers();
520                        } else {
521                            error!(chain=%self.cfg.chain, report=%finished,
522                            "The TIP follower failed, restarting it.");
523                        }
524
525                        // Start the Live Chain sync task again from where it left off.
526                        self.sync_tasks.push(sync_subchain(
527                            finished.retry(),
528                            self.event_channel.0.clone(),
529                        ));
530                    } else if let Some(result) = finished.result.as_ref() {
531                        match result {
532                            Ok(()) => {
533                                self.current_sync_tasks =
534                                    self.current_sync_tasks.checked_sub(1).unwrap_or_else(|| {
535                                        error!("current_sync_tasks -= 1 overflow");
536                                        0
537                                    });
538                                info!(chain=%self.cfg.chain, report=%finished,
539                                    "The Immutable follower completed successfully.");
540
541                                finished.last_indexed_block.as_ref().inspect(|block| {
542                                    self.dispatch_event(
543                                        event::ChainIndexerEvent::IndexedSlotProgressed {
544                                            slot: block.slot_or_default(),
545                                        },
546                                    );
547                                });
548                                self.dispatch_event(event::ChainIndexerEvent::SyncTasksChanged {
549                                    current_sync_tasks: self.current_sync_tasks,
550                                });
551
552                                // If we need more immutable chain followers to sync the block
553                                // chain, we can now start them.
554                                self.start_immutable_followers();
555                            },
556                            Err(error) => {
557                                error!(chain=%self.cfg.chain, report=%finished, error=%error,
558                                        "An Immutable follower failed, restarting it.");
559                                // Restart the Immutable Chain sync task again from where it left
560                                // off.
561                                self.sync_tasks.push(sync_subchain(
562                                    finished.retry(),
563                                    self.event_channel.0.clone(),
564                                ));
565                            },
566                        }
567                    } else {
568                        error!(chain=%self.cfg.chain, report=%finished,
569                        "BUG: The Immutable follower completed, but without a proper result.");
570                    }
571                },
572                Err(error) => {
573                    error!(chain=%self.cfg.chain, error=%error, "BUG: Sync task failed. Can not restart it, not enough information.  Sync is probably failed at this point.");
574                },
575            }
576
577            // IF there is only 1 chain follower left in sync_tasks, then all
578            // immutable followers have finished.
579            // When this happens we need to purge the live index of any records that exist
580            // before the current immutable tip.
581            // Note: to prevent a data race when multiple nodes are syncing, we probably
582            // want to put a gap in this, so that there are X slots of overlap
583            // between the live chain and immutable chain.  This gap should be
584            // a parameter.
585            if self.sync_tasks.len() == 1 {
586                self.dispatch_event(event::ChainIndexerEvent::SyncCompleted);
587
588                // Purge data up to this slot
589                // Slots arithmetic has saturating semantic, so this is ok.
590                #[allow(clippy::arithmetic_side_effects)]
591                let purge_to_slot =
592                    self.immutable_tip_slot - Settings::purge_backward_slot_buffer();
593                let purge_condition = PurgeCondition::PurgeBackwards(purge_to_slot);
594                info!(chain=%self.cfg.chain, purge_to_slot=?purge_to_slot, "Backwards purging volatile data.");
595                if let Err(error) = roll_forward::purge_live_index(purge_condition).await {
596                    error!(chain=%self.cfg.chain, error=%error, "BUG: Purging volatile data task failed.");
597                } else {
598                    self.dispatch_event(event::ChainIndexerEvent::BackwardDataPurged);
599                }
600            }
601        }
602
603        error!(chain=%self.cfg.chain,"BUG: Sync tasks have all stopped.  This is an unexpected error!");
604    }
605
606    /// Start immutable followers, if we can
607    fn start_immutable_followers(&mut self) {
608        // Start the Immutable Chain sync tasks, as required.
609        // We will start at most the number of configured sync tasks.
610        // The live chain sync task is not counted as a sync task for this config value.
611
612        // Nothing to do if the start_slot is not less than the end of the immutable chain.
613        if self.start_slot < self.immutable_tip_slot {
614            // Will also break if there are no more slots left to sync.
615            while self.current_sync_tasks < self.cfg.sync_tasks {
616                let end_slot = self.immutable_tip_slot.min(
617                    (u64::from(self.start_slot).saturating_add(self.cfg.sync_chunk_max_slots))
618                        .into(),
619                );
620
621                if let Some((first_point, last_point)) =
622                    self.get_syncable_range(self.start_slot, end_slot)
623                {
624                    self.sync_tasks.push(sync_subchain(
625                        SyncParams::new(self.cfg.chain, first_point, last_point.clone()),
626                        self.event_channel.0.clone(),
627                    ));
628                    self.current_sync_tasks = self.current_sync_tasks.saturating_add(1);
629
630                    self.dispatch_event(event::ChainIndexerEvent::SyncTasksChanged {
631                        current_sync_tasks: self.current_sync_tasks,
632                    });
633                }
634
635                // The one slot overlap is deliberate, it doesn't hurt anything and prevents all off
636                // by one problems that may occur otherwise.
637                self.start_slot = end_slot;
638
639                if end_slot == self.immutable_tip_slot {
640                    break;
641                }
642            }
643            // `start_slot` is still used, because it is used to keep syncing chunks as required
644            // while each immutable sync task finishes.
645            info!(chain=%self.cfg.chain, tasks=self.current_sync_tasks, until=?self.start_slot, "Persistent Indexing DB tasks started");
646        }
647    }
648
649    /// Check if the requested range has already been indexed.
650    /// If it hasn't just return the slots as points.
651    /// If it has, return a subset that hasn't been indexed if any, or None if its been
652    /// completely indexed already.
653    fn get_syncable_range(&self, start: Slot, end: Slot) -> Option<(Point, Point)> {
654        for sync_block in &self.sync_status {
655            // Check if we start within a previously synchronized block.
656            if start >= sync_block.start_slot && start <= sync_block.end_slot {
657                // Check if we are fully contained by the sync block, if so, nothing to sync.
658                if end <= sync_block.end_slot {
659                    return None;
660                }
661
662                // In theory, we could extend into another sync block, but because we could extend
663                // into an unbounded number of sync blocks, we would need to bust
664                // this range into an unbounded number of sub chunks.
665                // It is not a problem to sync the same data mutiple times, so for simplicity we do
666                // not account for this, if the requested range goes beyond the sync
667                // block it starts within we assume that the rest is not synced.
668                return Some((Point::fuzzy(sync_block.end_slot), Point::fuzzy(end)));
669            }
670        }
671
672        let start_slot = if start == 0.into() {
673            Point::ORIGIN
674        } else {
675            Point::fuzzy(start)
676        };
677
678        Some((start_slot, Point::fuzzy(end)))
679    }
680}

682impl event::EventTarget<event::ChainIndexerEvent> for SyncTask {
683    fn add_event_listener(&mut self, listener: event::EventListenerFn<event::ChainIndexerEvent>) {
684        let mut rx = self.event_channel.0.subscribe();
685        tokio::spawn(async move {
686            while let Ok(event) = rx.recv().await {
687                (listener)(&event);
688            }
689        });
690    }
691
692    fn dispatch_event(&self, message: event::ChainIndexerEvent) {
693        let _ = self.event_channel.0.send(message);
694    }
695}

697/// Start followers as per defined in the config
698pub(crate) async fn start_followers() -> anyhow::Result<()> {
699    let cfg = Settings::follower_cfg();
700
701    // Log the chain follower configuration.
702    cfg.log();
703
704    // Start Syncing the blockchain, so we can consume its data as required.
705    start_sync_for(&cfg).await?;
706    info!(chain=%cfg.chain,"Chain Sync is started.");
707
708    tokio::spawn(async move {
709        use self::event::ChainIndexerEvent as Event;
710        use crate::metrics::chain_indexer::reporter;
711
712        let api_host_names = Settings::api_host_names().join(",");
713        let service_id = Settings::service_id();
714        let network = cfg.chain.to_string();
715
716        let mut sync_task = SyncTask::new(cfg);
717
718        // add an event listener to watch for certain events to report as metrics
719        sync_task.add_event_listener(Box::new(move |event: &Event| {
720            if let Event::SyncStarted = event {
721                reporter::REACHED_TIP
722                    .with_label_values(&[&api_host_names, service_id, &network])
723                    .set(0);
724            }
725            if let Event::SyncCompleted = event {
726                reporter::REACHED_TIP
727                    .with_label_values(&[&api_host_names, service_id, &network])
728                    .set(1);
729            }
730            if let Event::SyncTasksChanged { current_sync_tasks } = event {
731                reporter::RUNNING_INDEXER_TASKS_COUNT
732                    .with_label_values(&[&api_host_names, service_id, &network])
733                    .set(From::from(*current_sync_tasks));
734            }
735            if let Event::LiveTipSlotChanged {
736                live_slot,
737                immutable_slot,
738            } = event
739            {
740                reporter::CURRENT_LIVE_TIP_SLOT
741                    .with_label_values(&[&api_host_names, service_id, &network])
742                    .set(i64::try_from(u64::from(*live_slot)).unwrap_or(-1));
743
744                reporter::SLOT_TIP_DIFF
745                    .with_label_values(&[&api_host_names, service_id, &network])
746                    .set(
747                        i64::try_from(
748                            u64::from(*immutable_slot)
749                                .checked_sub(u64::from(*live_slot))
750                                .unwrap_or_default(),
751                        )
752                        .unwrap_or(-1),
753                    );
754            }
755            if let Event::ImmutableTipSlotChanged {
756                live_slot,
757                immutable_slot,
758            } = event
759            {
760                reporter::CURRENT_IMMUTABLE_TIP_SLOT
761                    .with_label_values(&[&api_host_names, service_id, &network])
762                    .set(i64::try_from(u64::from(*immutable_slot)).unwrap_or(-1));
763
764                reporter::SLOT_TIP_DIFF
765                    .with_label_values(&[&api_host_names, service_id, &network])
766                    .set(
767                        i64::try_from(
768                            u64::from(*immutable_slot)
769                                .checked_sub(u64::from(*live_slot))
770                                .unwrap_or_default(),
771                        )
772                        .unwrap_or(-1),
773                    );
774            }
775            if let Event::IndexedSlotProgressed { slot } = event {
776                reporter::HIGHEST_COMPLETE_INDEXED_SLOT
777                    .with_label_values(&[&api_host_names, service_id, &network])
778                    .set(i64::try_from(u64::from(*slot)).unwrap_or(-1));
779            }
780            if let Event::BackwardDataPurged = event {
781                reporter::TRIGGERED_BACKWARD_PURGES_COUNT
782                    .with_label_values(&[&api_host_names, service_id, &network])
783                    .inc();
784            }
785            if let Event::ForwardDataPurged { purge_slots } = event {
786                reporter::TRIGGERED_FORWARD_PURGES_COUNT
787                    .with_label_values(&[&api_host_names, service_id, &network])
788                    .inc();
789                reporter::PURGED_SLOTS
790                    .with_label_values(&[&api_host_names, service_id, &network])
791                    .set(i64::try_from(*purge_slots).unwrap_or(-1));
792            }
793        }));
794
795        sync_task.run().await;
796    });
797
798    Ok(())
799}
