1//! Implementation of the GET `/rbac/registrations` endpoint.
2use anyhow::anyhow;
3use futures::StreamExt;
4use poem_openapi::{payload::Json, ApiResponse, Object};
5use tracing::error;
6
7use crate::{
8    db::index::{
9        queries::rbac::get_registrations::{
10            GetRegistrationsByChainRootQuery, GetRegistrationsByChainRootQueryParams,
11        },
12        session::CassandraSession,
13    },
14    service::common::{
15        objects::cardano::hash::Hash, responses::WithErrorResponses,
16        types::headers::retry_after::RetryAfterOption,
17    },
18};
19
20/// GET RBAC registrations by chain root response list item.
21#[derive(Object)]
22pub(crate) struct RbacRegistration {
23    /// Registration transaction hash.
24    tx_hash: Hash,
25}
26
27/// GET RBAC registrations by chain root response.
28#[derive(Object)]
29pub(crate) struct RbacRegistrationsResponse {
30    /// Registrations by RBAC chain root.
31    #[oai(validator(max_items = "100000"))]
32    registrations: Vec<RbacRegistration>,
33}
34
35/// Endpoint responses.
36#[derive(ApiResponse)]
37pub(crate) enum Responses {
38    /// ## Ok
39    ///
40    /// Success returns a list of registration transaction ids.
41    #[oai(status = 200)]
42    Ok(Json<RbacRegistrationsResponse>),
43    /// ## Unprocessable Content
44    ///
45    /// Response for unprocessable content.
46    #[oai(status = 422)]
47    UnprocessableContent,
48}
49
50pub(crate) type AllResponses = WithErrorResponses<Responses>;
51
52/// Get chain root endpoint.
53pub(crate) async fn endpoint(chain_root: String) -> AllResponses {
54    let Some(session) = CassandraSession::get(true) else {
55        error!("Failed to acquire db session");
56        let err = anyhow::anyhow!("Failed to acquire db session");
57        return AllResponses::service_unavailable(&err, RetryAfterOption::Default);
58    };
59
60    let Ok(decoded_chain_root) = hex::decode(chain_root) else {
61        return Responses::UnprocessableContent.into();
62    };
63
64    let query_res = GetRegistrationsByChainRootQuery::execute(
65        &session,
66        GetRegistrationsByChainRootQueryParams {
67            chain_root: decoded_chain_root,
68        },
69    )
70    .await;
71
72    let mut row_iter = match query_res {
73        Ok(row_iter) => row_iter,
74        Err(err) => {
75            error!(
76                error = ?err,
77                "Failed to execute get registrations by chain root query"
78            );
79            return AllResponses::internal_error(&err);
80        },
81    };
82
83    let mut registrations = Vec::new();
84    while let Some(row_res) = row_iter.next().await {
85        let row = match row_res {
86            Ok(row) => row,
87            Err(err) => {
88                error!(error = ?err, "Failed to parse get registrations by chain root query row");
89                let err = anyhow!(err);
90                return AllResponses::internal_error(&err);
91            },
92        };
93
94        let item = RbacRegistration {
95            tx_hash: row.transaction_id.into(),
96        };
97        registrations.push(item);
98    }
99
100    Responses::Ok(Json(RbacRegistrationsResponse { registrations })).into()
101}
