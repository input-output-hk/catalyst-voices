28/// Cardano Follower API Endpoints
29pub(crate) struct Api;
30
31#[OpenApi(tag = "ApiTags::Cardano")]
32impl Api {
33    /// Get Cardano slot to the provided date-time.
34    ///
35    /// This endpoint returns the closest cardano slot info to the provided
36    /// date-time.
37    #[oai(
38        path = "/draft/cardano/date_time_to_slot_number",
39        method = "get",
40        operation_id = "dateTimeToSlotNumberGet",
41        transform = "schema_version_validation"
42    )]
43    async fn date_time_to_slot_number_get(
44        &self,
45        /// The date-time for which the slot number should be calculated.
46        /// If omitted current date time is used.
47        date_time: Query<Option<DateTime>>,
48        /// Cardano network type.
49        /// If omitted `mainnet` network type is defined.
50        /// As `preprod` and `preview` network types in the stake address encoded as a
51        /// `testnet`, to specify `preprod` or `preview` network type use this
52        /// query parameter.
53        network: Query<Option<Network>>,
54        /// No Authorization required, but Token permitted.
55        _auth: NoneOrRBAC,
56    ) -> date_time_to_slot_number_get::AllResponses {
57        date_time_to_slot_number_get::endpoint(date_time.0, network.0).await
58    }
59
60    #[oai(
61        path = "/draft/rbac/chain_root/:stake_address",
62        method = "get",
63        operation_id = "rbacChainRootGet"
64    )]
65    /// Get RBAC chain root
66    ///
67    /// This endpoint returns the RBAC certificate chain root for a given stake address.
68    async fn rbac_chain_root_get(
69        &self,
70        /// Stake address to get the chain root for.
71        Path(stake_address): Path<Cip19StakeAddress>,
72        /// No Authorization required, but Token permitted.
73        _auth: NoneOrRBAC,
74    ) -> rbac::chain_root_get::AllResponses {
75        rbac::chain_root_get::endpoint(stake_address).await
76    }
77
78    #[oai(
79        path = "/draft/rbac/registrations/:chain_root",
80        method = "get",
81        operation_id = "rbacRegistrations"
82    )]
83    /// Get registrations by RBAC chain root
84    ///
85    /// This endpoint returns the registrations for a given chain root.
86    async fn rbac_registrations_get(
87        &self,
88        /// Chain root to get the registrations for.
89        #[oai(validator(max_length = 66, min_length = 64, pattern = "0x[0-9a-f]{64}"))]
90        Path(chain_root): Path<String>,
91        /// No Authorization required, but Token permitted.
92        _auth: NoneOrRBAC,
93    ) -> rbac::registrations_get::AllResponses {
94        rbac::registrations_get::endpoint(chain_root).await
95    }
96
97    #[oai(
98        path = "/draft/rbac/role0_chain_root/:role0_key",
99        method = "get",
100        operation_id = "rbacRole0KeyChainRoot"
101    )]
102    /// Get RBAC chain root for a given role0 key.
103    ///
104    /// This endpoint returns the RBAC certificate chain root for a given role 0 key.
105    async fn rbac_role0_key_chain_root(
106        &self,
107        /// Role0 key to get the chain root for.
108        #[oai(validator(min_length = 34, max_length = 34, pattern = "0x[0-9a-f]{32}"))]
109        Path(role0_key): Path<String>,
110        /// No Authorization required, but Token permitted.
111        _auth: NoneOrRBAC,
112    ) -> rbac::role0_chain_root_get::AllResponses {
113        rbac::role0_chain_root_get::endpoint(role0_key).await
114    }
115}
