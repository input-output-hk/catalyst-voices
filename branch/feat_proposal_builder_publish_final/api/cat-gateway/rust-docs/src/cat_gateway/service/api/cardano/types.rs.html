1//! Cardano Specific Types
2//!
3//! These are temporary types are needed to prevent breakage due to the removal of the
4//! Event DB logic for chain-sync.  They should be replaced with proper types in a better
5//! place.
6
7use serde::{Deserialize, Serialize};
8
9/// Pub key
10#[derive(Serialize, Deserialize, Debug, Clone, PartialEq)]
11pub(crate) struct PubKey(Vec<u8>);
12
13impl PubKey {
14    /// Get bytes
15    pub(crate) fn bytes(&self) -> &[u8] {
16        &self.0
17    }
18}
19
20/// The source of voting power for a given registration
21///
22/// The voting power can either come from:
23///  - a single wallet, OR
24///  - a set of delegations
25#[derive(Serialize, Deserialize)]
26#[serde(untagged)]
27#[derive(Debug, Clone, PartialEq)]
28pub(crate) enum VotingInfo {
29    /// Direct voting
30    ///
31    /// Voting power is based on the staked ada of the given key
32    Direct(PubKey),
33
34    /// Delegated voting
35    ///
36    /// Voting power is based on the staked ada of the delegated keys
37    /// order of elements is important and must be preserved.
38    Delegated(Vec<(PubKey, i64)>),
39}
40
41/// Block time
42pub(crate) type DateTime = chrono::DateTime<chrono::offset::Utc>;
43/// Slot
44pub(crate) type SlotNumber = i64;
45/// Transaction id
46pub(crate) type TxId = Vec<u8>;
47/// Public voting key
48pub(crate) type PublicVotingInfo = VotingInfo;
49/// Payment address
50pub(crate) type PaymentAddress = Vec<u8>;
51/// Nonce
52pub(crate) type Nonce = i64;
53/// Stake amount.
54pub(crate) type StakeAmount = i64;
