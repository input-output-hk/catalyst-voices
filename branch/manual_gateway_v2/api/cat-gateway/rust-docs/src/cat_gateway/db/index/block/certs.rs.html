15/// Insert Cert Queries
16pub(crate) struct CertInsertQuery {
17    /// Stake Registration Data captured during indexing.
18    stake_reg_data: Vec<StakeRegistrationInsertQuery>,
19}
20
21impl CertInsertQuery {
22    /// Create new data set for Cert Insert Query Batch.
23    pub(crate) fn new() -> Self {
24        CertInsertQuery {
25            stake_reg_data: Vec::new(),
26        }
27    }

29    /// Get the stake address for a hash, return an empty address if one can not be found.
30    #[allow(clippy::too_many_arguments)]
31    fn stake_address(
32        &mut self, cred: &alonzo::StakeCredential, slot_no: Slot, txn: TxnIndex,
33        register: Option<bool>, deregister: Option<bool>, cip36: Option<bool>,
34        delegation: Option<Vec<u8>>, block: &MultiEraBlock,
35    ) {
36        let (stake_address, pubkey, script) = match cred {
37            conway::StakeCredential::AddrKeyhash(cred) => {
38                let stake_address = StakeAddress::new(block.network(), false, (*cred).into());
39                let addr = block.witness_for_tx(&VKeyHash::from(*cred), txn);
40                // Note: it is totally possible for the Registration Certificate to not be
41                // witnessed.
42                (stake_address, addr, false)
43            },
44            conway::StakeCredential::Scripthash(h) => {
45                (
46                    StakeAddress::new(block.network(), true, (*h).into()),
47                    None,
48                    true,
49                )
50            },
51        };
52
53        if pubkey.is_none() && !script && deregister.is_some_and(|v| v) {
54            error!("Stake Deregistration Certificate {stake_address} is NOT Witnessed.");
55        }
56
57        if pubkey.is_none() && !script && delegation.is_some() {
58            error!("Stake Delegation Certificate {stake_address} is NOT Witnessed.");
59        }
60
61        if let Some(pubkey) = pubkey {
62            self.stake_reg_data.push(StakeRegistrationInsertQuery::new(
63                stake_address,
64                slot_no,
65                txn,
66                pubkey,
67                script,
68                register,
69                deregister,
70                cip36,
71                delegation,
72            ));
73        }
74    }

76    /// Index an Alonzo Era certificate into the database.
77    fn index_alonzo_cert(
78        &mut self, cert: &alonzo::Certificate, slot: Slot, index: TxnIndex, block: &MultiEraBlock,
79    ) {
80        #[allow(clippy::match_same_arms)]
81        match cert {
82            alonzo::Certificate::StakeRegistration(cred) => {
83                // This may not be witnessed, its normal but disappointing.
84                self.stake_address(cred, slot, index, Some(true), None, None, None, block);
85            },
86            alonzo::Certificate::StakeDeregistration(cred) => {
87                self.stake_address(cred, slot, index, None, Some(true), None, None, block);
88            },
89            alonzo::Certificate::StakeDelegation(cred, pool) => {
90                self.stake_address(
91                    cred,
92                    slot,
93                    index,
94                    None,
95                    None,
96                    None,
97                    Some(pool.to_vec()),
98                    block,
99                );
100            },
101            alonzo::Certificate::PoolRegistration { .. } => {},
102            alonzo::Certificate::PoolRetirement(..) => {},
103            alonzo::Certificate::GenesisKeyDelegation(..) => {},
104            alonzo::Certificate::MoveInstantaneousRewardsCert(_) => {},
105        }
106    }

108    /// Index a certificate from a conway transaction.
109    fn index_conway_cert(
110        &mut self, cert: &conway::Certificate, slot_no: Slot, txn: TxnIndex, block: &MultiEraBlock,
111    ) {
112        #[allow(clippy::match_same_arms)]
113        match cert {
114            conway::Certificate::StakeRegistration(cred) => {
115                // This may not be witnessed, its normal but disappointing.
116                self.stake_address(cred, slot_no, txn, Some(true), None, None, None, block);
117            },
118            conway::Certificate::StakeDeregistration(cred) => {
119                self.stake_address(cred, slot_no, txn, None, Some(true), None, None, block);
120            },
121            conway::Certificate::StakeDelegation(cred, pool) => {
122                self.stake_address(
123                    cred,
124                    slot_no,
125                    txn,
126                    None,
127                    None,
128                    None,
129                    Some(pool.to_vec()),
130                    block,
131                );
132            },
133            conway::Certificate::PoolRegistration { .. } => {},
134            conway::Certificate::PoolRetirement(..) => {},
135            conway::Certificate::Reg(..) => {},
136            conway::Certificate::UnReg(..) => {},
137            conway::Certificate::VoteDeleg(..) => {},
138            conway::Certificate::StakeVoteDeleg(..) => {},
139            conway::Certificate::StakeRegDeleg(..) => {},
140            conway::Certificate::VoteRegDeleg(..) => {},
141            conway::Certificate::StakeVoteRegDeleg(..) => {},
142            conway::Certificate::AuthCommitteeHot(..) => {},
143            conway::Certificate::ResignCommitteeCold(..) => {},
144            conway::Certificate::RegDRepCert(..) => {},
145            conway::Certificate::UnRegDRepCert(..) => {},
146            conway::Certificate::UpdateDRepCert(..) => {},
147        }
148    }

150    /// Index the certificates in a transaction.
151    pub(crate) fn index(
152        &mut self, txs: &pallas::ledger::traverse::MultiEraTx<'_>, slot: Slot, index: TxnIndex,
153        block: &MultiEraBlock,
154    ) {
155        #[allow(clippy::match_same_arms)]
156        txs.certs().iter().for_each(|cert| {
157            match cert {
158                pallas::ledger::traverse::MultiEraCert::NotApplicable => {},
159                pallas::ledger::traverse::MultiEraCert::AlonzoCompatible(cert) => {
160                    self.index_alonzo_cert(cert, slot, index, block);
161                },
162                pallas::ledger::traverse::MultiEraCert::Conway(cert) => {
163                    self.index_conway_cert(cert, slot, index, block);
164                },
165                _ => {},
166            }
167        });
168    }
169
170    /// Execute the Certificate Indexing Queries.
171    ///
172    /// Consumes the `self` and returns a vector of futures.
173    pub(crate) fn execute(self, session: &Arc<CassandraSession>) -> FallibleQueryTasks {
174        let mut query_handles: FallibleQueryTasks = Vec::new();
175
176        let inner_session = session.clone();
177
178        query_handles.push(tokio::spawn(async move {
179            inner_session
180                .execute_batch(
181                    PreparedQuery::StakeRegistrationInsertQuery,
182                    self.stake_reg_data,
183                )
184                .await
185        }));
186
187        query_handles
188    }
