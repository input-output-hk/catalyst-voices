173impl TryFrom<SignedDocBody> for IndexedDocumentVersionDocumented {
174    type Error = anyhow::Error;
175
176    fn try_from(doc: SignedDocBody) -> Result<Self, Self::Error> {
177        let mut doc_ref = None;
178        let mut reply = None;
179        let mut template = None;
180        let mut brand = None;
181        let mut campaign = None;
182        let mut category = None;
183        if let Some(json_meta) = doc.metadata() {
184            let meta: catalyst_signed_doc::ExtraFields = serde_json::from_value(json_meta.clone())?;
185            doc_ref = meta.doc_ref().map(Into::into);
186            reply = meta.reply().map(Into::into);
187            template = meta.template().map(Into::into);
188            brand = meta.brand_id().map(Into::into);
189            campaign = meta.campaign_id().map(Into::into);
190            category = meta.campaign_id().map(Into::into);
191        }
192
193        Ok(IndexedDocumentVersionDocumented(IndexedDocumentVersion {
194            ver: DocumentVer::new_unchecked(doc.ver().to_string()),
195            doc_type: DocumentType::new_unchecked(doc.doc_type().to_string()),
196            doc_ref,
197            reply,
198            template,
199            brand,
200            campaign,
201            category,
202        }))
203    }
