19/// This is to prevent the init function from accidentally being called multiple times.
20static IS_INITIALIZED: AtomicBool = AtomicBool::new(false);
21
22/// Starts a background thread to periodically update Chain Follower metrics.
23///
24/// This function spawns a thread that updates the Chain Follower metrics
25/// at regular intervals defined by `METRICS_FOLLOWER_INTERVAL`.
26pub(crate) fn init_metrics_reporter() {
27    if IS_INITIALIZED.swap(true, Ordering::SeqCst) {
28        return;
29    }
30
31    let api_host_names = Settings::api_host_names().join(",");
32    let service_id = Settings::service_id();
33    let network = Settings::cardano_network().to_string();
34
35    // for sending metrics periodically
36    thread::spawn(move || {
37        reporter::MAX_CACHE_SIZE
38            .with_label_values(&[&api_host_names, service_id, &network])
39            .set(i64::try_from(Settings::rbac_cache_max_mem_size()).unwrap_or(-1));
40
41        loop {
42            let rbac_entries = RBAC_CACHE.rbac_entries();
43
44            let chain_size = size_of::<RegistrationChain>();
45            let key_size = size_of::<CatalystId>();
46
47            let approx_mem_used = (chain_size + key_size) * rbac_entries;
48
49            reporter::CACHING_RBAC_ENTRIES
50                .with_label_values(&[&api_host_names, service_id, &network])
51                .set(i64::try_from(rbac_entries).unwrap_or(-1));
52            reporter::CACHE_SIZE
53                .with_label_values(&[&api_host_names, service_id, &network])
54                .set(i64::try_from(approx_mem_used).unwrap_or(-1));
55
56            thread::sleep(Settings::metrics_rbac_cache_interval());
57        }
58    });
59
60    let api_host_names = Settings::api_host_names().join(",");
61    let service_id = Settings::service_id();
62    let network = Settings::cardano_network().to_string();
63
64    // for sending metrics on events
65    RBAC_CACHE.add_event_listener(Box::new(move |event: &Event| {
66        if let Event::Initialized { start_up_time } = event {
67            reporter::START_UP_TIME
68                .with_label_values(&[&api_host_names, service_id, &network])
69                .set(i64::try_from(start_up_time.as_millis()).unwrap_or(-1));
70        }
71        if let Event::RbacRegistrationChainAdded { .. } = event {}
72        if let Event::CacheAccessed {
73            is_found, latency, ..
74        } = event
75        {
76            reporter::CACHE_ACCESS
77                .with_label_values(&[&api_host_names, service_id, &network])
78                .inc();
79
80            if *is_found {
81                reporter::CACHE_HIT
82                    .with_label_values(&[&api_host_names, service_id, &network])
83                    .inc();
84                reporter::LATENCY
85                    .with_label_values(&[&api_host_names, service_id, &network])
86                    .observe(latency.as_secs_f64());
87            } else {
88                reporter::CACHE_MISS
89                    .with_label_values(&[&api_host_names, service_id, &network])
90                    .inc();
91            }
92        }
93    }));
94}

96/// All the related RBAC Registration Chain Caching reporting metrics to the Prometheus
97/// service are inside this module.
98pub(crate) mod reporter {
99    use std::sync::LazyLock;
100
101    use prometheus::{
102        register_counter_vec, register_histogram_vec, register_int_gauge_vec, CounterVec,
103        HistogramVec, IntGaugeVec,
104    };
105
106    /// Labels for the metrics.
107    const METRIC_LABELS: [&str; 3] = ["api_host_names", "service_id", "network"];
108
109    /// Total count of cache hits.
110    pub(crate) static CACHE_HIT: LazyLock<CounterVec> = LazyLock::new(|| {
111        register_counter_vec!("cache_hit", "Total count of cache hits", &METRIC_LABELS).unwrap()
112    });
113
114    /// Total count of cache misses.
115    pub(crate) static CACHE_MISS: LazyLock<CounterVec> = LazyLock::new(|| {
116        register_counter_vec!("cache_miss", "Total count of cache misses", &METRIC_LABELS).unwrap()
117    });
118
119    /// Total count of cache access attempts.
120    pub(crate) static CACHE_ACCESS: LazyLock<CounterVec> = LazyLock::new(|| {
121        register_counter_vec!(
122            "cache_access",
123            "Total count of cache access attempts",
124            &METRIC_LABELS
125        )
126        .unwrap()
127    });
128
129    /// Duration measured in milliseconds for accessing cache on a cache hit.
130    pub(crate) static LATENCY: LazyLock<HistogramVec> = LazyLock::new(|| {
131        register_histogram_vec!(
132            "latency",
133            "Duration measured in milliseconds for accessing cache on a cache hit",
134            &METRIC_LABELS
135        )
136        .unwrap()
137    });
138
139    /// Total space in bytes currently using for RBAC caching.
140    pub(crate) static CACHE_SIZE: LazyLock<IntGaugeVec> = LazyLock::new(|| {
141        register_int_gauge_vec!(
142            "cache_size",
143            "Total space in bytes currently using for RBAC caching",
144            &METRIC_LABELS
145        )
146        .unwrap()
147    });
148
149    /// Maximum space in bytes for RBAC caching.
150    pub(crate) static MAX_CACHE_SIZE: LazyLock<IntGaugeVec> = LazyLock::new(|| {
151        register_int_gauge_vec!(
152            "max_cache_size",
153            "Maximum space in bytes for RBAC caching",
154            &METRIC_LABELS
155        )
156        .unwrap()
157    });
158
159    /// Total time in milliseconds taken to set up RBAC caching service to be operational.
160    pub(crate) static START_UP_TIME: LazyLock<IntGaugeVec> = LazyLock::new(|| {
161        register_int_gauge_vec!(
162            "start_up_time",
163            "Total time in milliseconds taken to set up RBAC caching service to be operational",
164            &METRIC_LABELS
165        )
166        .unwrap()
167    });
168
169    /// Total number of RBAC registration chain entries being stored as cache.
170    pub(crate) static CACHING_RBAC_ENTRIES: LazyLock<IntGaugeVec> = LazyLock::new(|| {
171        register_int_gauge_vec!(
172            "caching_rbac_entries",
173            "Total number of RBAC registration chain entries being stored as cache",
174            &METRIC_LABELS
175        )
176        .unwrap()
177    });
178}
