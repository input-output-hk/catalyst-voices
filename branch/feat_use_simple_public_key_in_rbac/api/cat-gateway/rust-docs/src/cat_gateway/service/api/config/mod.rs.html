21/// Configuration API struct
22pub(crate) struct ConfigApi;
23
24/// Get configuration endpoint responses.
25#[derive(ApiResponse)]
26enum GetConfigResponses {
27    /// Configuration result.
28    #[oai(status = 200)]
29    Ok(Json<JsonObject>),
30
31    /// No frontend config defined.
32    #[oai(status = 404)]
33    NotFound,
34}
35/// Get configuration all responses.
36type GetConfigAllResponses = WithErrorResponses<GetConfigResponses>;
37
38/// Set configuration endpoint responses.
39#[derive(ApiResponse)]
40enum SetConfigResponse {
41    /// Configuration Update Successful.
42    #[oai(status = 204)]
43    Ok,
44}
45/// Set configuration all responses.
46type SetConfigAllResponses = WithErrorResponses<SetConfigResponse>;
47
48#[OpenApi(tag = "ApiTags::Config")]
49impl ConfigApi {
50    /// Get the configuration for the frontend.
51    ///
52    /// Get the frontend configuration for the requesting client.
53    ///
54    /// ### Security
55    ///
56    /// Does not require any Catalyst RBAC Token to access.
57    #[oai(
58        path = "/v1/config/frontend",
59        method = "get",
60        operation_id = "get_config_frontend"
61    )]
62    async fn get_frontend(&self, ip_address: RealIp, _auth: NoneOrRBAC) -> GetConfigAllResponses {
63        let general_config: JsonObject = match Config::get(ConfigKey::Frontend)
64            .await
65            .and_then(TryInto::try_into)
66        {
67            Ok(value) => value,
68            Err(err) if err.is::<NotFoundError>() => return GetConfigResponses::NotFound.into(),
69            Err(err) => {
70                error!(id="get_frontend_config_general", error=?err, "Failed to get general frontend configuration");
71                return GetConfigAllResponses::handle_error(&err);
72            },
73        };
74
75        // Attempt to fetch the IP configuration
76        let ip_config: Option<JsonObject> = if let Some(ip) = ip_address.0 {
77            match Config::get(ConfigKey::FrontendForIp(ip))
78                .await
79                .and_then(TryInto::try_into)
80            {
81                Ok(value) => Some(value),
82                Err(err) if err.is::<NotFoundError>() => None,
83                Err(err) => {
84                    error!(id = "get_frontend_config_ip", error = ?err, "Failed to get frontend configuration for IP");
85                    return GetConfigAllResponses::handle_error(&err);
86                },
87            }
88        } else {
89            None
90        };
91        if let Some(ip_config) = ip_config {
92            let config = merge_configs(general_config, ip_config);
93            GetConfigResponses::Ok(Json(config)).into()
94        } else {
95            GetConfigResponses::Ok(Json(general_config)).into()
96        }
97    }
98
99    /// Set the frontend configuration.
100    ///
101    /// Store the given config as either global front end configuration, or configuration
102    /// for a client at a specific IP address.
103    ///
104    /// ### Security
105    ///
106    /// Requires Admin Authoritative RBAC Token.
107    #[oai(
108        path = "/v1/config/frontend",
109        method = "put",
110        operation_id = "put_config_frontend",
111        hidden = true
112    )]
113    async fn put_frontend(
114        &self,
115        /// *OPTIONAL* The IP Address to set the configuration for.
116        #[oai(name = "IP")]
117        Query(ip_address): Query<Option<IpAddr>>,
118        Json(json_config): Json<JsonObject>, _auth: InternalApiKeyAuthorization,
119    ) -> SetConfigAllResponses {
120        if let Some(ip) = ip_address {
121            set(ConfigKey::FrontendForIp(ip.into()), json_config.into()).await
122        } else {
123            set(ConfigKey::Frontend, json_config.into()).await
124        }
125    }
126}
127
128/// Helper function to merge two JSON values.
129fn merge_configs(general: JsonObject, ip_specific: JsonObject) -> JsonObject {
130    let mut merged = general;
131
132    for (key, value) in Map::<String, Value>::from(ip_specific) {
133        if let Some(existing_value) = merged.get_mut(&key) {
134            *existing_value = value;
135        } else {
136            merged.insert(key, value);
137        }
138    }
139
140    merged
141}
142
143/// Helper function to handle set.
144async fn set(key: ConfigKey, value: Value) -> SetConfigAllResponses {
145    match Config::set(key, value).await {
146        Ok(()) => SetConfigResponse::Ok.into(),
147        Err(err) => {
148            error!(id="set_config_frontend", error=?err, "Failed to set frontend configuration");
149            SetConfigAllResponses::handle_error(&err)
150        },
151    }
152}
