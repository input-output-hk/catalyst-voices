22/// Title.
23const TITLE: &str = "28 Byte Hash";
24/// Description.
25const DESCRIPTION: &str = "This is a 28 Byte Hex encoded Hash.";
26/// Example.
27const EXAMPLE: &str = "0x8eee77e5894c22268d5d12e6484ba713e7ddd595abba308d88d36943";
28/// Length of the hex encoded string;
29const ENCODED_LENGTH: usize = EXAMPLE.len();
30/// Length of the hash itself;
31const HASH_LENGTH: usize = BLAKE_2B224_SIZE;
32/// Validation Regex Pattern
33const PATTERN: &str = concatcp!("^0x", "[A-Fa-f0-9]{", HASH_LENGTH * 2, "}$");
34
35/// Schema.
36static SCHEMA: LazyLock<MetaSchema> = LazyLock::new(|| {
37    MetaSchema {
38        title: Some(TITLE.to_owned()),
39        description: Some(DESCRIPTION),
40        example: Some(Value::String(EXAMPLE.to_string())),
41        max_length: Some(ENCODED_LENGTH),
42        min_length: Some(ENCODED_LENGTH),
43        pattern: Some(PATTERN.to_string()),
44        ..poem_openapi::registry::MetaSchema::ANY
45    }
46});
47
48/// Validate `HexEncodedHash28` This part is done separately from the `PATTERN`
49fn is_valid(hash: &str) -> bool {
50    /// Regex to validate `HexEncodedHash28`
51    #[allow(clippy::unwrap_used)] // Safe because the Regex is constant.
52    static RE: LazyLock<Regex> = LazyLock::new(|| Regex::new(PATTERN).unwrap());
53
54    if RE.is_match(hash) {
55        if let Some(h) = hash.strip_prefix("0x") {
56            return hex::decode(h).is_ok();
57        }
58    }
59    false
60}
61
62impl_string_types!(
63    HexEncodedHash28,
64    "string",
65    "hex:hash(28)",
66    Some(SCHEMA.clone()),
67    is_valid
68);
69
70impl Example for HexEncodedHash28 {
71    /// An example 32 bytes ED25519 Public Key.
72    fn example() -> Self {
73        Self(EXAMPLE.to_owned())
74    }
75}

77impl TryFrom<Vec<u8>> for HexEncodedHash28 {
78    type Error = anyhow::Error;
79
80    fn try_from(value: Vec<u8>) -> Result<Self, Self::Error> {
81        if value.len() != HASH_LENGTH {
82            bail!("Hash Length Invalid.")
83        }
84        Ok(Self(as_hex_string(&value)))
85    }
86}
87
88// Because it is impossible for the Encoded Hash to not be valid (due to `is_valid`), we
89// can ensure this method is infallible.
90// All creation of this type should come only from one of the deserialization methods.
91impl From<HexEncodedHash28> for Vec<u8> {
92    fn from(val: HexEncodedHash28) -> Self {
93        #[allow(clippy::expect_used)]
94        from_hex_string(&val.0).expect("This can only fail if the type was invalidly constructed.")
95    }
96}
