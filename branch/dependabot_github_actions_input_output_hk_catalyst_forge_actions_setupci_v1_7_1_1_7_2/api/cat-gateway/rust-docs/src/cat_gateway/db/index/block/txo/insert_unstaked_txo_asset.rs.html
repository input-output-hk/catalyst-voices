1//! Insert Unstaked TXO Native Assets into the DB.
2
3use std::sync::Arc;
4
5use cardano_blockchain_types::{Slot, TransactionId, TxnIndex, TxnOutputOffset};
6use scylla::{SerializeRow, Session};
7use tracing::error;
8
9use crate::{
10    db::{
11        index::queries::{PreparedQueries, SizedBatch},
12        types::{DbSlot, DbTransactionId, DbTxnIndex, DbTxnOutputOffset},
13    },
14    settings::cassandra_db,
15};
16
17/// Unstaked TXO Asset by Stake Address Indexing Query
18const INSERT_UNSTAKED_TXO_ASSET_QUERY: &str = include_str!("./cql/insert_unstaked_txo_asset.cql");
19
20/// Insert TXO Asset Query Parameters
21/// (Superset of data to support both Staked and Unstaked TXO records.)
22#[derive(SerializeRow, Debug)]
23pub(crate) struct Params {
24    /// Transactions hash.
25    txn_id: DbTransactionId,
26    /// Transaction Output Offset inside the transaction.
27    txo: DbTxnOutputOffset,
28    /// Policy hash of the asset
29    policy_id: Vec<u8>,
30    /// Policy name of the asset
31    asset_name: Vec<u8>,
32    /// Block Slot Number
33    slot_no: DbSlot,
34    /// Transaction Offset inside the block.
35    txn_index: DbTxnIndex,
36    /// Value of the asset
37    value: num_bigint::BigInt,
38}
39
40impl Params {
41    /// Create a new record for this transaction.
42    ///
43    /// Note Value can be either a u64 or an i64, so use a i128 to represent all possible
44    /// values.
45    #[allow(clippy::too_many_arguments)]
46    pub(crate) fn new(
47        txn_id: TransactionId, txo: TxnOutputOffset, policy_id: &[u8], asset_name: &[u8],
48        slot_no: Slot, txn_index: TxnIndex, value: i128,
49    ) -> Self {
50        Self {
51            txn_id: txn_id.into(),
52            txo: txo.into(),
53            policy_id: policy_id.to_vec(),
54            asset_name: asset_name.to_vec(),
55            slot_no: slot_no.into(),
56            txn_index: txn_index.into(),
57            value: value.into(),
58        }
59    }
60
61    /// Prepare Batch of Staked Insert TXO Asset Index Data Queries
62    pub(crate) async fn prepare_batch(
63        session: &Arc<Session>, cfg: &cassandra_db::EnvVars,
64    ) -> anyhow::Result<SizedBatch> {
65        PreparedQueries::prepare_batch(
66            session.clone(),
67            INSERT_UNSTAKED_TXO_ASSET_QUERY,
68            cfg,
69            scylla::statement::Consistency::Any,
70            true,
71            false,
72        )
73        .await
74        .inspect_err(
75            |error| error!(error=%error,"Failed to prepare Insert Unstaked TXO Asset Query."),
76        )
77        .map_err(|error| anyhow::anyhow!("{error}\n--\n{INSERT_UNSTAKED_TXO_ASSET_QUERY}"))
78    }
79}
