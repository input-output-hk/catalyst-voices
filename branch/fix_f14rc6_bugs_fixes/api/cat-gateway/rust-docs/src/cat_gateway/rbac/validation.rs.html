22/// Validates a new registration by either starting a new chain or adding it to the
23/// existing one.
24///
25/// In case of failure a problem report from the given registration is updated and
26/// returned.
27pub async fn validate_rbac_registration(
28    reg: Cip509, is_persistent: bool, context: &mut RbacIndexingContext,
29) -> RbacValidationResult {
30    match reg.previous_transaction() {
31        Some(previous_txn) => update_chain(reg, previous_txn, is_persistent, context).await,
32        None => start_new_chain(reg, is_persistent).await,
33    }
34}

36async fn update_chain(
37    reg: Cip509, previous_txn: TransactionId, is_persistent: bool,
38    context: &mut RbacIndexingContext,
39) -> RbacValidationResult {
40    let purpose = reg.purpose();
41    let report = reg.report().to_owned();
42
43    // Find a chain this registration belongs to.
44    let catalyst_id = match catalyst_id_from_txn_id(previous_txn, is_persistent, context).await? {
45        Some(id) => id,
46        None => {
47            // We are unable to determine a Catalyst ID, so there is no sense to update the
48            // problem report because we would be unable to store this registration anyway.
49            return Err(RbacValidationError::UnknownCatalystId);
50        },
51    };
52    let chain = match chain(&catalyst_id, is_persistent, context).await? {
53        Some(c) => c,
54        None => {
55            return Err(anyhow!("{catalyst_id} is present in 'catalyst_id_for_txn_id' table, but not in 'rbac_registration'").into());
56        },
57    };
58
59    // Check that addresses from the new registration aren't used in other chains.
60    let previous_addresses = chain.role_0_stake_addresses();
61    let reg_addresses = reg.role_0_stake_addresses();
62    let new_addresses: Vec<_> = reg_addresses.difference(&previous_addresses).collect();
63    for address in &new_addresses {
64        match catalyst_id_from_stake_address(address, is_persistent, context).await? {
65            None => {
66                // All good: the address wasn't used before.
67            },
68            Some(_) => {
69                report.functional_validation(
70                    &format!("{address} stake addresses is already used"),
71                    "It isn't allowed to use same stake address in multiple registration chains",
72                );
73            },
74        }
75    }
76
77    // Store values before consuming the registration.
78    let txn_id = reg.txn_hash();
79    let stake_addresses = reg.role_0_stake_addresses();
80    let origin = reg.origin().to_owned();
81
82    // Try to add a new registration to the chain.
83    let new_chain = chain.update(reg).map_err(|e| {
84        report.other(
85            &format!("{e:?}"),
86            "Failed to apply update the registration chain",
87        );
88        RbacValidationError::InvalidRegistration {
89            catalyst_id: catalyst_id.clone(),
90            purpose,
91            report: report.clone(),
92        }
93    })?;
94
95    // Check that new public keys aren't used by other chains.
96    let keys = validate_public_keys(&new_chain, is_persistent, &report, context).await?;
97
98    if report.is_problematic() {
99        return Err(RbacValidationError::InvalidRegistration {
100            catalyst_id,
101            purpose,
102            report,
103        });
104    }
105
106    // Everything is fine: update the context.
107    context.insert_transaction(txn_id, catalyst_id.clone());
108    context.insert_addresses(stake_addresses, catalyst_id.clone());
109    context.insert_public_keys(&keys, catalyst_id.clone());
110    context.insert_registration(
111        catalyst_id.clone(),
112        txn_id,
113        origin.point().slot_or_default(),
114        origin.txn_index(),
115        Some(previous_txn),
116        // Only a new chain can remove stake addresses from an existing one.
117        HashSet::new(),
118    );
119
120    Ok(vec![RbacUpdate {
121        catalyst_id,
122        removed_stake_addresses: HashSet::new(),
123    }])
124}

233/// Returns a Catalyst ID corresponding to the given transaction hash.
234async fn catalyst_id_from_txn_id(
235    txn_id: TransactionId, is_persistent: bool, context: &mut RbacIndexingContext,
236) -> Result<Option<CatalystId>> {
237    use crate::db::index::queries::rbac::get_catalyst_id_from_transaction_id::Query;
238
239    // Check the context first.
240    if let Some(catalyst_id) = context.find_transaction(&txn_id) {
241        return Ok(Some(catalyst_id.to_owned()));
242    }
243
244    // Then try to find in the persistent database.
245    let session =
246        CassandraSession::get(true).context("Failed to get Cassandra persistent session")?;
247    if let Some(r) = Query::get(&session, txn_id).await? {
248        return Ok(Some(r.catalyst_id));
249    };
250
251    // Conditionally check the volatile database.
252    if !is_persistent {
253        let session =
254            CassandraSession::get(false).context("Failed to get Cassandra volatile session")?;
255        return Query::get(&session, txn_id)
256            .map_ok(|r| r.map(|r| r.catalyst_id))
257            .await;
258    }
259
260    Ok(None)
261}

263/// Returns either persistent or "latest" (persistent + volatile) registration chain for
264/// the given Catalyst ID.
265async fn chain(
266    id: &CatalystId, is_persistent: bool, context: &mut RbacIndexingContext,
267) -> Result<Option<RegistrationChain>> {
268    let chain = if is_persistent {
269        persistent_rbac_chain(id).await?
270    } else {
271        latest_rbac_chain(id).await?.map(|i| i.chain)
272    };
273
274    // Apply additional registrations from context if any.
275    if let Some(regs) = context.find_registrations(id) {
276        let regs = regs.iter().cloned();
277        match chain {
278            Some(c) => apply_regs(c, regs).await.map(Some),
279            None => build_rbac_chain(regs).await,
280        }
281    } else {
282        Ok(chain)
283    }
284}

316/// Checks that a new registration doesn't contain a signing key that was used by any
317/// other chain. Returns a list of public keys in the registration.
318async fn validate_public_keys(
319    chain: &RegistrationChain, is_persistent: bool, report: &ProblemReport,
320    context: &mut RbacIndexingContext,
321) -> Result<Vec<VerifyingKey>> {
322    let mut keys = Vec::new();
323
324    let roles: Vec<_> = chain.role_data_history().keys().collect();
325    let catalyst_id = chain.catalyst_id();
326
327    for role in roles {
328        if let Some((key, _)) = chain.get_latest_signing_pk_for_role(role) {
329            keys.push(key);
330            if let Some(previous) = catalyst_id_from_public_key(key, is_persistent, context).await?
331            {
332                if &previous != catalyst_id {
333                    report.functional_validation(
334                        &format!("An update to {catalyst_id} registration chain uses the same public key ({key:?}) as {previous} chain"),
335                        "It isn't allowed to use role 0 signing (certificate subject public) key in different chains",
336                    );
337                }
338            }
339        }
340    }
341
342    Ok(keys)
343}
