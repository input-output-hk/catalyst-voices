34/// Endpoint responses.
35#[derive(ApiResponse)]
36pub(crate) enum Responses {
37    /// ## Ok
38    ///
39    /// The amount of ADA staked by the queried stake address, as at the indicated slot.
40    #[oai(status = 200)]
41    Ok(Json<FullStakeInfo>),
42    /// ## Not Found
43    ///
44    /// The queried stake address was not found at the requested slot number.
45    #[oai(status = 404)]
46    NotFound,
47}
48
49/// All responses.
50pub(crate) type AllResponses = WithErrorResponses<Responses>;
51
52/// # GET `/staked_ada`
53#[allow(clippy::unused_async, clippy::no_effect_underscore_binding)]
54pub(crate) async fn endpoint(
55    stake_address: Cip19StakeAddress, _provided_network: Option<Network>,
56    slot_num: Option<SlotNumber>,
57) -> AllResponses {
58    let persistent_res = calculate_stake_info(true, stake_address.clone(), slot_num).await;
59    let persistent_stake_info = match persistent_res {
60        Ok(stake_info) => stake_info,
61        Err(err) => return AllResponses::handle_error(&err),
62    };
63
64    let volatile_res = calculate_stake_info(false, stake_address, slot_num).await;
65    let volatile_stake_info = match volatile_res {
66        Ok(stake_info) => stake_info,
67        Err(err) => return AllResponses::handle_error(&err),
68    };
69
70    if persistent_stake_info.is_none() && volatile_stake_info.is_none() {
71        return Responses::NotFound.into();
72    }
73
74    Responses::Ok(Json(FullStakeInfo {
75        volatile: volatile_stake_info.unwrap_or_default(),
76        persistent: persistent_stake_info.unwrap_or_default(),
77    }))
78    .into()
79}

81/// TXO asset information.
82struct TxoAssetInfo {
83    /// Asset hash.
84    id: Vec<u8>,
85    /// Asset name.
86    name: AssetName,
87    /// Asset amount.
88    amount: num_bigint::BigInt,
89}

91/// TXO information used when calculating a user's stake info.
92struct TxoInfo {
93    /// TXO value.
94    value: num_bigint::BigInt,
95    /// TXO transaction hash.
96    txn_hash: Vec<u8>,
97    /// TXO transaction index within the slot.
98    txn: i16,
99    /// TXO index.
100    txo: i16,
101    /// TXO transaction slot number.
102    slot_no: num_bigint::BigInt,
103    /// Whether the TXO was spent.
104    spent_slot_no: Option<num_bigint::BigInt>,
105    /// TXO assets.
106    assets: HashMap<Vec<u8>, Vec<TxoAssetInfo>>,
107}

109/// Calculate the stake info for a given stake address.
110///
111/// This function also updates the spent column if it detects that a TXO was spent
112/// between lookups.
113async fn calculate_stake_info(
114    persistent: bool, stake_address: Cip19StakeAddress, slot_num: Option<SlotNumber>,
115) -> anyhow::Result<Option<StakeInfo>> {
116    let Some(session) = CassandraSession::get(persistent) else {
117        anyhow::bail!("Failed to acquire db session");
118    };
119
120    let address: StakeAddress = stake_address.try_into()?;
121    let stake_address_bytes = address.payload().as_hash().to_vec();
122
123    let mut txos_by_txn = get_txo_by_txn(&session, stake_address_bytes.clone(), slot_num).await?;
124    if txos_by_txn.is_empty() {
125        return Ok(None);
126    }
127
128    check_and_set_spent(&session, &mut txos_by_txn).await?;
129    // TODO: This could be executed in the background, it does not actually matter if it
130    // succeeds. This is just an optimization step to reduce the need to query spent
131    // TXO's.
132    update_spent(&session, stake_address_bytes, &txos_by_txn).await?;
133
134    let stake_info = build_stake_info(txos_by_txn)?;
135
136    Ok(Some(stake_info))
137}

139/// Returns a map of TXO infos by transaction hash for the given stake address.
140async fn get_txo_by_txn(
141    session: &CassandraSession, stake_address: Vec<u8>, slot_num: Option<SlotNumber>,
142) -> anyhow::Result<HashMap<Vec<u8>, HashMap<i16, TxoInfo>>> {
143    let adjusted_slot_num = num_bigint::BigInt::from(slot_num.unwrap_or(i64::MAX));
144
145    let mut txo_map = HashMap::new();
146    let mut txos_iter = GetTxoByStakeAddressQuery::execute(
147        session,
148        GetTxoByStakeAddressQueryParams::new(stake_address.clone(), adjusted_slot_num.clone()),
149    )
150    .await?;
151
152    // Aggregate TXO info.
153    while let Some(row_res) = txos_iter.next().await {
154        let row = row_res?;
155
156        // Filter out already known spent TXOs.
157        if row.spent_slot.is_some() {
158            continue;
159        }
160
161        let key = (row.slot_no.clone(), row.txn, row.txo);
162        txo_map.insert(key, TxoInfo {
163            value: row.value,
164            txn_hash: row.txn_hash,
165            txn: row.txn,
166            txo: row.txo,
167            slot_no: row.slot_no,
168            spent_slot_no: None,
169            assets: HashMap::new(),
170        });
171    }
172
173    // Augment TXO info with asset info.
174    let mut assets_txos_iter = GetAssetsByStakeAddressQuery::execute(
175        session,
176        GetAssetsByStakeAddressParams::new(stake_address, adjusted_slot_num),
177    )
178    .await?;
179
180    while let Some(row_res) = assets_txos_iter.next().await {
181        let row = row_res?;
182
183        let txo_info_key = (row.slot_no.clone(), row.txn, row.txo);
184        let Some(txo_info) = txo_map.get_mut(&txo_info_key) else {
185            continue;
186        };
187
188        let entry = txo_info
189            .assets
190            .entry(row.policy_id.clone())
191            .or_insert_with(Vec::new);
192
193        match entry.iter_mut().find(|item| item.id == row.policy_id) {
194            Some(item) => item.amount += row.value,
195            None => {
196                entry.push(TxoAssetInfo {
197                    id: row.policy_id,
198                    name: row.asset_name.into(),
199                    amount: row.value,
200                });
201            },
202        }
203    }
204
205    let mut txos_by_txn = HashMap::new();
206    for txo_info in txo_map.into_values() {
207        let txn_map = txos_by_txn
208            .entry(txo_info.txn_hash.clone())
209            .or_insert(HashMap::new());
210        txn_map.insert(txo_info.txo, txo_info);
211    }
212
213    Ok(txos_by_txn)
214}

216/// Checks if the given TXOs were spent and mark then as such.
217async fn check_and_set_spent(
218    session: &CassandraSession, txos_by_txn: &mut HashMap<Vec<u8>, HashMap<i16, TxoInfo>>,
219) -> anyhow::Result<()> {
220    let txn_hashes = txos_by_txn.keys().cloned().collect::<Vec<_>>();
221
222    for chunk in txn_hashes.chunks(100) {
223        let mut txi_iter = GetTxiByTxnHashesQuery::execute(
224            session,
225            GetTxiByTxnHashesQueryParams::new(chunk.to_vec()),
226        )
227        .await?;
228
229        while let Some(row_res) = txi_iter.next().await {
230            let row = row_res?;
231
232            if let Some(txn_map) = txos_by_txn.get_mut(&row.txn_hash) {
233                if let Some(txo_info) = txn_map.get_mut(&row.txo) {
234                    if row.slot_no >= num_bigint::BigInt::ZERO {
235                        txo_info.spent_slot_no = Some(row.slot_no);
236                    }
237                }
238            }
239        }
240    }
241
242    Ok(())
243}

245/// Sets TXOs as spent in the database if they are marked as spent in the map.
246async fn update_spent(
247    session: &CassandraSession, stake_address: Vec<u8>,
248    txos_by_txn: &HashMap<Vec<u8>, HashMap<i16, TxoInfo>>,
249) -> anyhow::Result<()> {
250    let mut params = Vec::new();
251    for txn_map in txos_by_txn.values() {
252        for txo_info in txn_map.values() {
253            if txo_info.spent_slot_no.is_none() {
254                continue;
255            }
256
257            if let Some(spent_slot) = &txo_info.spent_slot_no {
258                params.push(UpdateTxoSpentQueryParams {
259                    stake_address: stake_address.clone(),
260                    txn: txo_info.txn,
261                    txo: txo_info.txo,
262                    slot_no: txo_info.slot_no.clone(),
263                    spent_slot: spent_slot.clone(),
264                });
265            }
266        }
267    }
268
269    UpdateTxoSpentQuery::execute(session, params).await?;
270
271    Ok(())
272}

274/// Builds an instance of [`StakeInfo`] based on the TXOs given.
275fn build_stake_info(
276    txos_by_txn: HashMap<Vec<u8>, HashMap<i16, TxoInfo>>,
277) -> anyhow::Result<StakeInfo> {
278    let mut stake_info = StakeInfo::default();
279    for txn_map in txos_by_txn.into_values() {
280        for txo_info in txn_map.into_values() {
281            if txo_info.spent_slot_no.is_none() {
282                let value = i64::try_from(txo_info.value).map_err(|err| anyhow!(err))?;
283                stake_info.ada_amount =
284                    stake_info.ada_amount.checked_add(value).ok_or_else(|| {
285                        anyhow!(
286                            "Total stake amount overflow: {} + {value}",
287                            stake_info.ada_amount
288                        )
289                    })?;
290
291                for asset in txo_info.assets.into_values().flatten() {
292                    stake_info.native_tokens.push(StakedNativeTokenInfo {
293                        policy_hash: asset.id.try_into()?,
294                        asset_name: asset.name,
295                        amount: asset.amount.try_into()?,
296                    });
297                }
298
299                let slot_no = i64::try_from(txo_info.slot_no).map_err(|err| anyhow!(err))?;
300
301                if stake_info.slot_number < slot_no {
302                    stake_info.slot_number = slot_no;
303                }
304            }
305        }
306    }
307
308    Ok(stake_info)
