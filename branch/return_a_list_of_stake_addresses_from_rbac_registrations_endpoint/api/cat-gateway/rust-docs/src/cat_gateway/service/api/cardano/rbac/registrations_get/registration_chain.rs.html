18/// A chain of valid RBAC registrations.
19///
20/// A unified data of multiple RBAC registrations.
21#[derive(Object, Debug, Clone)]
22#[oai(example)]
23pub struct RbacRegistrationChain {
24    /// A Catalyst ID.
25    catalyst_id: CatalystId,
26    /// A list of stake addresses.
27    #[oai(skip_serializing_if_is_empty)]
28    stake_addresses: Cip19StakeAddressList,
29    /// An ID of the last persistent transaction.
30    #[oai(skip_serializing_if_is_none)]
31    last_persistent_txn: Option<TxnId>,
32    /// An ID of the last volatile transaction.
33    #[oai(skip_serializing_if_is_none)]
34    last_volatile_txn: Option<TxnId>,
35    /// A list of registration purposes.
36    #[oai(skip_serializing_if_is_empty)]
37    purpose: PurposeList,
38    /// A map of role number to role data.
39    // This map is never empty, so there is no need to add the `skip_serializing_if_is_empty`
40    // attribute.
41    roles: RoleMap,
42}
43
44impl Example for RbacRegistrationChain {
45    fn example() -> Self {
46        Self {
47            catalyst_id: CatalystId::example(),
48            stake_addresses: Cip19StakeAddressList::example(),
49            purpose: PurposeList::example(),
50            last_persistent_txn: Some(TxnId::example()),
51            last_volatile_txn: Some(TxnId::example()),
52            roles: RoleMap::example(),
53        }
54    }
55}
56
57impl RbacRegistrationChain {
58    /// Creates a new registration chain instance.
59    pub fn new(info: &ChainInfo) -> anyhow::Result<Self> {
60        let catalyst_id = info.chain.catalyst_id().clone().into();
61        let stake_addresses = info
62            .chain
63            .role_0_stake_addresses()
64            .into_iter()
65            .map(Into::into)
66            .collect::<Vec<_>>()
67            .into();
68        let last_persistent_txn = info.last_persistent_txn.map(Into::into);
69        let last_volatile_txn = info.last_volatile_txn.map(Into::into);
70        let purpose = info
71            .chain
72            .purpose()
73            .iter()
74            .copied()
75            .map(UUIDv4::from)
76            .collect::<Vec<_>>()
77            .into();
78        let roles = role_data(info)?.into();
79
80        Ok(Self {
81            catalyst_id,
82            stake_addresses,
83            last_persistent_txn,
84            last_volatile_txn,
85            purpose,
86            roles,
87        })
88    }
89}
90
91/// Gets and converts a role data from the given chain info.
92fn role_data(info: &ChainInfo) -> anyhow::Result<HashMap<u8, RbacRoleData>> {
93    info.chain
94        .role_data_history()
95        .iter()
96        .map(|(&number, data)| {
97            RbacRoleData::new(data, info.last_persistent_slot, &info.chain)
98                .map(|r| (number.into(), r))
99        })
100        .collect()
101}
