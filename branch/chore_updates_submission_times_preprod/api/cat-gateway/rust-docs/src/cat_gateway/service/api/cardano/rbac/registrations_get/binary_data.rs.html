19/// A title.
20const TITLE: &str = "Hex encoded binary data";
21/// A description.
22const DESCRIPTION: &str = "A hex encoded binary data.";
23/// An example.
24const EXAMPLE: &str = "0x56CDD154355E078A0990F9E633F9553F7D43A68B2FF9BEF78B9F5C71C808A7C8";
25/// A validation regex pattern.
26const PATTERN: &str = "^0x[A-Fa-f0-9]+$";
27/// A format.
28const FORMAT: &str = "hex";
29
30/// A schema.
31static SCHEMA: LazyLock<MetaSchema> = LazyLock::new(|| {
32    MetaSchema {
33        title: Some(TITLE.to_owned()),
34        description: Some(DESCRIPTION),
35        example: Some(Value::String(EXAMPLE.to_string())),
36        pattern: Some(PATTERN.to_string()),
37        ..MetaSchema::ANY
38    }
39});
40
41/// Validate the `HexEncodedBinaryData`.
42/// This part is done separately from the `PATTERN`
43fn is_valid(hash: &str) -> bool {
44    /// Regex to validate `HexEncodedBinaryData`
45    #[allow(clippy::unwrap_used)] // Safe because the Regex is constant
46    static RE: LazyLock<Regex> = LazyLock::new(|| Regex::new(PATTERN).unwrap());
47
48    if RE.is_match(hash) {
49        if let Some(h) = hash.strip_prefix("0x") {
50            return hex::decode(h).is_ok();
51        }
52    }
53    false
54}
55
56impl_string_types!(
57    /// A hex encoded binary data.
58    HexEncodedBinaryData,
59    "string",
60    FORMAT,
61    Some(SCHEMA.clone()),
62    is_valid
63);

65impl From<Vec<u8>> for HexEncodedBinaryData {
