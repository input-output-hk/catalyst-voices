1//! A role extended data.
2
3use std::{collections::HashMap, sync::LazyLock};
4
5use poem_openapi::{
6    registry::{MetaSchema, MetaSchemaRef},
7    types::{Example, ParseError, ParseFromJSON, ParseResult, ToJSON, Type},
8};
9
10/// A title.
11const TITLE: &str = "Role extended data";
12/// A description.
13const DESCRIPTION: &str = "An arbitrary extended data for the role";
14
15/// A schema.
16static SCHEMA: LazyLock<MetaSchema> = LazyLock::new(|| {
17    MetaSchema {
18        title: Some(TITLE.to_owned()),
19        description: Some(DESCRIPTION),
20        example: ExtendedData::example().to_json(),
21        ..MetaSchema::ANY
22    }
23});
24
25/// A role extended data.
26#[derive(Debug, Eq, PartialEq, Clone)]
27pub(crate) struct ExtendedData(HashMap<u8, Vec<u8>>);
28
29impl From<HashMap<u8, Vec<u8>>> for ExtendedData {
30    fn from(value: HashMap<u8, Vec<u8>>) -> Self {
31        Self(value)
32    }
33}
34
35impl Type for ExtendedData {
36    type RawElementValueType = Self;
37    type RawValueType = Self;
38
39    const IS_REQUIRED: bool = true;
40
41    fn name() -> std::borrow::Cow<'static, str> {
42        "ExtendedData".into()
43    }
44
45    fn schema_ref() -> MetaSchemaRef {
46        let schema_ref =
47            MetaSchemaRef::Inline(Box::new(MetaSchema::new_with_format("object", "map")));
48        schema_ref.merge(SCHEMA.clone())
49    }
50
51    fn as_raw_value(&self) -> Option<&Self::RawValueType> {
52        Some(self)
53    }
54
55    fn raw_element_iter<'a>(
56        &'a self,
57    ) -> Box<dyn Iterator<Item = &'a Self::RawElementValueType> + 'a> {
58        Box::new(self.as_raw_value().into_iter())
59    }
60
61    fn is_empty(&self) -> bool {
62        self.0.is_empty()
63    }
64}
65
66impl ToJSON for ExtendedData {
67    fn to_json(&self) -> Option<serde_json::Value> {
68        Some(serde_json::Value::Object(
69            self.0
70                .iter()
71                .map(|(key, val)| (key.to_string(), val.clone().into()))
72                .collect(),
73        ))
74    }
75}
76
77impl ParseFromJSON for ExtendedData {
78    fn parse_from_json(value: Option<serde_json::Value>) -> ParseResult<Self> {
79        HashMap::parse_from_json(value)
80            .map_err(ParseError::propagate)
81            .map(Self)
82    }
83}
84
85impl Example for ExtendedData {
86    fn example() -> Self {
87        Self([(10, vec![1, 2, 3])].into_iter().collect())
88    }
