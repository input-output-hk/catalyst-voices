15/// Title.
16const TITLE: &str = "Cardano Native Asset Name";
17/// Description.
18const DESCRIPTION: &str = r"The name given to the native asset when it was minted.
19If the name can be converted to UTF8, its string is represented directly.
20Otherwise it will be represented as escaped ascii.
21Any `\` present in the name will be replaced with `\\` in all cases.";
22
23/// Example.
24const EXAMPLE: &str = "My Cool\nAsset";
25/// Maximum asset name in bytes.
26const ASSET_NAME_MAX_BYTES: usize = 32;
27/// Minimum length.
28const MIN_LENGTH: usize = 0;
29/// Maximum length calculated from maximum length of escaped string * max number of bytes
30/// of asset name. <https://www.gnu.org/software/gawk/manual/html_node/Escape-Sequences.html>
31const MAX_LENGTH: usize = ASSET_NAME_MAX_BYTES * 4;
32/// Validation Regex Pattern
33/// Can be anything
34const PATTERN: &str = concatcp!(r"^[\S\s]{", "0,", MAX_LENGTH, "}$");
35
36/// Schema.
37static SCHEMA: LazyLock<MetaSchema> = LazyLock::new(|| {
38    MetaSchema {
39        title: Some(TITLE.to_owned()),
40        description: Some(DESCRIPTION),
41        example: Some(Value::String(EXAMPLE.to_string())),
42        max_length: Some(MAX_LENGTH),
43        min_length: Some(MIN_LENGTH),
44        pattern: Some(PATTERN.to_string()),
45        ..poem_openapi::registry::MetaSchema::ANY
46    }
47});
48
49/// Validate `AssetName` This part is done separately from the `PATTERN`
50fn is_valid(name: &str) -> bool {
51    /// Regex to validate `AssetName`
52    #[allow(clippy::unwrap_used)] // Safe because the Regex is constant.
53    static RE: LazyLock<Regex> = LazyLock::new(|| Regex::new(PATTERN).unwrap());
54    RE.is_match(name)
55}
56
57impl_string_types!(
58    AssetName,
59    "string",
60    "cardano:asset_name",
61    Some(SCHEMA.clone()),
62    is_valid
63);
64
65impl Example for AssetName {
66    fn example() -> Self {
67        Self(EXAMPLE.to_owned())
68    }
69}
70
71impl From<&Vec<u8>> for AssetName {
72    fn from(value: &Vec<u8>) -> Self {
73        match String::from_utf8(value.clone()) {
74            Ok(name) => {
75                // UTF8 - Yay
76                // Escape any `\` so its consistent with escaped ascii below.
77                let name = name.replace('\\', r"\\");
78                Self(name)
79            },
80            Err(_) => Self(value.escape_ascii().to_string()),
81        }
82    }
83}
84
85impl From<String> for AssetName {
86    fn from(value: String) -> Self {
87        Self(value)
88    }
89}

96    fn test_asset_name() {
97        let escape_octal = r"\nnn".repeat(32);
98        // Test Data
99        // <https://preprod.cardanoscan.io/tokens>
100        let valid = [
101            EXAMPLE,
102            "This_Is_A_Very_Long_String______",
103            &escape_octal,
104            "SPLASH",
105        ];
106        for v in valid {
107            assert!(AssetName::parse_from_parameter(v).is_ok());
108        }
109        let invalid = [
110            // Add additional char
111            &format!("{escape_octal}a"),
112        ];
113        for v in invalid {
114            assert!(AssetName::parse_from_parameter(v).is_err());
115        }
