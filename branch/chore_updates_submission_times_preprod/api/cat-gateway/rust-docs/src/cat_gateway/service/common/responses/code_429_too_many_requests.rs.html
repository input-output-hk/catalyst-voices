1//! Define `TooManyRequests` response type.
2
3use poem_openapi::{types::Example, Object};
4use uuid::Uuid;
5
6use crate::service::common;
7
8#[derive(Object)]
9#[oai(example)]
10/// The client has sent too many requests in a given amount of time.
11pub(crate) struct TooManyRequests {
12    /// Unique ID of this Server Error so that it can be located easily for debugging.
13    id: common::types::generic::error_uuid::ErrorUuid,
14    /// Error message.
15    // Will not contain sensitive information, internal details or backtraces.
16    msg: common::types::generic::error_msg::ErrorMessage,
17}
18
19impl TooManyRequests {
20    /// Create a new Server Error Response Payload.
21    pub(crate) fn new(msg: Option<String>) -> Self {
22        let msg = msg.unwrap_or(
23            "Too Many Requests. You have exceeded the rate limit for this endpoint.".to_string(),
24        );
25        let id = Uuid::new_v4();
26
27        Self {
28            id: id.into(),
29            msg: msg.into(),
30        }
31    }
32}
33
34impl Example for TooManyRequests {
35    /// Example for the Too Many Requests Payload.
36    fn example() -> Self {
37        Self::new(None)
38    }
39}
