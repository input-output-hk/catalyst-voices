21/// Title.
22const TITLE: &str = "Signed Document Type";
23/// Description.
24const DESCRIPTION: &str = "Document Type.  UUIDv4 Formatted 128bit value.";
25/// Example.
26pub(crate) const EXAMPLE: &str = "7808d2ba-d511-40af-84e8-c0d1625fdfdc";
27/// External Documentation URI
28const URI: &str =
29    "https://input-output-hk.github.io/catalyst-libs/architecture/08_concepts/signed_doc/spec/#type";
30/// Description of the URI
31const URI_DESCRIPTION: &str = "Specification";
32/// Length of the hex encoded string
33pub(crate) const ENCODED_LENGTH: usize = uuidv4::ENCODED_LENGTH;
34/// Validation Regex Pattern
35pub(crate) const PATTERN: &str = uuidv4::PATTERN;
36/// Format
37pub(crate) const FORMAT: &str = uuidv4::FORMAT;
38
39/// Schema
40static SCHEMA: LazyLock<MetaSchema> = LazyLock::new(|| {
41    MetaSchema {
42        title: Some(TITLE.to_owned()),
43        description: Some(DESCRIPTION),
44        example: Some(Value::String(EXAMPLE.to_string())),
45        max_length: Some(ENCODED_LENGTH),
46        min_length: Some(ENCODED_LENGTH),
47        pattern: Some(PATTERN.to_string()),
48        external_docs: Some(MetaExternalDocument {
49            url: URI.to_owned(),
50            description: Some(URI_DESCRIPTION.to_owned()),
51        }),
52
53        ..poem_openapi::registry::MetaSchema::ANY
54    }
55});
56
57/// Because ALL the constraints are defined above, we do not ever need to define them in
58/// the API. BUT we do need to make a validator.
59/// This helps enforce uniform validation.
60fn is_valid(uuid: &str) -> bool {
61    uuidv4::UUIDv4::try_from(uuid).is_ok()
62}
63
64impl_string_types!(
65    DocumentType,
66    "string",
67    FORMAT,
68    Some(SCHEMA.clone()),
69    is_valid
70);
71
72impl DocumentType {
73    /// Creates a new `DocumentType` instance without validation.
74    /// **NOTE** could produce an invalid instance, be sure that passing `String` is a
75    /// valid `DocumentType`
76    pub(crate) fn new_unchecked(uuid: String) -> Self {
77        Self(uuid)
78    }
79}
80
81impl Example for DocumentType {

88impl TryFrom<&str> for DocumentType {
89    type Error = anyhow::Error;
90
91    fn try_from(value: &str) -> Result<Self, Self::Error> {
92        value.to_string().try_into()
93    }
94}
95
96impl TryFrom<String> for DocumentType {
97    type Error = anyhow::Error;
98
99    fn try_from(value: String) -> Result<Self, Self::Error> {
100        if !is_valid(&value) {
101            bail!("Invalid DocumentType '{value}', must be a valid UUIDv4")
102        }
103        Ok(Self(value))
104    }
105}
106
107impl From<uuidv4::UUIDv4> for DocumentType {
108    fn from(value: uuidv4::UUIDv4) -> Self {
109        Self(value.to_string())
110    }
