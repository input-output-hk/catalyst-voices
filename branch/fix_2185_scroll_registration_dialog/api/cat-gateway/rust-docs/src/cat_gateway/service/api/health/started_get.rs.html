1//! Implementation of the GET /health/started endpoint
2
3use std::sync::atomic::{AtomicBool, Ordering};
4
5use poem_openapi::ApiResponse;
6
7use crate::service::common::{
8    responses::WithErrorResponses, types::headers::retry_after::RetryAfterOption,
9};
10
11/// Flag to determine if the service has started
12static IS_STARTED: AtomicBool = AtomicBool::new(false);
13
14/// Set the started flag to `true`
15pub(crate) fn started() {
16    IS_STARTED.store(true, Ordering::Release);
17}
18/// Get the started flag
19fn is_started() -> bool {
20    IS_STARTED.load(Ordering::Acquire)
21}
22
23/// Endpoint responses.
24#[derive(ApiResponse)]
25pub(crate) enum Responses {
26    /// ## No Content
27    ///
28    /// Service is Started and can serve requests.
29    #[oai(status = 204)]
30    NoContent,
31}
32
33/// All responses.
34pub(crate) type AllResponses = WithErrorResponses<Responses>;
35
36/// # GET /health/started
37///
38/// Service Started endpoint.
39///
40/// Kubernetes (and others) use this endpoint to determine if the service has started
41/// properly and is able to serve requests.
42///
43/// In this service, started is guaranteed if this endpoint is reachable.
44/// So, it will always just return 204.
45///
46/// In theory it can also return 503 is the service has some startup processing
47/// to complete before it is ready to serve requests.
48///
49/// An example of not being started could be that bulk data needs to be read
50/// into memory or processed in some way before the API can return valid
51/// responses.  In that scenario this endpoint would return 503 until that
52/// startup processing was fully completed.
53#[allow(clippy::unused_async)]
54pub(crate) async fn endpoint() -> AllResponses {
55    if is_started() {
56        Responses::NoContent.into()
57    } else {
58        AllResponses::service_unavailable(
59            &anyhow::anyhow!("Service is not ready, do not send other requests."),
60            RetryAfterOption::Default,
61        )
62    }
63}
