33/// List of CIP36 Registration Data as found on-chain.
34#[derive(Object)]
35#[oai(example = true)]
36pub(crate) struct Cip36RegistrationList {
37    /// The Slot the Registrations are valid up until.
38    ///
39    /// Any registrations that occurred after this Slot are not included in the list.
40    /// Errors are reported only if they fall between the last valid registration and this
41    /// slot number.
42    /// Earlier errors are never reported.
43    pub slot: common::types::cardano::slot_no::SlotNo,
44    /// List of registrations associated with the query.
45    pub voting_key: Cip36RegistrationsForVotingPublicKeyList,
46    /// List of latest invalid registrations that were found, for the requested filter.
47    #[oai(skip_serializing_if_is_empty)]
48    pub invalid: common::types::cardano::registration_list::RegistrationCip36List,
49    /// Current Page
50    #[oai(skip_serializing_if_is_none)]
51    pub page: Option<Cip36RegistrationListPage>,
52}

120/// CIP36 Registration Data as found on-chain.
121#[derive(Object, Debug, Clone)]
122#[oai(example = true)]
123pub(crate) struct Cip36Details {
124    /// Blocks Slot Number that the registration certificate is in.
125    pub slot_no: common::types::cardano::slot_no::SlotNo,
126    /// Full Stake Address (not hashed, 32 byte ED25519 Public key).
127    #[oai(skip_serializing_if_is_none)]
128    pub stake_pub_key:
129        Option<common::types::generic::ed25519_public_key::Ed25519HexEncodedPublicKey>,
130    /// Voting Public Key (Ed25519 Public key).
131    #[oai(skip_serializing_if_is_none)]
132    pub vote_pub_key:
133        Option<common::types::generic::ed25519_public_key::Ed25519HexEncodedPublicKey>,
134    #[allow(clippy::missing_docs_in_private_items)] // Type is pre documented.
135    #[oai(skip_serializing_if_is_none)]
136    pub nonce: Option<common::types::cardano::nonce::Nonce>,
137    #[allow(clippy::missing_docs_in_private_items)] // Type is pre documented.
138    #[oai(skip_serializing_if_is_none)]
139    pub txn: Option<common::types::cardano::txn_index::TxnIndex>,
140    /// Cardano Cip-19 Formatted Shelley Payment Address.
141    #[oai(skip_serializing_if_is_none)]
142    pub payment_address: Option<common::types::cardano::cip19_shelley_address::Cip19ShelleyAddress>,
143    /// If the payment address is a script, then it can not be payed rewards.
144    #[oai(default)]
145    pub is_payable: common::types::cardano::boolean::IsPayable,
146    /// If this field is set, then the registration was in CIP15 format.
147    #[oai(default)]
148    pub cip15: common::types::cardano::boolean::IsCip15,
149    /// If there are errors with this registration, they are listed here.
150    /// This field is *NEVER* returned for a valid registration.
151    #[oai(skip_serializing_if_is_none)]
152    pub errors: Option<common::types::generic::error_msg::ErrorMessage>,
153}

181        let problem_report = ProblemReport::new("Cip36");

183        let errors = serde_json::to_string(&problem_report).unwrap_or_default();
184
185        Self {
186            slot_no: (common::types::cardano::slot_no::EXAMPLE + 135)
187                .try_into()
188                .unwrap_or_default(),
189            stake_pub_key: None,
190            vote_pub_key: Some(
191                common::types::generic::ed25519_public_key::Ed25519HexEncodedPublicKey::example(),
192            ),
193            nonce: Some((common::types::cardano::nonce::EXAMPLE + 97).into()),
194            txn: Some(common::types::cardano::txn_index::TxnIndex::example()),
195            payment_address: None,
196            is_payable: false.into(),
197            cip15: true.into(),
198            errors: Some(
199                crate::service::common::types::generic::error_msg::ErrorMessage::from(errors),
200            ),
201        }
202    }
