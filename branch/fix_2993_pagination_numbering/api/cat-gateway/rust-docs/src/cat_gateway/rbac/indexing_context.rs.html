1//! A RBAC context used during indexing.
2
3use std::collections::HashMap;
4
5use cardano_blockchain_types::{StakeAddress, TransactionId};
6use catalyst_types::catalyst_id::CatalystId;
7
8/// A RBAC context used during indexing.
9///
10/// During indexing the information is written to the database only after processing the
11/// whole block. If there are multiple RBAC registrations in one block, then when
12/// validating subsequent transactions we wouldn't be able to find information about the
13/// previous ones in the database. This context is used to hold such information during
14/// block processing in order to mitigate that issue.
15pub struct RbacIndexingContext {
16    /// A map containing pending data that will be written in the `catalyst_id_for_txn_id`
17    /// table.
18    transactions: HashMap<TransactionId, CatalystId>,
19    /// A map contains pending data that will be written in the
20    /// `catalyst_id_for_stake_address` table.
21    addresses: HashMap<StakeAddress, CatalystId>,
22}
23
24impl RbacIndexingContext {
25    /// Creates a new context issue.
26    pub fn new() -> Self {
27        let transactions = HashMap::new();
28        let addresses = HashMap::new();
29
30        Self {
31            transactions,
32            addresses,
33        }
34    }
35
36    /// Adds a transaction to the context.
37    pub fn insert_transaction(&mut self, transaction: TransactionId, catalyst_id: CatalystId) {
38        self.transactions.insert(transaction, catalyst_id);
39    }
40
41    /// Returns a Catalyst ID of the given transaction ID if it exists in the context.
42    pub fn find_transaction(&self, transaction_id: &TransactionId) -> Option<&CatalystId> {
43        self.transactions.get(transaction_id)
44    }
45
46    /// Adds a stake address to the context.
47    pub fn insert_address(&mut self, address: StakeAddress, catalyst_id: CatalystId) {
48        self.addresses.insert(address, catalyst_id);
49    }
50
51    /// Returns a Catalyst ID corresponding the the given stake address.
52    pub fn find_address(&self, address: &StakeAddress) -> Option<&CatalystId> {
53        self.addresses.get(address)
54    }
55}
