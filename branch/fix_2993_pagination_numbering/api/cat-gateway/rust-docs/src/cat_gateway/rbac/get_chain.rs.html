22/// Returns the latest (including the volatile part) registration chain by the given
23/// Catalyst ID.
24pub async fn latest_rbac_chain(id: &CatalystId) -> Result<Option<ChainInfo>> {
25    let volatile_session =
26        CassandraSession::get(false).context("Failed to get volatile Cassandra session")?;
27    let (chain, volatile_regs) = try_join(
28        persistent_rbac_chain(id),
29        indexed_regs(&volatile_session, id),
30    )
31    .await?;
32
33    let mut last_persistent_txn = None;
34    let mut last_persistent_slot = 0.into();
35
36    let chain = match chain {
37        Some(c) => {
38            last_persistent_txn = Some(c.current_tx_id_hash());
39            last_persistent_slot = c.current_point().slot_or_default();
40            Some(apply_regs(c, volatile_regs).await?)
41        },
42        None => build_rbac_chain(volatile_regs).await?,
43    };
44
45    Ok(chain.map(|chain| {
46        let last_txn = Some(chain.current_tx_id_hash());
47        let last_volatile_txn = if last_persistent_txn == last_txn {
48            None
49        } else {
50            last_txn
51        };
52
53        ChainInfo {
54            chain,
55            last_persistent_txn,
56            last_volatile_txn,
57            last_persistent_slot,
58        }
59    }))
60}

62/// Returns the latest (including the volatile part) registration chain by the given stake
63/// address.
64pub async fn latest_rbac_chain_by_address(address: &StakeAddress) -> Result<Option<ChainInfo>> {
65    let persistent_session =
66        CassandraSession::get(true).context("Failed to get persistent Cassandra session")?;
67    let volatile_session =
68        CassandraSession::get(false).context("Failed to get volatile Cassandra session")?;
69
70    let id = match CatalystIdQuery::latest(&volatile_session, address).await? {
71        Some(id) => id.catalyst_id,
72        None => {
73            match CatalystIdQuery::latest(&persistent_session, address).await? {
74                Some(id) => id.catalyst_id,
75                None => return Ok(None),
76            }
77        },
78    };
79
80    latest_rbac_chain(&id).await
81}

83/// Returns only the persistent part of a registration chain by the given Catalyst ID.
84pub async fn persistent_rbac_chain(id: &CatalystId) -> Result<Option<RegistrationChain>> {
85    if let Some(chain) = cached_persistent_rbac_chain(id) {
86        return Ok(Some(chain));
87    }
88
89    let session = CassandraSession::get(true).context("Failed to get Cassandra session")?;
90
91    let regs = indexed_regs(&session, id).await?;
92    let chain = build_rbac_chain(regs).await?.inspect(|c| {
93        cache_persistent_rbac_chain(id.clone(), c.clone());
94    });
95    Ok(chain)
96}

98/// Queries indexed RBAC registrations from the database.
99async fn indexed_regs(session: &CassandraSession, id: &CatalystId) -> Result<Vec<RbacQuery>> {
100    RbacQuery::execute(session, RbacQueryParams {
101        catalyst_id: id.clone().into(),
102    })
103    .and_then(|r| r.try_collect().map_err(Into::into))
104    .await
105}

107/// Builds a chain from the given registrations.
108async fn build_rbac_chain(
109    regs: impl IntoIterator<Item = RbacQuery>,
110) -> Result<Option<RegistrationChain>> {
111    let mut regs = regs.into_iter();
112    let Some(root) = regs.next() else {
113        return Ok(None);
114    };
115    if !root.removed_stake_addresses.is_empty() {
116        // This set contains addresses that were removed from the chain. It is impossible to
117        // remove an address before the chain was even started.
118        bail!("The root registration shouldn't contain removed stake addresses");
119    }
120    let root = cip509(
121        Settings::cardano_network(),
122        root.slot_no.into(),
123        root.txn_index.into(),
124    )
125    .await?;
126
127    let chain = RegistrationChain::new(root)?;
128    let chain = apply_regs(chain, regs).await?;
129    Ok(Some(chain))
130}

132/// Applies the given registration to the given chain.
133async fn apply_regs(
134    mut chain: RegistrationChain, regs: impl IntoIterator<Item = RbacQuery>,
135) -> Result<RegistrationChain> {
136    let network = Settings::cardano_network();
137
138    for reg in regs {
139        if !reg.removed_stake_addresses.is_empty() {
140            // TODO: This should be handled as a part of the
141            // https://github.com/input-output-hk/catalyst-voices/issues/2599 task.
142            continue;
143        }
144        let reg = cip509(network, reg.slot_no.into(), reg.txn_index.into()).await?;
145        chain = chain.update(reg)?;
146    }
147
148    Ok(chain)
149}

151/// Loads and parses a `Cip509` registration from a block using chain follower.
152async fn cip509(network: Network, slot: Slot, txn_index: TxnIndex) -> Result<Cip509> {
153    let point = Point::fuzzy(slot);
154    let block = ChainFollower::get_block(network, point)
155        .await
156        .context("Unable to get block")?
157        .data;
158    if block.point().slot_or_default() != slot {
159        // The `ChainFollower::get_block` function can return the next consecutive block if it
160        // cannot find the exact one. This shouldn't happen, but we need to check anyway.
161        anyhow::bail!(
162            "Unable to find exact {slot:?} block. Found block slot {:?}",
163            block.point().slot_or_default()
164        );
165    }
166    // We perform validation during indexing, so this normally should never fail.
167    Cip509::new(&block, txn_index, &[])
168        .with_context(|| {
169            format!("Invalid RBAC registration, slot = {slot:?}, transaction index = {txn_index:?}")
170        })?
171        .with_context(|| {
172            format!("No RBAC registration, slot = {slot:?}, transaction index = {txn_index:?}")
173        })
174}
