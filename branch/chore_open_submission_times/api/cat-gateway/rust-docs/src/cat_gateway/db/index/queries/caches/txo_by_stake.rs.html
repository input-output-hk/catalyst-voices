1//! Cache for TXO Assets by Stake Address Queries
2use std::sync::{Arc, LazyLock};
3
4use moka::{ops::compute::Op, policy::EvictionPolicy, sync::Cache};
5
6use crate::{
7    db::{
8        index::queries::staked_ada::{
9            get_txo_by_stake_address::{GetTxoByStakeAddressQuery, GetTxoByStakeAddressQueryKey},
10            update_txo_spent::UpdateTxoSpentQueryParams,
11        },
12        types::DbStakeAddress,
13    },
14    metrics::caches::txo_assets::{txo_assets_hits_inc, txo_assets_misses_inc},
15    settings::Settings,
16};
17
18/// In memory cache of the most recent Cardano TXO assets by Stake Address.
19static ASSETS_CACHE: LazyLock<Cache<DbStakeAddress, Arc<Vec<GetTxoByStakeAddressQuery>>>> =
20    LazyLock::new(|| {
21        Cache::builder()
22            .name("Cardano TXO Assets")
23            .eviction_policy(EvictionPolicy::lru())
24            .max_capacity(Settings::cardano_assets_cache().utxo_cache_size())
25            .build()
26    });
27
28/// Get TXO Assets entry from Cache.
29pub(crate) fn get(stake_address: &DbStakeAddress) -> Option<Arc<Vec<GetTxoByStakeAddressQuery>>> {
30    ASSETS_CACHE
31        .get(stake_address)
32        .inspect(|_| txo_assets_hits_inc())
33        .or_else(|| {
34            txo_assets_misses_inc();
35            None
36        })
37}
38
39/// Insert TXO Assets entry in Cache.
40pub(crate) fn insert(stake_address: DbStakeAddress, rows: Arc<Vec<GetTxoByStakeAddressQuery>>) {
41    ASSETS_CACHE.insert(stake_address, rows);
42}
43
44/// Update spent TXO Assets in Cache.
45pub(crate) fn update(params: Vec<UpdateTxoSpentQueryParams>) {
46    for txo_update in params {
47        let stake_address = &txo_update.stake_address;
48        let update_key = &GetTxoByStakeAddressQueryKey {
49            txn_index: txo_update.txn_index,
50            txo: txo_update.txo,
51            slot_no: txo_update.slot_no,
52        };
53        let _entry = ASSETS_CACHE
54            .entry(stake_address.clone())
55            .and_compute_with(|maybe_entry| {
56                maybe_entry.map_or_else(
57                    || {
58                        tracing::debug!(
59                        ?txo_update,
60                        %stake_address,
61                        "Stake Address not found in TXO Assets Cache, cannot update.");
62                    },
63                    |entry| {
64                        if let Some(txo) = entry
65                            .into_value()
66                            .iter()
67                            .find(|t| t.key.as_ref() == update_key)
68                        {
69                            let mut value = txo.value.write().unwrap_or_else(|error| {
70                                tracing::error!(
71                                ?txo_update,
72                                %stake_address,
73                                %error,
74                                cache_name = ?ASSETS_CACHE.name(),
75                                "RwLock for cache entry is poisoned, recovering.");
76                                txo.value.clear_poison();
77                                error.into_inner()
78                            });
79                            value.spent_slot = Some(txo_update.spent_slot);
80                            tracing::debug!(
81                            ?txo_update,
82                            %stake_address,
83                            "Updated TXO for Stake Address in Assets Cache");
84                        } else {
85                            tracing::debug!(
86                            ?txo_update,
87                            %stake_address,
88                            "TXOs not found for Stake Address in Assets Cache");
89                        }
90                    },
91                );
92                Op::Nop
93            });
94    }
95}
96
97/// Empty the TXO Assets cache.
98pub(crate) fn drop() {
99    ASSETS_CACHE.invalidate_all();
100}
101
102/// Size of TXO Assets cache.
103pub(crate) fn size() -> u64 {
104    ASSETS_CACHE.run_pending_tasks();
105    ASSETS_CACHE.weighted_size()
106}
107/// Number of entries in TXO Assets cache.
108pub(crate) fn entry_count() -> u64 {
109    ASSETS_CACHE.run_pending_tasks();
110    ASSETS_CACHE.entry_count()
111}
