18/// Validates a new registration by either starting a new chain or adding it to the
19/// existing one.
20///
21/// In case of failure a problem report from the given registration is updated and
22/// returned.
23pub async fn validate_rbac_registration(
24    reg: Cip509, is_persistent: bool, context: &mut RbacIndexingContext,
25) -> RbacValidationResult {
26    match reg.previous_transaction() {
27        Some(previous_txn) => update_chain(reg, previous_txn, is_persistent, context).await,
28        None => start_new_chain(reg, is_persistent).await,
29    }
30}

244/// Returns a Catalyst ID corresponding to the given transaction hash.
245async fn catalyst_id_from_txn_id(
246    txn_id: TransactionId, is_persistent: bool, context: &mut RbacIndexingContext,
247) -> Result<Option<CatalystId>> {
248    use crate::db::index::queries::rbac::get_catalyst_id_from_transaction_id::Query;
249
250    // Check the context first.
251    if let Some(catalyst_id) = context.find_transaction(&txn_id) {
252        return Ok(Some(catalyst_id.to_owned()));
253    }
254
255    // Then try to find in the persistent database.
256    let session =
257        CassandraSession::get(true).context("Failed to get Cassandra persistent session")?;
258    if let Some(r) = Query::get(&session, txn_id).await? {
259        return Ok(Some(r.catalyst_id));
260    };
261
262    // Conditionally check the volatile database.
263    if !is_persistent {
264        let session =
265            CassandraSession::get(false).context("Failed to get Cassandra volatile session")?;
266        return Query::get(&session, txn_id)
267            .map_ok(|r| r.map(|r| r.catalyst_id))
268            .await;
269    }
270
271    Ok(None)
272}

274/// Returns either persistent or "latest" (persistent + volatile) registration chain for
275/// the given Catalyst ID.
276async fn chain(
277    id: &CatalystId, is_persistent: bool, context: &mut RbacIndexingContext,
278) -> Result<Option<RegistrationChain>> {
279    if is_persistent {
280        persistent_rbac_chain(id).await
281    } else {
282        Ok(latest_rbac_chain(id).await?.map(|i| i.chain))
283    }
284
285    // TODO: FIXME: Use context to potentially append registrations!
286}

288/// Returns a Catalyst ID corresponding to the given stake address.
289async fn catalyst_id_from_stake_address(
290    address: &StakeAddress, is_persistent: bool, context: &mut RbacIndexingContext,
291) -> Result<Option<CatalystId>> {
292    use crate::db::index::queries::rbac::get_catalyst_id_from_stake_address::Query;
293
294    // Check the context first.
295    if let Some(catalyst_id) = context.find_address(address) {
296        return Ok(Some(catalyst_id.to_owned()));
297    }
298
299    // Then try to find in the persistent database.
300    let session =
301        CassandraSession::get(true).context("Failed to get Cassandra persistent session")?;
302    if let Some(r) = Query::latest(&session, address).await? {
303        return Ok(Some(r.catalyst_id));
304    };
305
306    // Conditionally check the volatile database.
307    if !is_persistent {
308        let session =
309            CassandraSession::get(false).context("Failed to get Cassandra volatile session")?;
310        return Query::latest(&session, address)
311            .map_ok(|r| r.map(|r| r.catalyst_id))
312            .await;
313    }
314
315    Ok(None)
