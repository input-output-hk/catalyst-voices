13/// Get document from the database
14pub(crate) async fn get_document(
15    document_id: &uuid::Uuid, version: Option<&uuid::Uuid>,
16) -> anyhow::Result<CatalystSignedDocument> {
17    // Find the doc in the static templates first
18    if let Some(doc) = get_doc_static_template(document_id) {
19        return Ok(doc);
20    }
21
22    // If doesn't exist in the static templates, try to find it in the database
23    let db_doc = FullSignedDoc::retrieve(document_id, version).await?;
24    db_doc.raw().try_into()
25}

27/// A struct which implements a
28/// `catalyst_signed_doc::providers::CatalystSignedDocumentProvider` trait
29pub(crate) struct DocProvider;
30
31impl catalyst_signed_doc::providers::CatalystSignedDocumentProvider for DocProvider {
32    async fn try_get_doc(
33        &self, doc_ref: &catalyst_signed_doc::DocumentRef,
34    ) -> anyhow::Result<Option<CatalystSignedDocument>> {
35        let id = doc_ref.id.uuid();
36        let ver = doc_ref.ver.map(|uuid| uuid.uuid());
37        match get_document(&id, ver.as_ref()).await {
38            Ok(doc) => Ok(Some(doc)),
39            Err(err) if err.is::<NotFoundError>() => Ok(None),
40            Err(err) => Err(err),
41        }
42    }
43}

66impl VerifyingKeyProvider {
67    /// Attempts to construct an instance of `Self` by validating and resolving a list of
68    /// Catalyst Document KIDs against a provided RBAC token.
69    ///
70    /// This method performs the following steps:
71    /// 1. Verifies that **all** provided `kids` match the Catalyst ID from the RBAC
72    ///    token.
73    /// 2. Extracts the role index from each KID and attempts to build a registration
74    ///    chain from the indexed registration data.
75    /// 3. Retrieves the latest signing public key and rotation state associated with the
76    ///    role for each KID from the registration chain.
77    /// 4. Collects and returns a vector of tuples containing the KID, signing key, and
78    ///    rotation.
79    ///
80    /// # Errors
81    ///
82    /// Returns an `anyhow::Error` if:
83    /// - Any KID's short Catalyst ID does not match the one in the token.
84    /// - The role index parsing fails.
85    /// - Indexed registration queries or chain building fail.
86    /// - The latest signing key for a required role cannot be found.
87    pub(crate) fn try_from_kids(
88        token: &CatalystRBACTokenV1, kids: &[catalyst_signed_doc::IdUri],
89    ) -> anyhow::Result<Self> {
90        // validate rbac token and document KIDs (ignoring the role/rotation)
91        if kids
92            .iter()
93            .any(|kid| kid.as_short_id() != token.catalyst_id().as_short_id())
94        {
95            anyhow::bail!("RBAC Token CatID does not match with the document KIDs");
96        }
97
98        let mut result = Vec::with_capacity(kids.len());
99        for kid in kids {
100            let (role_index, _) = kid.role_and_rotation();
101            let role_index = RoleNumber::from(role_index.to_string().parse::<u8>()?);
102
103            let Some(reg_chain) = token.reg_chain() else {
104                anyhow::bail!("Failed to retrieve a registration chain for {kid} Catalyst ID")
105            };
106
107            let (latest_pk, rotation) = reg_chain
108                .get_latest_signing_pk_for_role(&role_index)
109                .ok_or_else(|| {
110                    anyhow::anyhow!(
111                        "Failed to get last signing key for the proposer role for {kid} Catalyst ID"
112                    )
113                })?;
114
115            result.push((kid.clone(), latest_pk, rotation));
116        }
117
118        Ok(Self(result))
119    }
