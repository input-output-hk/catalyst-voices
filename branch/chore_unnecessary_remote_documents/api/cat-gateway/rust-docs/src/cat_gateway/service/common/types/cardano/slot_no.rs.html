14/// Title.
15const TITLE: &str = "Cardano Blockchain Slot Number";
16/// Description.
17const DESCRIPTION: &str = "The Slot Number of a Cardano Block on the chain.";
18/// Example.
19pub(crate) const EXAMPLE: u64 = 1_234_567;
20/// Minimum.
21const MINIMUM: u64 = 0;
22/// Maximum.
23const MAXIMUM: u64 = u64::MAX / 2;
24
25/// Schema.
26#[allow(clippy::cast_precision_loss)]
27static SCHEMA: LazyLock<MetaSchema> = LazyLock::new(|| {
28    MetaSchema {
29        title: Some(TITLE.to_owned()),
30        description: Some(DESCRIPTION),
31        example: Some(EXAMPLE.into()),
32        maximum: Some(MAXIMUM as f64),
33        minimum: Some(MINIMUM as f64),
34        ..MetaSchema::ANY
35    }
36});
37
38/// Slot number
39#[derive(Debug, Eq, PartialEq, Hash, Clone, Copy, PartialOrd, Ord)]
40pub(crate) struct SlotNo(u64);
41
42impl SlotNo {
43    /// Is the Slot Number valid?
44    fn is_valid(value: u64) -> bool {
45        (MINIMUM..=MAXIMUM).contains(&value)
46    }
47
48    /// Generic conversion of `Option<T>` to `Option<SlotNo>`.
49    pub(crate) fn into_option<T: Into<SlotNo>>(value: Option<T>) -> Option<SlotNo> {
50        value.map(std::convert::Into::into)
51    }
52}
