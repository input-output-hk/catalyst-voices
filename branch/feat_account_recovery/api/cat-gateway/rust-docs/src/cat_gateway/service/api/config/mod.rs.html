28/// Get configuration endpoint responses.
29#[derive(ApiResponse)]
30enum GetConfigResponses {
31    /// Configuration result.
32    #[oai(status = 200)]
33    Ok(Json<FrontendConfig>),
34}
35/// Get configuration all responses.
36type GetConfigAllResponses = WithErrorResponses<GetConfigResponses>;
37
38/// Get configuration schema endpoint responses.
39#[derive(ApiResponse)]
40enum GetConfigSchemaResponses {
41    /// Configuration result.
42    #[oai(status = 200)]
43    Ok(Json<FrontendConfig>),
44}
45/// Get configuration schema all responses.
46type GetConfigSchemaAllResponses = WithErrorResponses<GetConfigSchemaResponses>;
47
48/// Set configuration endpoint responses.
49#[derive(ApiResponse)]
50enum SetConfigResponse {
51    /// Configuration Update Successful.
52    #[oai(status = 204)]
53    Ok,
54    /// Set configuration unprocessable content.
55    #[oai(status = 422)]
56    UnprocessableContent(Json<ConfigUnprocessableContent>),
57}
58/// Set configuration all responses.
59type SetConfigAllResponses = WithErrorResponses<SetConfigResponse>;

74#[OpenApi(tag = "ApiTags::Config")]
75impl ConfigApi {
76    /// Get the configuration for the frontend.
77    ///
78    /// Get the frontend configuration for the requesting client.
79    ///
80    /// ### Security
81    ///
82    /// Does not require any Catalyst RBAC Token to access.
83    #[oai(
84        path = "/draft/config/frontend",
85        method = "get",
86        operation_id = "get_config_frontend"
87    )]
88    async fn get_frontend(&self, ip_address: RealIp, _auth: NoneOrRBAC) -> GetConfigAllResponses {
89        // Fetch the general configuration
90        let general_config = Config::get(ConfigKey::Frontend).await;
91
92        // Attempt to fetch the IP configuration
93        let ip_config = if let Some(ip) = ip_address.0 {
94            match Config::get(ConfigKey::FrontendForIp(ip)).await {
95                Ok(value) => Some(value),
96                Err(err) => {
97                    error!(id="get_frontend_config_ip", error=?err, "Failed to get frontend configuration for IP");
98                    return GetConfigAllResponses::handle_error(&err);
99                },
100            }
101        } else {
102            None
103        };
104
105        // Handle the response
106        match general_config {
107            Ok(general) => {
108                // If there is IP specific config, replace any key in the general config with
109                // the keys from the IP specific config
110                let response_config = if let Some(ip_specific) = ip_config {
111                    merge_configs(&general, &ip_specific)
112                } else {
113                    general
114                };
115
116                // Convert the merged Value to FrontendConfig
117                let frontend_config: FrontendConfig =
118                    FrontendConfig::parse_from_json(Some(response_config)).unwrap_or_default(); // Handle error as needed
119
120                GetConfigResponses::Ok(Json(frontend_config)).into()
121            },
122            Err(err) => {
123                error!(id="get_frontend_config_general", error=?err, "Failed to get general frontend configuration");
124                GetConfigAllResponses::handle_error(&err)
125            },
126        }
127    }
128
129    /// Get the frontend configuration JSON schema.
130    ///
131    /// Returns the JSON schema which defines the data which can be read or written for
132    /// the frontend configuration.
133    ///
134    /// ### Security
135    ///
136    /// Does not require any Catalyst RBAC Token to access.
137    #[oai(
138        path = "/draft/config/frontend/schema",
139        method = "get",
140        operation_id = "get_config_frontend_schema"
141    )]
142    #[allow(clippy::unused_async)]
143    async fn get_frontend_schema(&self, _auth: NoneOrRBAC) -> GetConfigSchemaAllResponses {
144        // Schema for both IP specific and general are identical
145        let schema_value = ConfigKey::Frontend.schema().clone();
146        let frontend_config =
147            FrontendConfig::parse_from_json(Some(schema_value)).unwrap_or_default();
148        GetConfigSchemaResponses::Ok(Json(frontend_config)).into()
149    }
150
151    /// Set the frontend configuration.
152    ///
153    /// Store the given config as either global front end configuration, or configuration
154    /// for a client at a specific IP address.
155    ///
156    /// ### Security
157    ///
158    /// Requires Admin Authoritative RBAC Token.
159    #[oai(
160        path = "/draft/config/frontend",
161        method = "put",
162        operation_id = "put_config_frontend",
163        hidden = true
164    )]
165    async fn put_frontend(
166        &self,
167        /// *OPTIONAL* The IP Address to set the configuration for.
168        #[oai(name = "IP")]
169        ip_query: Query<Option<IpAddr>>,
170        body: Json<FrontendConfig>, _auth: InternalApiKeyAuthorization,
171    ) -> SetConfigAllResponses {
172        let body_value = body.0.to_json();
173
174        match body_value {
175            Some(value) => {
176                match ip_query.0 {
177                    Some(ip) => set(ConfigKey::FrontendForIp(ip.0), value).await,
178                    None => set(ConfigKey::Frontend, value).await,
179                }
180            },
181            None => {
182                SetConfigResponse::UnprocessableContent(Json(ConfigUnprocessableContent::new(
183                    "Invalid JSON data".to_string(),
184                    None,
185                )))
186                .into()
187            },
188        }
189    }
190}
191
192/// Helper function to merge two JSON values.
193fn merge_configs(general: &Value, ip_specific: &Value) -> Value {
194    let mut merged = general.clone();
195
196    if let Some(ip_specific_obj) = ip_specific.as_object() {
197        if let Some(merged_obj) = merged.as_object_mut() {
198            for (key, value) in ip_specific_obj {
199                if let Some(existing_value) = merged_obj.get_mut(key) {
200                    *existing_value = value.clone();
201                } else {
202                    merged_obj.insert(key.clone(), value.clone());
203                }
204            }
205        }
206    }
207
208    merged
209}

211/// Helper function to handle set.
212async fn set(key: ConfigKey, value: Value) -> SetConfigAllResponses {
213    match Config::set(key, value).await {
214        Ok(validate) => {
215            match validate {
216                BasicOutput::Valid(_) => SetConfigResponse::Ok.into(),
217                BasicOutput::Invalid(errors) => {
218                    let schema_errors: Vec<ErrorMessage> = errors
219                        .iter()
220                        .map(|error| error.error_description().clone().into_inner().into())
221                        .collect();
222                    SetConfigResponse::UnprocessableContent(Json(ConfigUnprocessableContent::new(
223                        "Invalid JSON data validating against JSON schema".to_string(),
224                        Some(schema_errors.into()),
225                    )))
226                    .into()
227                },
228            }
229        },
230        Err(err) => {
231            error!(id="set_config_frontend", error=?err, "Failed to set frontend configuration");
232            SetConfigAllResponses::handle_error(&err)
233        },
234    }
235}
