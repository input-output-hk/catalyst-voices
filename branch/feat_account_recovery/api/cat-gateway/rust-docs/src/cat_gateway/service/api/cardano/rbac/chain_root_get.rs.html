3use poem_openapi::{payload::Json, types::Example, ApiResponse, Object};

17/// GET RBAC chain root response.
18#[derive(Object)]
19#[oai(example = true)]
20pub(crate) struct ChainRootGetResponse {
21    /// RBAC certificate chain root.
22    chain_root: Hash256,
23}
24
25/// Endpoint responses.
26#[allow(dead_code)]
27#[derive(ApiResponse)]
28pub(crate) enum Responses {
29    /// ## Ok
30    ///
31    /// Success returns the chain root hash.
32    #[oai(status = 200)]
33    Ok(Json<ChainRootGetResponse>),
34    /// ## Not Found
35    ///
36    /// No chain root found for the given stake address.
37    #[oai(status = 404)]
38    NotFound,
39}
40
41pub(crate) type AllResponses = WithErrorResponses<Responses>;
42
43/// Get chain root endpoint.
44#[allow(clippy::unused_async)]
45pub(crate) async fn endpoint(_stake_address: Cip19StakeAddress) -> AllResponses {
46    let Some(_session) = CassandraSession::get(true) else {
47        error!("Failed to acquire db session");
48        let err = anyhow::anyhow!("Failed to acquire db session");
49        return AllResponses::service_unavailable(&err, RetryAfterOption::Default);
50    };
51
52    // TODO: This endpoint needs to be removed or updated because "chain root" was replaced by
53    // Catalyst ID.
54    AllResponses::forbidden(None)
55}
